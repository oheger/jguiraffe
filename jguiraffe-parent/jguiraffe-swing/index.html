JGUIraffe Swing Integration

This module implements the builder interfaces defined by the Core project using the Swing platform.

In Maven projects this module can be added using the following dependencies: 

  ...
  <!-- Import JGUIraffe BOM -->
  <dependencyManagement>
    <dependencies>
      <dependency>
        <groupId>net.sf.jguiraffe</groupId>
        <artifactId>jguiraffe-bom</artifactId>
        <version>1.3.1</version>
        <type>pom</type>
        <scope>import</scope>
      </dependency>
    </dependencies>
  </dependencyManagement>

  <dependencies>

    <dependency>
      <groupId>net.sf.jguiraffe</groupId>
      <artifactId>jguiraffe-swing</artifactId>
    </dependency>
