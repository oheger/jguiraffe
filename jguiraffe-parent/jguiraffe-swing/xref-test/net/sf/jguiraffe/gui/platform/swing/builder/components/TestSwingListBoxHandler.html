28  /**
29   * Test class for SwingListBoxHandler.
30   *
31   * @author Oliver Heger
32   * @version $Id: TestSwingListBoxHandler.java 205 2012-01-29 18:29:57Z oheger $
33   */
34  public class TestSwingListBoxHandler extends
35          AbstractListBoxComponentHandlerTest
36  {
37      /** The handler to be tested. */
38      private SwingListBoxHandler handler;
39  
40      @Override
41      @Before
42      public void setUp() throws Exception
43      {
44          super.setUp();
45          handler = new SwingListBoxHandler(component, setUpListModel(), 0, 0);
46      }
47  
48      @Override
49      protected SwingListModelHandler getListModelHandler()
50      {
51          return handler;
52      }
53  
54      /**
55       * Tests fetching the handler's data when no list element is selected.
56       */
57      @Test
58      public void testGetDataUnselected()
59      {
60          assertNull("Data returned for no selection", handler.getData());
61      }
62  
63      /**
64       * Tests fetching the handler's data when an element is selected.
65       */
66      @Test
67      public void testGetDataSelected()
68      {
69          component.setSelectedIndex(3);
70          assertEquals("Wrong data", ListModelImpl.VALUE_PREFIX + "3", handler
71                  .getData());
72      }
73  
74      /**
75       * Tests setting valid data for the handler.
76       */
77      @Test
78      public void testSetDataValid()
79      {
80          handler.setData(ListModelImpl.VALUE_PREFIX + "4");
81          assertEquals("Wrong selected index", 4, component.getSelectedIndex());
82      }
83  
84      /**
85       * Tests setting data to null. This should clear the selection.
86       */
87      @Test
88      public void testSetDataNull()
89      {
90          component.setSelectedIndex(1);
91          handler.setData(null);
92          assertEquals("Selection not cleared for null data", -1, component
93                  .getSelectedIndex());
94      }
95  
96      /**
97       * Tests setting an invalid data object. This should clear the selection.
98       */
99      @Test
100     public void testSetDataInvalid()
101     {
102         component.setSelectedIndex(1);
103         handler.setData("No list element");
104         assertEquals("Selection not cleared for invalid data", -1, component
105                 .getSelectedIndex());
106     }
107 
108     /**
109      * Tests whether a scroll pane can be passed to the constructor.
110      */
111     @Test
112     public void testInitWithScrollPane()
113     {
114         JList list = new JList();
115         ListModel listModel = setUpListModel();
116         JScrollPane scr = new JScrollPane();
117         SwingListBoxHandler listHandler =
118                 new SwingListBoxHandler(list, listModel, scr);
119 
120         assertEquals("Wrong list", list, listHandler.getComponent());
121         assertEquals("Wrong list model", MODEL_SIZE, listModel.size());
122         assertEquals("Wrong scroll pane", scr, listHandler.getOuterComponent());
123     }
