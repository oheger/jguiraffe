102     /**
103      * Fully initializes a table tag for the test table based on the provided
104      * parameters.
105      *
106      * @param data the data collection; can be <b>null</b>, then default data
107      *        will be created
108      * @param width if set to <b>true</b>, the widths of the columns are set
109      * @param scrollWidth the scroll width as string (can be <b>null</b>)
110      * @param scrollHeight the scroll height as string (can be <b>null</b>)
111      * @return the initialized tag
112      * @throws JellyTagException if the tag is used incorrectly
113      * @throws FormBuilderException if an error occurs
114      */
115     public static TableTagTestImpl setUpTableTag(Collection<?> data,
116             boolean width, String scrollWidth, String scrollHeight)
117             throws JellyTagException, FormBuilderException
118     {
119         ComponentManager cm = EasyMock.createNiceMock(ComponentManager.class);
120         TransformerContext tctx = EasyMock.createNiceMock(TransformerContext.class);
121         EasyMock.replay(cm, tctx);
122         JellyContext ctx = new JellyContext();
123         ComponentBuilderData builderData = new ComponentBuilderData();
124         builderData.setComponentManager(cm);
125         builderData.initializeForm(tctx, new BeanBindingStrategy());
126         builderData.setBeanContext(new DefaultBeanContext(
127                 new JellyContextBeanStore(ctx, null)));
128         builderData.put(ctx);
129         ctx.setVariable(VAR_MODEL, (data != null) ? data : setUpTestData());
130         TableTagTestImpl tt = new TableTagTestImpl();
131         tt.setContext(ctx);
132         tt.setModel(VAR_MODEL);
133         tt.setScrollWidth(scrollWidth);
134         tt.setScrollHeight(scrollHeight);
135         processTableTag(tt);
136 
137         for (int i = 0; i < COLUMN_NAMES.length; i++)
138         {
139             // add a special column tag that can be directly executed
140             TableColumnTagTestImpl colTag = new TableColumnTagTestImpl(tt);
141             colTag.setContext(ctx);
142             colTag.setHeader(COLUMN_NAMES[i]);
143             colTag.setName(COLUMN_NAMES[i]);
144             colTag.setColumnClass(COLUMN_TYPES[i]);
145             colTag.setEditable(EDIT_FLAGS[i]);
146             if (i == IDX_RENDERER_COLUMN)
147             {
148                 colTag.installRenderer(new JLabel());
149             }
150             if (width)
151             {
152                 colTag.setWidth(COLUMN_WIDTHS[i] + Unit.DLU.getUnitName());
153             }
154             colTag.processBeforeBody();
155         }
156 
157         return tt;
158     }

193     /**
194      * Invokes the {@code processBeforeBody()} method on the passed in
195      * table tag. This method can be used if the tag in its processed form is
196      * needed. It assumes that the passed in tag has been created by the
197      * {@code setUpTableTag()} method.
198      *
199      * @param tt the table tag
200      * @throws FormBuilderException if an exception occurs
201      * @throws JellyTagException if the tag is used incorrectly
202      */
203     public static void processTableTag(TableTag tt) throws JellyTagException,
204             FormBuilderException
205     {
206         ComponentBuilderData builderData =
207                 ComponentBuilderData.get(tt.getContext());
208         builderData.setBeanContext(new DefaultBeanContext(
209                 new JellyContextBeanStore(tt.getContext(),
210                         createParentBeanStore())));
211         ((TableTagTestImpl) tt).processBeforeBody();
212     }

214     /**
215      * Creates a bean store which contains an application bean. This is needed
216      * when processing a table tag.
217      *
218      * @return the bean store
219      */
220     private static BeanStore createParentBeanStore()
221     {
222         DefaultBeanStore store = new DefaultBeanStore();
223         Application app = new Application();
224         app.setApplicationContext(new ApplicationContextImpl());
225         store.addBeanProvider(Application.BEAN_APPLICATION,
226                 ConstantBeanProvider.getInstance(Application.class, app));
227         return store;
228     }

230     /**
231      * Returns a collection with test data.
232      *
233      * @return the data collection
234      */
235     public static Collection<PersonBean> setUpTestData()
236     {
237         List<PersonBean> data = new LinkedList<PersonBean>();
238         for (Object[] td : TEST_DATA)
239         {
240             PersonBean bean = new PersonBean();
241             bean.setFirstName(td[0].toString());
242             bean.setLastName(td[1].toString());
243             bean.setIdNo(((Integer) td[2]).intValue());
244             bean.setSalary(((Double) td[3]).doubleValue());
245             data.add(bean);
246         }
247         return data;
248     }

250     /**
251      * Creates a table model. If a table tag has already been created, this tag
252      * is used for the new model. Otherwise a new tag is created and initialized
253      * now.
254      *
255      * @return the table model
256      */
257     protected SwingTableModel createTableModel()
258     {
259         try
260         {
261             if (tableTag == null)
262             {
263                 tableTag = setUpTableTag(null);
264             }
265             return new SwingTableModel(tableTag, new JTable());
266         }
267         catch (Exception ex)
268         {
269             throw new RuntimeException(ex);
270         }
271     }
