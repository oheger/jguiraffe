32  /**
33   * Test class for SwingSelectionHandler.
34   *
35   * @author Oliver Heger
36   * @version $Id: TestSwingSelectionHandler.java 205 2012-01-29 18:29:57Z oheger $
37   */
38  public class TestSwingSelectionHandler extends AbstractTableModelTest
39  {
40      /** Stores the handler to be tested. */
41      private SwingTableSelectionHandler handler;
42  
43      /** Stores the table tag. */
44      private TableTag tableTag;
45  
46      /** Stores the component. */
47      private JComponent component;
48  
49      @Before
50      public void setUp() throws Exception
51      {
52          tableTag = setUpTableTag(null);
53          handler = new SwingTableSelectionHandler();
54          component = createCellComponent();
55      }
56  
57      /**
58       * Creates the table needed for testing.
59       *
60       * @return the test table
61       */
62      private JTable createTable()
63      {
64          return new JTable();
65      }
66  
67      /**
68       * Creates the component that will be passed to the handler.
69       *
70       * @return the cell renderer or editor component
71       */
72      private JComponent createCellComponent()
73      {
74          return new JLabel();
75      }
76  
77      /**
78       * Test for the default status: not selected and not focused.
79       */
80      @Test
81      public void testPrepareComponentDefault()
82      {
83          JTable table = createTable();
84          handler
85                  .prepareComponent(table, tableTag, component, false, false, 0,
86                          0);
87          assertNull("Component has a border", component.getBorder());
88          assertEquals("Wrong background color", table.getBackground(), component
89                  .getBackground());
90          assertEquals("Wrong foreground color", table.getForeground(), component
91                  .getForeground());
92      }
93  
94      /**
95       * Tests the selected status.
96       */
97      @Test
98      public void testPrepareComponentSelected()
99      {
100         JTable table = createTable();
101         handler.prepareComponent(table, tableTag, component, true, false, 0, 0);
102         assertNull("Component has a border", component.getBorder());
103         assertEquals("Wrong background color", table.getSelectionBackground(),
104                 component.getBackground());
105         assertEquals("Wrong foreground color", table.getSelectionForeground(),
106                 component.getForeground());
107     }
108 
109     /**
110      * Tests the focused status.
111      */
112     @Test
113     public void testPrepareComponentFocused()
114     {
115         JTable table = createTable();
116         handler.prepareComponent(table, tableTag, component, false, true, 0, 0);
117         assertTrue("Component has no line border",
118                 component.getBorder() instanceof LineBorder);
119         assertEquals("Wrong background color", table.getBackground(), component
120                 .getBackground());
121         assertEquals("Wrong foreground color", table.getForeground(), component
122                 .getForeground());
123     }
124 
125     /**
126      * Tests the selected and focused status.
127      */
128     @Test
129     public void testPrepareComponentSelectedFocused()
130     {
131         JTable table = createTable();
132         handler.prepareComponent(table, tableTag, component, true, true, 0, 0);
133         assertTrue("Component has no line border",
134                 component.getBorder() instanceof LineBorder);
135         assertEquals("Wrong background color", table.getSelectionBackground(),
136                 component.getBackground());
137         assertEquals("Wrong foreground color", table.getSelectionForeground(),
138                 component.getForeground());
139     }
