16  package net.sf.jguiraffe.gui.platform.swing.builder.event;
17  
18  import java.awt.List;
19  import java.awt.event.ItemEvent;
20  
21  import net.sf.jguiraffe.gui.builder.event.FormChangeEvent;
22  import net.sf.jguiraffe.gui.builder.event.FormChangeListener;
23  import net.sf.jguiraffe.gui.builder.event.FormEventManager;
24  
25  import org.junit.Before;
26  import org.junit.Test;
27  
28  /**
29   * Test class for {@code ChangeEventAdapter}.
30   *
31   * @author Oliver Heger
32   * @version $Id: TestChangeEventAdapter.java 205 2012-01-29 18:29:57Z oheger $
33   */
34  public class TestChangeEventAdapter
35  {
36      /** Constant for the test change event. */
37      private static final ItemEvent CHANGE_EVENT = new ItemEvent(new List(), 42,
38              "Test", ItemEvent.SELECTED);
39  
40      /** The test event manager. */
41      private EventManagerImpl eventManager;
42  
43      @Before
44      public void setUp() throws Exception
45      {
46          eventManager = new EventManagerImpl();
47      }
48  
49      /**
50       * Tests whether change events can be passed to the event manager.
51       */
52      @Test
53      public void testComponentChangedEventManager()
54      {
55          ChangeEventAdapter adapter = new ChangeEventAdapter(eventManager,
56                  eventManager.getHandler(), EventManagerImpl.NAME);
57          adapter.componentChanged(CHANGE_EVENT);
58          eventManager.checkFormEvent(FormChangeEvent.class, CHANGE_EVENT);
59      }
60  
61      /**
62       * Tests whether change events can be passed to a specific event listener.
63       */
64      @Test
65      public void testComponentChangedEventListener()
66      {
67          FormChangeListenerTestImpl l = new FormChangeListenerTestImpl();
68          ChangeEventAdapter adapter = new ChangeEventAdapter(l, eventManager
69                  .getHandler(), EventManagerImpl.NAME);
70          adapter.componentChanged(CHANGE_EVENT);
71          eventManager.checkFormEvent(l.event, CHANGE_EVENT);
72      }
73  
74      /**
75       * Tries to create a change adapter without an event manager. This should
76       * cause an exception.
77       */
78      @Test(expected = IllegalArgumentException.class)
79      public void testInitNoEventManager()
80      {
81          new ChangeEventAdapter((FormEventManager) null, eventManager
82                  .getHandler(), EventManagerImpl.NAME);
83      }
84  
85      /**
86       * Tries to create an instance without a listener. This should cause an
87       * exception.
88       */
89      @Test(expected = IllegalArgumentException.class)
90      public void testInitNoListener()
91      {
92          new ChangeEventAdapter((FormChangeListener) null, eventManager
93                  .getHandler(), EventManagerImpl.NAME);
94      }
95  
96      /**
97       * A test listener implementation for testing whether change events are
98       * correctly sent.
99       */
100     private static class FormChangeListenerTestImpl implements
101             FormChangeListener
102     {
103         /** The received change event. */
104         FormChangeEvent event;
105 
106         /**
107          * Records this invocation.
108          */
109         public void elementChanged(FormChangeEvent e)
110         {
111             event = e;
112         }
113     }
