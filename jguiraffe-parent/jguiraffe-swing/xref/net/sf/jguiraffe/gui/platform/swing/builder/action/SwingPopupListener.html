16  package net.sf.jguiraffe.gui.platform.swing.builder.action;

18  import java.awt.event.MouseAdapter;
19  import java.awt.event.MouseEvent;
20  
21  import net.sf.jguiraffe.gui.builder.action.ActionBuilder;
22  import net.sf.jguiraffe.gui.builder.action.ActionManager;
23  import net.sf.jguiraffe.gui.builder.action.FormActionException;
24  import net.sf.jguiraffe.gui.builder.action.PopupMenuBuilder;
25  import net.sf.jguiraffe.gui.builder.action.PopupMenuHandler;
26  import net.sf.jguiraffe.gui.builder.components.ComponentBuilderData;

29  /**
30   * <p>
31   * A specialized mouse listener for bringing up a popup menu for a GUI
32   * component.
33   * </p>
34   * <p>
35   * An instance of this class is used by the
36   * {@code registerPopupMenuHandler()} method of
37   * {@link SwingActionManager}. It is initialized with the
38   * {@link PopupMenuHandler} object responsible for this menu.
39   * Whenever an event is detected which triggers a popup menu a Swing-specific
40   * {@link PopupMenuBuilder} is created and passed to the
41   * {@link PopupMenuHandler}. Using this object the handler can
42   * define and display the menu.
43   * </p>
44   *
45   * @author Oliver Heger
46   * @version $Id: SwingPopupListener.java 205 2012-01-29 18:29:57Z oheger $
47   */
48  class SwingPopupListener extends MouseAdapter
49  {
50      /** Stores the menu handler for constructing the menu. */
51      private final PopupMenuHandler menuHandler;
52  
53      /** Stores a reference to the component builder data. */
54      private final ComponentBuilderData componentBuilderData;
55  
56      /** Stores the current action manager. */
57      private final ActionManager actionManager;
58  
59      /** Stores the action builder data object. */
60      private final ActionBuilder actionBuilder;
61  
62      /**
63       * Creates a new instance of {@code SwingPopupListener} and initializes it
64       * with the handler for constructing the menu.
65       *
66       * @param handler the menu handler (must not be <b>null</b>)
67       * @param compData the component builder data object (must not be
68       *        <b>null</b>)
69       * @param actMan the current {@code ActionManager}
70       * @param builder the {@code ActionBuilder}
71       * @throws IllegalArgumentException if a required parameter is <b>null</b>
72       */
73      public SwingPopupListener(PopupMenuHandler handler,
74              ComponentBuilderData compData, ActionManager actMan,
75              ActionBuilder builder)
76      {
77          if (handler == null)
78          {
79              throw new IllegalArgumentException("Menu handler must not be null!");
80          }
81          if (compData == null)
82          {
83              throw new IllegalArgumentException(
84                      "ComponentBuilderData must not be null!");
85          }
86  
87          menuHandler = handler;
88          componentBuilderData = compData;
89          actionManager = actMan;
90          actionBuilder = builder;
91      }
92  
93      /**
94       * Returns the menu handler used by this object.
95       *
96       * @return the menu handler
97       */
98      public PopupMenuHandler getMenuHandler()
99      {
100         return menuHandler;
101     }
102 
103     /**
104      * Returns the {@code ComponentBuilderData} object. This object will be
105      * passed to the menu handler.
106      *
107      * @return the {@code ComponentBuilderData}
108      */
109     public ComponentBuilderData getComponentBuilderData()
110     {
111         return componentBuilderData;
112     }
113 
114     /**
115      * Returns the current {@code ActionManager}.
116      *
117      * @return the {@code ActionManager}
118      */
119     public ActionManager getActionManager()
120     {
121         return actionManager;
122     }
123 
124     /**
125      * Returns the {@code ActionBuilder}.
126      *
127      * @return the {@code ActionBuilder}
128      */
129     public ActionBuilder getActionBuilder()
130     {
131         return actionBuilder;
132     }
133 
134     /**
135      * A mouse button was pressed. This implementation delegates to
136      * {@code maybeShowPopup()}.
137      *
138      * @param event the mouse event
139      */
140     @Override
141     public void mousePressed(MouseEvent event)
142     {
143         maybeShowPopup(event);
144     }
145 
146     /**
147      * A mouse button was released. This implementation delegates to
148      * {@code maybeShowPopup()}.
149      *
150      * @param event the mouse event
151      */
152     @Override
153     public void mouseReleased(MouseEvent event)
154     {
155         maybeShowPopup(event);
156     }
157 
158     /**
159      * Checks whether the passed in event is a trigger for a popup menu and
160      * invokes the menu handler if this is the case. This method is called by
161      * other event handler methods.
162      *
163      * @param event the mouse event
164      * @throws FormBuilderRuntimeException if the creation of the menu causes an
165      *         error
166      */
167     protected void maybeShowPopup(MouseEvent event)
168     {
169         if (event.isPopupTrigger())
170         {
171             try
172             {
173                 getMenuHandler().constructPopup(createMenuBuilder(event),
174                         getComponentBuilderData());
175             }
176             catch (FormActionException faex)
177             {
178                 throw new FormBuilderRuntimeException(
179                         "Exception when creating popup menu", faex);
180             }
181         }
182     }
183 
184     /**
185      * Creates a {@code PopupMenuBuilder} for defining a menu. This method
186      * is called when an event is detected that triggers a context menu. The
187      * builder created by this method is passed to the
188      * {@code PopupMenuHandler}.
189      *
190      * @param event the mouse event
191      * @return the {@code PopupMenuBuilder} to be used for constructing the
192      *         menu
193      */
194     protected PopupMenuBuilder createMenuBuilder(MouseEvent event)
195     {
196         return new SwingPopupMenuBuilder(getActionManager(),
197                 getActionBuilder(), event);
198     }
199 }
