79      /**
80       * Converts the passed in Swing (or AWT) color into a platform-independent
81       * logic color. <b>null</b> values are allowed as input values, the result
82       * will then be <b>null</b>, too.
83       *
84       * @param color the Swing color object
85       * @return the corresponding logic color object
86       */
87      public static Color swing2LogicColor(java.awt.Color color)
88      {
89          return (color != null) ? Color.newRGBInstance(color.getRed(), color
90                  .getGreen(), color.getBlue()) : null;
91      }

93      /**
94       * Converts the passed in logic color description into a Swing (or AWT)
95       * specific color object. If the color is not supported (if it is based on a
96       * logic color definition), this method returns <b>null</b>. <b>null</b>
97       * values are allowed as input values, the result will then be <b>null</b>,
98       * too.
99       *
100      * @param color the logic color description
101      * @return the corresponding Swing color object
102      */
103     public static java.awt.Color logic2SwingColor(Color color)
104     {
105         if (color != null)
106         {
107             if (!color.isLogicColor())
108             {
109                 return new java.awt.Color(color.getRed(), color.getGreen(),
110                         color.getBlue());
111             }
112             else
113             {
114                 LOG.warn("Cannot convert color: " + color
115                         + "! Logic colors are not supported by Swing.");
116             }
117         }
118 
119         return null;
120     }

258     /**
259      * Combines two tool tips. Some components use tool tips combined of two
260      * components. This method generates such a combined tool tip by creating
261      * the concatenation of the passed in strings. The strings can be
262      * <b>null</b> or empty; then no separator is placed in the resulting
263      * string. The resulting combined tip is also converted to HTML if necessary
264      * by calling {@link #toHtml(String)}.
265      *
266      * @param tip1 component 1 of the tool tip
267      * @param tip2 component 2 of the tool tip
268      * @param separator the separator between the tip components
269      * @return the combined tool tip
270      * @deprecated Functionality to combine tool tips is now available in
271      * {@link RadioGroupWidgetHandler}.
272      */
273     @Deprecated
274     public static String combineToolTips(String tip1, String tip2,
275             String separator)
276     {
277         String tip;
278         if (StringUtils.isEmpty(tip1))
279         {
280             tip = tip2;
281         }
282         else if (StringUtils.isEmpty(tip2))
283         {
284             tip = tip1;
285         }
286         else
287         {
288             StringBuilder buf = new StringBuilder();
289             buf.append(tip1);
290             buf.append(separator);
291             buf.append(tip2);
292             tip = buf.toString();
293         }
294

298     /**
299      * Transforms the specified string to an HTML string. Swing components can
300      * deal with HTML code. This method is called if HTML may be required to
301      * display a text correctly, e.g. because it contains line breaks. This
302      * implementation checks whether a transformation to HTML is necessary. If
303      * this is the case, HTML code is returned. Otherwise, result is the same
304      * text as passed in. We check here for the existence of line breaks. If
305      * some are found, HTML code is returned which contains {@code <br>} tags for the
306      * line breaks.
307      *
308      * @param s the string to be processed (can be <b>null</b>)
309      * @return the processed string
310      */
311     public static String toHtml(String s)
312     {
313         if (s != null)
314         {
315             if (s.indexOf(LF) >= 0)
316             {
317                 StringBuilder buf = new StringBuilder();
318                 buf.append(HTML_START);
319                 buf.append(s.replace(LF, TAG_BR));
320                 buf.append(HTML_END);
321                 return buf.toString();
322             }
323         }
324 
325         return s;
326     }
327 
328     /**
329      * Removes HTML markup from the given string if it is contained. This
330      * function is the reverse operation of {@link #toHtml(String)}. It can be
331      * used to obtain the original content of a text that has been converted to
332      * Swing-like HTML.
333      *
334      * @param s the string to be processed (can be <b>null</b>)
335      * @return the processed string
336      * @since 1.4
337      */
338     public static String removeHtml(String s)
339     {
340         if (s != null && s.startsWith(HTML_START) && s.endsWith(HTML_END))
341         {
342             String unwrapped = s.substring(HTML_START.length(),
343                     s.length() - HTML_END.length());
344             return unwrapped.replace(TAG_BR, LF);
345         }
346         return s;
347     }

364      * Creates a scroll pane for the specified component. The preferred with of
365      * the scroll pane is properly initialized. If values greater than 0 are

397     /**
398      * Creates a scroll pane for the specified component that is initialized in
399      * a callback when the builder operation is finished. This is necessary to
400      * calculate the correct preferred size if it is defined in certain units
401      * (that require an initialized parent component). Therefore, this method
402      * creates and returns an uninitialized scroll pane and adds a callback
403      * which completes the initialization later.
404      *
405      * @param comp the component to be added to the scroll pane
406      * @param scrWidth the preferred scroll width
407      * @param scrHeight the preferred scroll height
408      * @param sizeHandler the size handler
409      * @param tag the current tag
410      * @return the scroll pane
411      * @since 1.4
412      */
413     public static JScrollPane scrollPaneLazyInit(final Scrollable comp,
414             final NumberWithUnit scrWidth, final NumberWithUnit scrHeight,
415             final SwingSizeHandler sizeHandler, final FormBaseTag tag)
416     {
417         final JScrollPane scr = scrollPaneFor(comp);
418         ComponentBuilderData builderData =
419                 FormBaseTag.getBuilderData(tag.getContext());
420         builderData.addCallBack(new ComponentBuilderCallBack()
421         {
422             public void callBack(ComponentBuilderData builderData,
423                     Object params) throws FormBuilderException
424             {
425                 Object container = tag.findContainer().getContainer();
426                 int width = scrWidth.toPixel(sizeHandler, container, false);
427                 int height = scrHeight.toPixel(sizeHandler, container, true);
428                 scr.setPreferredSize(
429                         calculatePreferredScrollSize(comp, width, height, scr));
430             }
431         }, null);
432         return scr;
433     }

435     /**
436      * Calculates the preferred size of a scroll pane based on the given
437      * parameters.
438      *
439      * @param comp the component to be added to the scroll pane
440      * @param scrWidth the preferred scroll width (&lt;= 0 for undefined)
441      * @param scrHeight the preferred scroll height (&lt;= 0 for undefined)
442      * @param scr the scroll pane
443      * @return the preferred scroll size for this scroll pane
444      */
445     private static Dimension calculatePreferredScrollSize(Scrollable comp,
446             int scrWidth, int scrHeight, JScrollPane scr)
447     {
448         Dimension prefSize;
449         Insets is;
450 
451         if (scrWidth <= 0 || scrHeight <= 0)
452         {
453             is = scr.getInsets();
454             prefSize = comp.getPreferredScrollableViewportSize();
455         }
456         else
457         {
458             is = null;
459             prefSize = new Dimension();
460         }
461 
462         if (scrWidth <= 0)
463         {
464             prefSize.width +=
465                     scr.getVerticalScrollBar().getPreferredSize().getWidth()
466                             + is.left + is.right;
467         }
468         else
469         {
470             prefSize.width = scrWidth;
471         }
472 
473         if (scrHeight <= 0)
474         {
475             prefSize.height +=
476                     scr.getHorizontalScrollBar().getPreferredSize().getHeight()
477                             + is.top + is.bottom;
478         }
479         else
480         {
481             prefSize.height = scrHeight;
482         }
483         return prefSize;
484     }
