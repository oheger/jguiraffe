56      /**
57       * Creates a frame window. This implementation returns a wrapper for a
58       * <code>javax.swing.JFrame</code> object.
59       *
60       * @param builderData the builder data object
61       * @param data the data for the window
62       * @param wnd the window to be initialized
63       * @return the window
64       * @throws WindowBuilderException if an error occurs
65       */
66      public Window createFrame(WindowBuilderData builderData, WindowData data,
67              Window wnd) throws WindowBuilderException
68      {
69          FrameAdapter result;
70  
71          if (wnd == null)
72          {
73              result = new FrameAdapter(new JFrame(), data.isCenter());
74          }
75  
76          else
77          {
78              result = (FrameAdapter) wnd;
79              initWindowBounds(result, data, builderData.getParentWindow());
80              initSwingWindowProperties(result, data, builderData
81                      .getParentWindow());
82              JFrame frame = result.getFrame();
83              if (data.getIcon() != null)
84              {
85                  frame.setIconImage(((ImageIcon) data.getIcon()).getImage());
86              }
87              frame.setJMenuBar((JMenuBar) data.getMenuBar());
88              frame.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
89          }
90  
91          return result;
92      }

94      /**
95       * Creates and initializes an internal frame window and adds it to its
96       * parent frame. This implementation will return an object derived from
97       * Swing's <code>JInternalFrame</code> class. This internal frame will
98       * already have been added to the parent frame's desktop. For this to work
99       * the parent window must be a Swing <code>JFrame</code> and it must
100      * contain a <code>JDesktopPane</code> instance; otherwise an exception
101      * will be thrown.
102      *
103      * @param builderData the builder data object
104      * @param data the data for the window
105      * @param wnd the window to be initialized
106      * @return the window
107      * @throws WindowBuilderException if an error occurs
108      */
109     public Window createInternalFrame(WindowBuilderData builderData,
110             WindowData data, Window wnd) throws WindowBuilderException
111     {
112         if (wnd == null)
113         {
114             return new InternalFrameAdapter(data.isCenter());
115         }
116 
117         else
118         {
119             InternalFrameAdapter iframe = (InternalFrameAdapter) wnd;
120             initInternalFrameBounds(iframe, data, builderData.getParentWindow());
121             initSwingWindowProperties(iframe, data, builderData
122                     .getParentWindow());
123             iframe.setDefaultCloseOperation(JInternalFrame.DO_NOTHING_ON_CLOSE);
124             iframe.setJMenuBar((JMenuBar) data.getMenuBar());
125             iframe.setFrameIcon((Icon) data.getIcon());
126             iframe.setClosable(data.isClosable());
127             iframe.setIconifiable(data.isIconifiable());
128             iframe.setMaximizable(data.isMaximizable());
129             iframe.setResizable(data.isResizable());
130 
131             return iframe;
132         }
133     }

135     /**
136      * Creates a dialog window. This implementation returns a wrapper for a
137      * <code>javax.swing.JDialog</code> object.
138      *
139      * @param builderData the builder data object
140      * @param data the data for the window
141      * @param modal the modal flag
142      * @param wnd the window to be initialized
143      * @return the window
144      * @throws WindowBuilderException if an error occurs
145      */
146     public Window createDialog(WindowBuilderData builderData, WindowData data,
147             boolean modal, Window wnd) throws WindowBuilderException
148     {
149         DialogAdapter result;
150 
151         if (wnd == null)
152         {
153             result = new DialogAdapter(createJDialog(builderData),
154                     data.isCenter());
155         }
156 
157         else
158         {
159             result = (DialogAdapter) wnd;
160             initWindowBounds(result, data, builderData.getParentWindow());
161             initSwingWindowProperties(result, data, builderData
162                     .getParentWindow());
163             JDialog dlg = result.getDialog();
164             dlg.setDefaultCloseOperation(JDialog.DO_NOTHING_ON_CLOSE);
165             dlg.setModal(modal);
166             dlg.setResizable(data.isResizable());
167             dlg.setJMenuBar((JMenuBar) data.getMenuBar());
168         }
169 
170         return result;
171     }

200     /**
201      * Initializes the given Swing window implementation from the given data
202      * object.
203      *
204      * @param window the window
205      * @param data the data object
206      * @param parent the parent window
207      */
208     protected void initSwingWindowProperties(SwingWindow window,
209             WindowData data, Window parent)
210     {
211         window.setTitle(data.getTitle());
212         window.setWindowController(data.getController());
213         window.setParentWindow(parent);
214 
215         if (data.isAutoClose())
216         {
217             window.registerAutoCloseListener();
218         }
219 
220         handleCloseOnEsc(window, data);
221         handleDefaultButton(window, data);
222     }

224     /**
225      * Creates a new dialog object. This method ensures that the correct owner
226      * window will be set.
227      *
228      * @param builderData the builder data object (contains the parent window)
229      * @return the new dialog
230      */
231     protected JDialog createJDialog(WindowBuilderData builderData)
232     {
233         if (builderData.getParentWindow() != null)
234         {
235             if (builderData.getParentWindow() instanceof FrameAdapter)
236             {
237                 return new JDialog(((FrameAdapter) (builderData
238                         .getParentWindow())).getFrame());
239             }
240 
241             if (builderData.getParentWindow() instanceof DialogAdapter)
242             {
243                 return new JDialog(((DialogAdapter) builderData
244                         .getParentWindow()).getDialog());
245             }
246         }
247 
248         return new JDialog();
249     }

251     /**
252      * Initializes the internal frame's bounds. It will also be added to its
253      * parent's desktop (if this is not possible, an exception will be thrown).
254      *
255      * @param iframe the internal frame to initialize
256      * @param data the data
257      * @param parent the parent window
258      * @throws WindowBuilderException if the internal frame cannot be added to
259      * its parent
260      */
261     private void initInternalFrameBounds(InternalFrameAdapter iframe,
262             WindowData data, Window parent) throws WindowBuilderException
263     {
264         if (!(parent instanceof FrameAdapter))
265         {
266             throw new WindowBuilderException(
267                     "Parent window must be defined and a Swing frame window!");
268         }
269 
270         JDesktopPane desktop = WindowHelper
271                 .findDesktopPane(((FrameAdapter) parent).getFrame());
272         if (desktop == null)
273         {
274             throw new WindowBuilderException(
275                     "No JDesktopPane found for adding the internal frame! "
276                             + "Please use the DesktopPanelTag to define the parent "
277                             + "frame's desktop panel.");
278         }
279 
280         WindowHelper.initComponentBounds(iframe, data);
281         if (data.isCenter())
282         {
283             iframe.setLocation((desktop.getWidth() - iframe.getWidth()) / 2,
284                     (desktop.getHeight() - iframe.getHeight()) / 2);
285         }
286 
287         desktop.add(iframe);
288     }

290     /**
291      * Takes care about closing the window when the {@code ESCAPE} key is
292      * pressed. This method checks whether the close on escape flag is set. If
293      * so, it registers an action for this key that either invokes the cancel
294      * button (if defined) or directly closes the window.
295      *
296      * @param window the window to be handled
297      * @param data the window data object
298      */
299     @SuppressWarnings("serial")
300     private void handleCloseOnEsc(final SwingWindow window, WindowData data)
301     {
302         if (data.isCloseOnEsc())
303         {
304             ComponentBuilderData builderData = data.getComponentBuilderData();
305             final JButton cancelButton = (JButton) builderData
306                     .getComponent(builderData.getCancelButtonName());
307             Action escAction = (cancelButton != null) ? new AbstractAction()
308             {
309                 public void actionPerformed(ActionEvent e)
310                 {
311                     cancelButton.doClick();
312                 }
313             } : new AbstractAction()
314             {
315                 public void actionPerformed(ActionEvent e)
316                 {
317                     window.close(false);
318                 }
319             };
320 
321             JRootPane rootPane = window.getRootPane();
322             rootPane.getInputMap(JComponent.WHEN_IN_FOCUSED_WINDOW).put(
323                     KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0), CMD_ESCAPE);
324             rootPane.getActionMap().put(CMD_ESCAPE, escAction);
325         }
326     }

328     /**
329      * Takes care about the default button of the specified window. This method
330      * checks whether a default button is specified. If so, it is obtained from
331      * the current {@link ComponentBuilderData} object and set as the window's
332      * root pane's default button.
333      *
334      * @param window the window to be initialized
335      * @param data the window data object
336      */
337     private void handleDefaultButton(SwingWindow window, WindowData data)
338     {
339         ComponentBuilderData builderData = data.getComponentBuilderData();
340         if (builderData.getDefaultButtonName() != null)
341         {
342             JButton defBtn = (JButton) builderData.getComponent(builderData
343                     .getDefaultButtonName());
344             window.getRootPane().setDefaultButton(defBtn);
345         }
346     }
