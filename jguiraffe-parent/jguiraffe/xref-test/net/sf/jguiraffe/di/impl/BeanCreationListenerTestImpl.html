16  package net.sf.jguiraffe.di.impl;

18  import static org.junit.Assert.assertEquals;
19  import static org.junit.Assert.assertFalse;
20  
21  import java.util.LinkedList;
22  import java.util.List;
23  
24  import net.sf.jguiraffe.di.BeanContext;
25  import net.sf.jguiraffe.di.BeanCreationEvent;
26  import net.sf.jguiraffe.di.BeanCreationListener;
27  import net.sf.jguiraffe.di.BeanProvider;
28  import net.sf.jguiraffe.di.DependencyProvider;
29  
30  /**
31   * A test implementation of a bean creation listener for testing whether events
32   * are correctly sent.
33   *
34   * @author Oliver Heger
35   * @version $Id: BeanCreationListenerTestImpl.java 205 2012-01-29 18:29:57Z oheger $
36   */
37  class BeanCreationListenerTestImpl implements BeanCreationListener
38  {
39      /** A list with the events received. */
40      private final List<BeanCreationEvent> events = new LinkedList<BeanCreationEvent>();
41  
42      /**
43       * Records this invocation and stores the passed in event.
44       */
45      public void beanCreated(BeanCreationEvent event)
46      {
47          events.add(event);
48      }
49  
50      /**
51       * Tests the next received event.
52       *
53       * @param expCtx the expected bean context
54       * @param expBean the expected bean
55       * @param expProvider the expected bean provider
56       * @param expDepProvider the expected dependency provider
57       */
58      public void checkNextEvent(BeanContext expCtx, Object expBean,
59              BeanProvider expProvider, DependencyProvider expDepProvider)
60      {
61          assertFalse("No more events", events.isEmpty());
62          BeanCreationEvent event = events.remove(0);
63          assertEquals("Wrong bean context", expCtx, event.getBeanContext());
64          assertEquals("Wrong bean", expBean, event.getBean());
65          assertEquals("Wrong provider", expProvider, event.getBeanProvider());
66          assertEquals("Wrong dependency provider", expDepProvider, event
67                  .getDependencyProvider());
68      }
69  
70      /**
71       * Returns a flag whether more events are available.
72       *
73       * @return a flag if there are more events
74       */
75      public boolean hasMoreEvents()
76      {
77          return !events.isEmpty();
78      }
79  }
