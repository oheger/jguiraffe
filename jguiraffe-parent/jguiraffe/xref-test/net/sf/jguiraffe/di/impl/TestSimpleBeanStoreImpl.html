152     @Test
153     public void testProviderNames()
154     {
155         final int count = 10;
156         for (int i = 0; i < count; i++)
157         {
158             store.addBean(BEAN_NAME + i, Integer.valueOf(i));
159         }
160         store.addBeanContributor(new SimpleBeanStoreImpl.BeanContributor()
161         {
162             public void beanNames(Set<String> names)
163             {
164                 // return some test bean names
165                 for (int i = 0; i < count; i++)
166                 {
167                     names.add(BEAN_NAME + (count + i));
168                 }
169             }
170 
171             public Object getBean(String name)
172             {
173                 // should not be called
174                 throw new UnsupportedOperationException("Not implemented!");
175             }
176 
177         });
178 
179         Set<String> names = store.providerNames();
180         assertEquals("Wrong number of names", 2 * count, names.size());
181         for (int i = 0; i < 2 * count; i++)
182         {
183             String name = BEAN_NAME + i;
184             assertTrue("Bean name not found: " + name, names.contains(name));
185         }
