68      /**
69       * Creates a mock object of the specified class. The mock is also added to
70       * an internal list. The <code>verify()</code> method verifies all mocks.
71       *
72       * @param mockCls the class of the mock object
73       * @return the mock
74       */
75      protected <T> T createMock(Class<T> mockCls)
76      {
77          T mock = EasyMock.createMock(mockCls);
78          if (mocks == null)
79          {
80              mocks = new LinkedList<Object>();
81          }
82          mocks.add(mock);
83          return mock;
84      }
85  
86      /**
87       * Verifies all mock objects that have been created using the
88       * <code>createMock()</code> method.
89       */
90      protected void verify()
91      {
92          EasyMock.verify(mocks.toArray());
93      }
