200     /**
201      * Creates and populates some test action stores. This data is used for
202      * testing recursive methods operating on action stores and their groups.
203      *
204      * @return the test action store
205      */
206     private ActionStore setUpStores()
207     {
208         final int storeCount = 3;
209         final int overlapCount = 2;
210         final int actionCount = 5;
211 
212         ActionStore parent = null;
213         ActionStore store = null;
214         int actionIdx = 0;
215         for (int i = 0; i < storeCount; i++)
216         {
217             // create a store and fill in some actions
218             store = new ActionStore(parent);
219             for (int j = 0; j < actionCount; j++, actionIdx++)
220             {
221                 FormActionImpl action = new FormActionImpl(ACTION_PREFIX
222                         + actionIdx);
223                 action.setData(String.valueOf(i));
224                 store.addAction(action);
225             }
226             // ensure that there are actions with same names in the next store
227             actionIdx -= overlapCount;
228             parent = store;
229         }
230 
231         // add some groups
232         store.addActionToGroup(ACTION_PREFIX + "6", GROUP_PREFIX);
233         store.addActionToGroup(ACTION_PREFIX + "7", GROUP_PREFIX);
234         store.getParent().addActionToGroup(ACTION_PREFIX + "6", GROUP_PREFIX);
235         store.getParent().addActionToGroup(ACTION_PREFIX + "5", GROUP_PREFIX);
236         store.getParent().getParent().addActionToGroup(ACTION_PREFIX + "0",
237                 GROUP_PREFIX + "test");
238 
239         return store;
240     }

331     /**
332      * Searches for a certain action.
333      *
334      * @param actions the actions collection
335      * @param no the index of this action
336      * @param store the store number
337      */
338     private void findAction(Collection<FormAction> actions, int no, int store)
339     {
340         boolean found = false;
341 
342         for (FormAction action : actions)
343         {
344             String actionName = ACTION_PREFIX + no;
345             if (actionName.equals(action.getName()))
346             {
347                 if (String.valueOf(store).equals(
348                         ((FormActionImpl) action).getData()))
349                 {
350                     found = true;
351                     break;
352                 }
353             }
354         }
355 
356         assertTrue("Cannot find action " + no + " in store " + store, found);
357     }
