16  package net.sf.jguiraffe.gui.builder.action.tags;
17  
18  import net.sf.jguiraffe.gui.builder.components.FormBuilderException;
19  import net.sf.jguiraffe.gui.builder.event.FormListenerType;
20  
21  import org.apache.commons.jelly.JellyContext;
22  import org.apache.commons.jelly.JellyTagException;
23  import org.apache.commons.jelly.MissingAttributeException;
24  import org.junit.Before;
25  import org.junit.Test;
26  
27  /**
28   * Test class for {@code EventListenerTypeTag}. This class only tests some error
29   * conditions. A successful execution of the tag is tested by other test classes
30   * that execute a test script.
31   *
32   * @author Oliver Heger
33   * @version $Id: TestEventListenerTypeTag.java 205 2012-01-29 18:29:57Z oheger $
34   */
35  public class TestEventListenerTypeTag
36  {
37      /** The tag to be tested. */
38      private EventListenerTypeTag tag;
39  
40      @Before
41      public void setUp() throws Exception
42      {
43          tag = new EventListenerTypeTag();
44          tag.setParent(new FormEventListenerTagTestImpl());
45          JellyContext ctx = new JellyContext();
46          tag.setContext(ctx);
47      }
48  
49      /**
50       * Tests whether an invalid parent tag is detected.
51       */
52      @Test(expected = JellyTagException.class)
53      public void testProcessWrongParent() throws JellyTagException,
54              FormBuilderException
55      {
56          tag.setType(FormListenerType.ACTION.name());
57          tag.setParent(null);
58          tag.process();
59      }
60  
61      /**
62       * Tries to execute a tag without a type attribute.
63       */
64      @Test(expected = MissingAttributeException.class)
65      public void testProcessNoType() throws JellyTagException,
66              FormBuilderException
67      {
68          tag.process();
69      }
70  
71      /**
72       * Tries to execute a tag for a non-standard type without a listener class.
73       */
74      @Test(expected = JellyTagException.class)
75      public void testProcessNoListenerClass() throws JellyTagException,
76              FormBuilderException
77      {
78          tag.setType("Expansion");
79          tag.process();
80      }
81  
82      /**
83       * A test implementation of a form event listener tag.
84       */
85      private static class FormEventListenerTagTestImpl extends
86              FormEventListenerTag
87      {
88          public FormEventListenerTagTestImpl()
89          {
90              super(FormListenerType.ACTION);
91          }
92      }
93  }
