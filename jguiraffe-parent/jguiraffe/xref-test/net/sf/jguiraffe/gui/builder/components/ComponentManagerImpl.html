150     public Object createPercentLayout(PercentLayoutTag tag)
151             throws FormBuilderException
152     {
153         StringBuilder buf = new StringBuilder("PercentLayout [ COLS = [");
154         if (tag.getColumns() != null)
155         {
156             buf.append(' ').append(tag.getColumns());
157         }
158         else
159         {
160             for (Iterator it = tag.getColConstraints().iterator(); it.hasNext();)
161             {
162                 buf.append(' ').append(it.next());
163             }
164         }
165         buf.append(" ] ROWS = [");
166         if (tag.getRows() != null)
167         {
168             buf.append(' ').append(tag.getRows());
169         }
170         else
171         {
172             for (Iterator it = tag.getRowConstraints().iterator(); it.hasNext();)
173             {
174                 buf.append(' ').append(it.next());
175             }
176         }
177         buf.append(" ]");
178         if (!tag.getColGroups().isEmpty())
179         {
180             buf.append(" COLGRPS = [ ");
181             for (Iterator it = tag.getColGroups().iterator(); it.hasNext();)
182             {
183                 buf.append(it.next()).append(' ');
184             }
185             buf.append(']');
186         }
187         if (!tag.getRowGroups().isEmpty())
188         {
189             buf.append(" ROWGRPS = [ ");
190             for (Iterator it = tag.getRowGroups().iterator(); it.hasNext();)
191             {
192                 buf.append(it.next()).append(' ');
193             }
194             buf.append(']');
195         }
196         if (tag.getPercentLayout().isCanShrink())
197         {
198             appendAttr(buf, "SHRINK", Boolean.TRUE);
199         }
200         buf.append(" ]");
201         return buf.toString();
202     }

306             StringBuilder b2 = new StringBuilder(toPropString("SPLITTER", buf).toString());
307             b2.append(" { ").append(tag.getFirstComponent()).append(", ");
308             b2.append(tag.getSecondComponent()).append(" }");
309             return b2.toString();

326                 buf.append(", ");

330         buf.append(" }");
331         return buf.toString();

347     /**
348      * Helper method for creating a component handler for a button tag.
349      *
350      * @param tag the tag
351      * @param create the create flag
352      * @param name the name of the component (i.e. the button type)
353      * @param furtherAttrs optional additional attributes to be added
354      * @return the component handler
355      * @throws FormBuilderException if an error occurs
356      */
357     private ComponentHandler createPushButton(ToggleButtonTag tag,
358             boolean create, String name, String furtherAttrs)
359             throws FormBuilderException
360     {
361         if (!create)
362         {
363             StringBuilder buf = new StringBuilder();
364             dumpTextIconData(buf, tag.getTextIconData());
365             appendAttr(buf, "CMD", tag.getCommand());
366             dumpInputComponent(buf, tag);
367             if (furtherAttrs != null)
368             {
369                 buf.append(' ').append(furtherAttrs);
370             }
371             return createHandler(name, buf, tag, Boolean.TYPE);
372         }
373         else
374         {
375             return null;
376         }
377     }

644     public ComponentHandler createTable(TableTag tag, boolean create)
645             throws FormBuilderException
646     {
647         if (create)
648         {
649             return null;
650         }
651         StringBuilder buf = new StringBuilder();
652         dumpInputComponent(buf, tag);
653         appendAttr(buf, "SELECTIONFG", tag.getSelectionForegroundColor());
654         appendAttr(buf, "SELECTIONBG", tag.getSelectionBackgroundColor());
655         appendAttr(buf, "EDITABLE", tag.isTableEditable());
656         appendAttr(buf, "VALIDATIONERRTITLE", tag.getValidationErrorTitle());
657         if (tag.isMultiSelection())
658         {
659             buf.append(" MULTISELECT");
660         }
661         appendNoUnitAttr(buf, "SCROLLWIDTH", tag.getPreferredScrollWidth());
662         appendNoUnitAttr(buf, "SCROLLHEIGHT", tag.getPreferredScrollHeight());
663 
664         buf.append(" COLUMNS {");
665         for (TableColumnTag col : tag.getColumns())
666         {
667             buf.append(" COLUMN [");
668             dumpInputComponent(buf, col);
669             if (col.getHeaderText().isDefined())
670             {
671                 appendAttr(buf, "HEADER", col.getHeaderText().getCaption());
672             }
673             if (col.getLogicDataClass() != null)
674             {
675                 appendAttr(buf, "LOGICCLASS", col.getLogicDataClass().name());
676             }
677             else if (col.getDataClass() != null)
678             {
679                 appendAttr(buf, "CLASS", col.getDataClass().getName());
680             }
681             appendAttr(buf, "WIDTH", col.getColumnWidth());
682             if (col.getPercentWidth() != 0)
683             {
684                 appendAttr(buf, "PERCENTWIDTH", col.getPercentWidth());
685             }
686             appendAttr(buf, "EDITABLE", tag.isColumnEditable(col));
687             if (col.getRendererComponent() != null)
688             {
689                 buf.append(" RENDERER { ").append(col.getRendererComponent());
690                 buf.append(" }");
691             }
692             if (col.getEditorComponent() != null)
693             {
694                 buf.append(" EDITOR { ").append(col.getEditorComponent());
695                 buf.append(" }");
696             }
697             buf.append(" ]");
698         }
699         buf.append(" }");
700         return createHandler("TABLE", buf, tag, Integer.TYPE);
701     }
