78      /**
79       * Replays the specified mock objects including the mock for the wrapped
80       * manager.
81       *
82       * @param mocks the mocks to be replayed
83       */
84      private void replay(Object... mocks)
85      {
86          EasyMock.replay(mocks);
87          EasyMock.replay(wrappedManager);
88      }
89  
90      /**
91       * Verifies the specified mock objects including the mock for the wrapped
92       * manager.
93       *
94       * @param mocks the mocks to be verified
95       */
96      private void verify(Object... mocks)
97      {
98          EasyMock.verify(mocks);
99          EasyMock.verify(wrappedManager);
100     }

102     /**
103      * Helper method for creating a mock for a component handler of a specific
104      * type.
105      *
106      * @param <T> the type of the handler
107      * @param cls the data class of the handler
108      * @return the mock handler
109      */
110     private <T> ComponentHandler<T> componentHandlerMock(Class<T> cls)
111     {
112         @SuppressWarnings("unchecked")
113         ComponentHandler<T> handler =
114                 EasyMock.createMock(ComponentHandler.class);
115         return handler;
116     }
