135     /**
136      * Tests whether the reference to the map with the attributes can be
137      * resolved.
138      */
139     @Test
140     public void testFetchAttributesMapReference() throws JellyTagException
141     {
142         BeanContext bctx = setUpBeanContext();
143         EasyMock.expect(bctx.containsBean(VAR)).andReturn(Boolean.TRUE);
144         Map<Object, Object> map = new HashMap<Object, Object>();
145         map.put("test", "value");
146         EasyMock.expect(bctx.getBean(VAR)).andReturn(map);
147         EasyMock.replay(bctx);
148         tag.setAttributes(VAR);
149         Map<?, ?> refMap = tag.fetchAttributesMap();
150         assertEquals("Wrong map", map, refMap);
151         assertNotSame("Map not copied", map, refMap);
152         EasyMock.verify(bctx);
153     }

243     private class FontTagTestImpl extends FontTag
