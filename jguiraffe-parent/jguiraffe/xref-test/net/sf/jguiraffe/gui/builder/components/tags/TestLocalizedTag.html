36  /**
37   * Test class for LocalizedPropertyTag.
38   *
39   * @author Oliver Heger
40   * @version $Id: TestLocalizedTag.java 205 2012-01-29 18:29:57Z oheger $
41   */
42  public class TestLocalizedTag
43  {
44      /** Constant for the locale. */
45      private static final Locale LOCALE = Locale.GERMAN;
46  
47      /** Constant for the resource ID. */
48      private static final String RES_ID = "myResID";
49  
50      /** Constant for the resource group. */
51      private static final String RES_GRP = "myResourceGroup";
52  
53      /** Constant for the resource value. */
54      private static final Object RES_VALUE = "ResourceValue";
55  
56      /** Stores the transformer context mock. */
57      private TransformerContext transformerContext;
58  
59      /** Stores the mock for the resource manager. */
60      private ResourceManager resourceManager;
61  
62      /** The parent tag. */
63      private SetPropertyTag parent;
64  
65      /** Stores the tag to be tested. */
66      private LocalizedTag tag;
67  
68      @Before
69      public void setUp() throws Exception
70      {
71          tag = new LocalizedTag();
72          tag.setContext(new JellyContext());
73          parent = new SetPropertyTag();
74          parent.setContext(tag.getContext());
75          tag.setParent(parent);
76      }
77  
78      /**
79       * Prepares a test that accesses resources. This method will setup a
80       * resource manager and store it in the context.
81       */
82      private void prepareResourceTest()
83      {
84          transformerContext = EasyMock.createMock(TransformerContext.class);
85          resourceManager = EasyMock.createMock(ResourceManager.class);
86          EasyMock.expect(transformerContext.getLocale()).andReturn(LOCALE);
87          EasyMock.expect(transformerContext.getResourceManager()).andReturn(
88                  resourceManager);
89          EasyMock.replay(transformerContext);
90          ComponentBuilderData cd = new ComponentBuilderData();
91          cd.initializeForm(transformerContext, new BeanBindingStrategy());
92          cd.put(tag.getContext());
93      }
94  
95      /**
96       * Tests processing a valid tag.
97       */
98      @Test
99      public void testProcess() throws JellyTagException, FormBuilderException
100     {
101         prepareResourceTest();
102         EasyMock.expect(resourceManager.getResource(LOCALE, RES_GRP, RES_ID))
103                 .andReturn(RES_VALUE);
104         EasyMock.replay(resourceManager);
105         tag.setResid(RES_ID);
106         tag.setResgrp(RES_GRP);
107         tag.process();
108         assertEquals("Wrong value", RES_VALUE, parent.getValueData().getValue());
109         EasyMock.verify(resourceManager, transformerContext);
110     }
111 
112     /**
113      * Tests the process() if when no resource group was specified. This is
114      * completely legal.
115      */
116     @Test
117     public void testProcessNoResGrp() throws JellyTagException,
118             FormBuilderException
119     {
120         prepareResourceTest();
121         EasyMock.expect(resourceManager.getResource(LOCALE, null, RES_ID))
122                 .andReturn(RES_VALUE);
123         EasyMock.replay(resourceManager);
124         tag.setResid(RES_ID);
125         tag.process();
126         assertEquals("Wrong value", RES_VALUE, parent.getValueData().getValue());
127         EasyMock.verify(resourceManager, transformerContext);
128     }
129 
130     /**
131      * Tests the process() method if no resource ID was specified. This should
132      * cause an exception.
133      */
134     @Test(expected = MissingAttributeException.class)
135     public void testProcessNoResID() throws JellyTagException,
136             FormBuilderException
137     {
138         tag.setResgrp(RES_GRP);
139         tag.process();
140     }
141 
142     /**
143      * Tests process() if the tag has an invalid parent. This should cause an
144      * exception.
145      */
146     @Test(expected = JellyTagException.class)
147     public void testProcessInvalidParent() throws JellyTagException,
148             FormBuilderException
149     {
150         tag.setParent(new LabelTag());
151         tag.setResid(RES_ID);
152         tag.process();
153     }
