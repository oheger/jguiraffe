64          tag = new ValidatorTag();

75      /**
76       * Returns the default transformer context.
77       *
78       * @return the default (i.e. global) transformer context
79       */
80      private TransformerContext getDefaultContext()
81      {
82          return ComponentBuilderData.get(tag.getContext())
83                  .getTransformerContext();
84      }

86      /**
87       * Tests whether a correct validator wrapper was set.
88       *
89       * @param wrapper the wrapper to test
90       * @param defCtx a flag whether the default transformer context is expected
91       */
92      private void checkValidator(ValidatorWrapper wrapper, boolean defCtx)
93      {
94          assertTrue("Wrong validator wrapper type: " + wrapper,
95                  wrapper instanceof DefaultValidatorWrapper);
96          DefaultValidatorWrapper vw = (DefaultValidatorWrapper) wrapper;
97          assertEquals("Wrong wrapped validator", validator, vw.getValidator());
98          assertEquals("Wrong context", defCtx, getDefaultContext().equals(
99                  vw.getTransformerContext()));
100     }

238     /**
239      * A test implementation of the ValidatorsTag allowing us to check whether
240      * the expected methods have been called.
241      */
242     private static class ValidatorsTagTestImpl extends ValidatorsTag
243     {
244         /** Stores the expected child validator. */
245         private Validator validator;
246 
247         /** Stores the expected properties. */
248         private Map<String, Object> properties;
249 
250         /** A flag whether the tag was called. */
251         boolean called;
252 
253         public ValidatorsTagTestImpl(Validator v, Map<String, Object> props)
254         {
255             validator = v;
256             properties = props;
257         }
258 
259         @Override
260         public void addChildValidator(Validator child, Map<String, Object> props)
261         {
262             called = true;
263             assertEquals("Wrong validator passed", validator, child);
264             assertSame("Wrong properties passed", properties, props);
265         }
