35  /**
36   * Test class for ValidatorsTag.
37   *
38   * @author Oliver Heger
39   * @version $Id: TestValidatorsTag.java 205 2012-01-29 18:29:57Z oheger $
40   */
41  public class TestValidatorsTag
42  {
43      /** An input component tag used as parent. */
44      private InputComponentTag input;
45  
46      /** An output object for executing the test tag. */
47      private XMLOutput output;
48  
49      /** The tag to be tested. */
50      private ValidatorsTag tag;
51  
52      @Before
53      public void setUp() throws Exception
54      {
55          JellyContext context = new JellyContext();
56          tag = new ValidatorsTag();
57          tag.setContext(context);
58          input = new TextFieldTag();
59          input.setContext(context);
60          tag.setParent(input);
61          tag.setBody(new TagScript());
62          tag.setAttribute(ValidatorTag.ATTR_PHASE, ValidationPhase.SYNTAX.name());
63          output = new XMLOutput();
64          ComponentBuilderData cdata = new ComponentBuilderData();
65          cdata.initializeForm(new TransformerContextImpl(),
66                  new BeanBindingStrategy());
67          cdata.put(context);
68      }
69  
70      /**
71       * Tests adding a child validator.
72       */
73      @Test
74      public void testAddChildValidator() throws JellyTagException
75      {
76          Validator v = EasyMock.createMock(Validator.class);
77          EasyMock.replay(v);
78          tag.addChildValidator(v, null);
79          tag.doTag(output);
80          DefaultValidatorWrapper wrapper = (DefaultValidatorWrapper) input
81                  .getFieldValidator();
82          ChainValidator cv = (ChainValidator) wrapper.getValidator();
83          assertEquals("Wrong number of child validators", 1, cv.size());
84          assertEquals("Wrong child validator", v, cv.getChildValidator(0));
85          EasyMock.verify(v);
86      }
87  
88      /**
89       * Tests whether a chain validator is correctly initialized when stored as a
90       * variable.
91       */
92      @Test
93      public void testStoreVariable() throws JellyTagException
94      {
95          Validator v = EasyMock.createMock(Validator.class);
96          EasyMock.replay(v);
97          tag.addChildValidator(v, null);
98          final String varName = "myVariable";
99          tag.setAttribute(ValidatorTag.ATTR_VAR, varName);
100         tag.setParent(null);
101         tag.doTag(output);
102         ChainValidator cv = (ChainValidator) tag.getContext().getVariable(
103                 varName);
104         assertEquals("Wrong number of child validators", 1, cv.size());
105         assertEquals("Wrong child validator", v, cv.getChildValidator(0));
106         EasyMock.verify(v);
107     }
108 
109     /**
110      * Tests a tag with a class attribute setting an invalid base class.
111      */
112     @Test(expected = JellyTagException.class)
113     public void testInvalidClass() throws JellyTagException
114     {
115         tag.setAttribute(ValidatorTag.ATTR_CLASS, getClass().getName());
116         tag.doTag(output);
117     }
