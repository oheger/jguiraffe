16  package net.sf.jguiraffe.gui.builder.event;
17  
18  import net.sf.jguiraffe.gui.forms.ComponentHandler;
19  
20  import org.apache.commons.lang.StringUtils;
21  
22  /**
23   * A test implementation of the <code>PlatformEventManager</code> interface.
24   * This implementation simply transforms call to the
25   * <code>registerListener()</code> method to plain text that can be checked by
26   * unit tests.
27   *
28   * @author Oliver Heger
29   * @version $Id: PlatformEventManagerImpl.java 205 2012-01-29 18:29:57Z oheger $
30   */
31  public class PlatformEventManagerImpl implements PlatformEventManager
32  {
33      /** Constant for the type separator. */
34      private static final String SEPARATOR1 = " -> ";
35  
36      /** Constant for the registration data separator. */
37      private static final String SEPARATOR2 = ", ";
38  
39      /** Stores the plain text. */
40      private final StringBuilder text = new StringBuilder();
41  
42      public void registerListener(String name, ComponentHandler<?> handler,
43              FormEventManager eventManager, FormListenerType type)
44      {
45          if (text.length() > 0)
46          {
47              text.append(SEPARATOR2);
48          }
49          text.append(name).append(SEPARATOR1).append(type.name());
50      }
51  
52      public void unregisterListener(String name, ComponentHandler<?> handler,
53              FormEventManager eventManager, FormListenerType type)
54      {
55          String regText = fetchRegistrationData(name, type);
56          int pos = text.indexOf(regText);
57          if (pos < 0)
58          {
59              throw new RuntimeException("No registration data found: " + regText);
60          }
61          text.delete(pos, pos + regText.length());
62          if (pos < text.length() - 1)
63          {
64              // remove separator
65              text.delete(pos, pos + SEPARATOR2.length());
66          }
67          if (StringUtils.countMatches(text.toString(), SEPARATOR2) == 1
68                  && StringUtils.countMatches(text.toString(), SEPARATOR1) == 1)
69          {
70              pos = text.indexOf(SEPARATOR2);
71              text.delete(pos, pos + SEPARATOR2.length());
72          }
73      }
74  
75      /**
76       * Returns information about registered listeners as plain text.
77       *
78       * @return text information about registered listeners
79       */
80      public String getRegistrationData()
81      {
82          return text.toString();
83      }
84  
85      /**
86       * Counts the number of occurrences of the given combination of component
87       * name and listener type. This is useful to find out how often a component
88       * has been registered.
89       *
90       * @param name the name of the component
91       * @param type the listener type; can be <b>null</b>, then all types are
92       *        counted
93       * @return the number of occurrences of the given combination
94       */
95      public int getNumberOf(String name, FormListenerType type)
96      {
97          String data = getRegistrationData();
98          int pos1 = 0;
99          int count = 0;
100 
101         while (pos1 < data.length())
102         {
103             int pos2 = data.indexOf(SEPARATOR1, pos1);
104             int posEnd = data.indexOf(SEPARATOR2, pos2);
105             if (posEnd < 0)
106             {
107                 posEnd = data.length();
108             }
109             if (data.substring(pos1, pos2).equals(name))
110             {
111                 if (type == null
112                         || data.substring(pos2 + SEPARATOR1.length(), posEnd)
113                                 .equals(type.name()))
114                 {
115                     count++;
116                 }
117             }
118             pos1 = posEnd + SEPARATOR2.length();
119         }
120 
121         return count;
122     }
123 
124     /**
125      * Creates registration data for the specified component name and listener
126      * type and adds it to the given buffer.
127      *
128      * @param buf the buffer
129      * @param name the name of the component
130      * @param type the listener type
131      */
132     private void appendRegistrationString(StringBuilder buf, String name,
133             FormListenerType type)
134     {
135         buf.append(name).append(SEPARATOR1).append(type.name());
136     }
137 
138     /**
139      * Creates the registration data string for the specified component name and
140      * event listener type.
141      *
142      * @param name the name of the component
143      * @param type the event listener type
144      * @return a string representation of the corresponding registration data
145      */
146     private String fetchRegistrationData(String name, FormListenerType type)
147     {
148         StringBuilder buf = new StringBuilder();
149         appendRegistrationString(buf, name, type);
150         return buf.toString();
151     }
152 }
