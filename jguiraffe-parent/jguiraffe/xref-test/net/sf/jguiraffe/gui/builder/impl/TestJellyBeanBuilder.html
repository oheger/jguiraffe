200     /**
201      * Tests accessing a bean from the test bean definitions that is a
202      * singleton. The bean is accessed multiple times, always checking that the
203      * same instance is returned.
204      *
205      * @param beanName the name of the singleton bean
206      * @throws BuilderException if an error occurs
207      */
208     private void checkSingleton(String beanName) throws BuilderException
209     {
210         BeanContext context = setUpContext();
211         Object bean = context.getBean(beanName);
212         checkTestObject(bean);
213         for (int i = 1; i < COUNT; i++)
214         {
215             assertSame("Multiple instances of bean " + beanName + " returned",
216                     bean, context.getBean(beanName));
217         }
218     }
219 
220     /**
221      * Tests accessing a bean from the test bean definitions that is a factory
222      * bean (i.e. no singleton). The bean is accessed multiple times, always
223      * ensuring that different instances are returned.
224      *
225      * @param beanName the name of the factory bean
226      * @throws BuilderException if an error occurs
227      */
228     private void checkFactory(String beanName) throws BuilderException
229     {
230         BeanContext context = setUpContext();
231         Set<Object> instances = new HashSet<Object>();
232         for (int i = 0; i < COUNT; i++)
233         {
234             Object bean = context.getBean(beanName);
235             checkTestObject(bean);
236             assertTrue("Same instance returned for bean " + beanName, instances
237                     .add(bean));
238         }
239     }
