77                  handler.getLogicValidator(), handler.getReadTransformer(),

209     /**
210      * Tests a failing validation at the form level.
211      */
212     @Test
213     public void testFailedValidation()
214     {
215         ValidationMessage vm = EasyMock.createMock(ValidationMessage.class);
216         EasyMock.expect(vm.getLevel()).andReturn(ValidationMessageLevel.ERROR)
217                 .anyTimes();
218         EasyMock.replay(vm);
219         prepareValidationTest();
220         DefaultValidationResult vr = new DefaultValidationResult.Builder()
221                 .addValidationMessage(vm).build();
222         EasyMock.expect(handler.getSyntaxValidator().isValid(TEST_DATA))
223                 .andReturn(DefaultValidationResult.VALID);
224         EasyMock.expect(handler.getReadTransformer().transform(TEST_DATA))
225                 .andReturn(TRANSFORMED_DATA);
226         EasyMock.expect(handler.getLogicValidator().isValid(TRANSFORMED_DATA))
227                 .andReturn(vr);
228         replayMocks();
229         assertTrue("Wrong result of field validation", handler.validate(
230                 ValidationPhase.SYNTAX).isValid());
231         assertSame("Wrong result of form validation", vr, handler
232                 .validate(ValidationPhase.LOGIC));
233         verifyMocks();
234         assertNull("Data of handler was not reset", handler.getData());
235         EasyMock.verify(vm);
236     }
