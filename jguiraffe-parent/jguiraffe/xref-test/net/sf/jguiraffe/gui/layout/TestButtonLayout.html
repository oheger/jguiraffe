144     /**
145      * Tests whether the specified constraint represents a gap of the given size
146      *
147      * @param cc the constraint
148      * @param size the size of the gap
149      * @param weight the expected weight factor
150      */
151     private void checkGap(CellConstraints cc, int size, int weight)
152     {
153         assertEquals("Got a cell size", CellSize.NONE, cc.getCellSize());
154         assertEquals("Wrong size", new NumberWithUnit(size), cc.getMinSize());
155         assertEquals("Wrong weight", weight, cc.getWeight());
156     }

158     /**
159      * Tests the column constraint for a button.
160      *
161      * @param cc the constraint to be tested
162      */
163     private void checkButtonConstr(CellConstraints cc)
164     {
165         assertEquals("Wrong alignment", CellAlignment.FULL, cc.getAlignment());
166         assertEquals("Wrong cell size", CellSize.PREFERRED, cc.getCellSize());
167         assertEquals("Wrong minimum size", NumberWithUnit.ZERO, cc.getMinSize());
168         assertEquals("Got a weight factor", 0, cc.getWeight());
169     }

171     /**
172      * Tests the constraints of the test layout.
173      *
174      * @param weightLeft the expected weight factor for the left margin
175      * @param weightRight the expected weight factor for the right margin
176      */
177     private void checkLayout(int weightLeft, int weightRight)
178     {
179         assertEquals("Wrong number of rows", 3, layout.getRowCount());
180         checkGap(layout.getRowConstraints(0), VERT_MARGIN, 0);
181         checkGap(layout.getRowConstraints(2), VERT_MARGIN, 0);
182         CellConstraints cc = layout.getRowConstraints(1);
183         assertEquals("Wrong cell alignment", CellAlignment.CENTER, cc
184                 .getAlignment());
185         assertEquals("Wrong cell size", CellSize.PREFERRED, cc.getCellSize());
186 
187         CellConstraints[] cccols = layout.getAllColumnConstraints();
188         assertEquals("Wrong number of constraints", 7, cccols.length);
189         checkGap(cccols[0], HOR_MARGIN, weightLeft);
190         checkButtonConstr(cccols[1]);
191         checkGap(cccols[2], GAP, 0);
192         checkButtonConstr(cccols[3]);
193         checkGap(cccols[4], GAP, 0);
194         checkButtonConstr(cccols[5]);
195         checkGap(cccols[6], HOR_MARGIN, weightRight);
196 
197         assertTrue("Got row groups", layout.getRowGroups().isEmpty());
198         Collection<CellGroup> colGroups = layout.getColumnGroups();
199         assertEquals("Wrong number of column groups", 1, colGroups.size());
200         CellGroup group = colGroups.iterator().next();
201         int[] sizes = new int[] {
202                 HOR_MARGIN, MAX_WIDTH - 10, GAP, MAX_WIDTH, GAP,
203                 MAX_WIDTH - 20, HOR_MARGIN
204         };
205         int[] expSizes = new int[] {
206                 HOR_MARGIN, MAX_WIDTH, GAP, MAX_WIDTH, GAP, MAX_WIDTH,
207                 HOR_MARGIN
208         };
209         group.apply(sizes);
210         assertTrue("Wrong sizes", Arrays.equals(expSizes, sizes));
211     }

243     /**
244      * Helper method for checking a PercentData object.
245      *
246      * @param col the expected column
247      * @param row the expected row
248      */
249     private void checkPercentData(int col, int row)
250     {
251         PercentData pd = layout.getPercentData(col, row);
252         assertEquals("Wrong column", col, pd.getColumn());
253         assertEquals("Wrong row", row, pd.getRow());
254         assertEquals("Wrong span X", 1, pd.getSpanX());
255         assertEquals("Wrong span Y", 1, pd.getSpanY());
256         assertEquals("Got target column", PercentData.POS_UNDEF, pd
257                 .getTargetColumn());
258         assertEquals("Got target row", PercentData.POS_UNDEF, pd.getTargetRow());
259         assertNull("Got col constraints", pd.getColumnConstraints());
260         assertNull("Got row constraints", pd.getRowConstraints());
261     }
