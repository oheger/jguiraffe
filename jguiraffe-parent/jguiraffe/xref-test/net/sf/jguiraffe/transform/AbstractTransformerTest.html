16  package net.sf.jguiraffe.transform;

18  import static org.junit.Assert.assertEquals;
19  import static org.junit.Assert.assertFalse;
20  
21  import java.util.Collection;
22  import java.util.HashMap;
23  import java.util.Locale;
24  import java.util.Map;

26  import org.easymock.EasyMock;
27  
28  /**
29   * A base class for testing transformers. This class provides some common
30   * functionality for tests of Transformer implementations. It especially
31   * maintains some mock objects for the transformer context and the validation
32   * message handler. It is possible to expect specific error messages.
33   *
34   * @author Oliver Heger
35   * @version $Id: AbstractTransformerTest.java 205 2012-01-29 18:29:57Z oheger $
36   */
37  public abstract class AbstractTransformerTest
38  {
39      /** Constant for the locale used by the tests. */
40      protected static Locale LOCALE = Locale.GERMAN;
41  
42      /** A mock object for the transformer context. */
43      protected TransformerContext context;
44  
45      /** A mock object for the validation message handler. */
46      private ValidationMessageHandler messageHandler;
47  
48      /** Stores a mock object for a validation message. */
49      private ValidationMessage message;
50  
51      /**
52       * Creates mock objects for the transformer context and the message
53       * validation handler.
54       *
55       * @param props the properties to be returned by the context mock (can be
56       *        <b>null</b>, then an empty map will be used)
57       */
58      protected void setUpContextMock(Map<String, Object> props)
59      {
60          context = EasyMock.createMock(TransformerContext.class);
61          messageHandler = EasyMock.createMock(ValidationMessageHandler.class);
62          EasyMock.expect(context.getValidationMessageHandler()).andStubReturn(
63                  messageHandler);
64          EasyMock.expect(context.properties()).andStubReturn(
65                  (props != null) ? props : new HashMap<String, Object>());
66          EasyMock.expect(context.getLocale()).andStubReturn(LOCALE);
67      }
68  
69      /**
70       * Replays the mock objects.
71       */
72      protected void replayMocks()
73      {
74          EasyMock.replay(context, messageHandler);
75      }
76  
77      /**
78       * Verifies the mock objects.
79       */
80      protected void verifyMocks()
81      {
82          EasyMock.verify(context, messageHandler);
83          if (message != null)
84          {
85              EasyMock.verify(message);
86          }
87      }
88  
89      /**
90       * Initializes the mock objects to expect an error code.
91       *
92       * @param errorKey the key of the error message
93       * @param params additional parameters for the error message
94       */
95      protected void expectError(String errorKey, Object... params)
96      {
97          message = EasyMock.createMock(ValidationMessage.class);
98          EasyMock.expect(message.getKey()).andStubReturn(errorKey);
99          EasyMock.expect(message.getLevel()).andReturn(ValidationMessageLevel.ERROR).anyTimes();
100         EasyMock.replay(message);
101         EasyMock.expect(
102                 messageHandler.getValidationMessage(context, errorKey, params))
103                 .andReturn(message);
104     }
105 
106     /**
107      * Tests whether the specified validation result object contains the
108      * expected error message.
109      *
110      * @param errorKey the error key
111      * @param vr the result object to check
112      */
113     protected void checkError(String errorKey, ValidationResult vr)
114     {
115         assertFalse("Date is valid", vr.isValid());
116         Collection<ValidationMessage> msgs = vr.getValidationMessages();
117         assertEquals("Wrong number of error messages", 1, msgs.size());
118         ValidationMessage msg = msgs.iterator().next();
119         assertEquals("Wrong message key", errorKey, msg.getKey());
120     }
121 }
