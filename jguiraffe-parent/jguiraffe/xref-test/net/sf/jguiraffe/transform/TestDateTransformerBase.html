53      /**
54       * Helper method for comparing the components of a date.
55       *
56       * @param dt the date
57       * @param year the expected year
58       * @param month the expected month
59       * @param day the expected day
60       * @param hour the expected hour
61       * @param min the expected minute
62       * @param sec the expected seconds
63       * @param millis the expected milliseconds
64       */
65      private void checkDate(Date dt, int year, int month, int day, int hour,
66              int min, int sec, int millis)
67      {
68          Calendar cal = Calendar.getInstance();
69          cal.setTime(dt);
70          assertEquals("Wrong year", year, cal.get(Calendar.YEAR));
71          assertEquals("Wrong month", month, cal.get(Calendar.MONTH));
72          assertEquals("Wrong day", day, cal.get(Calendar.DATE));
73          assertEquals("Wrong hour", hour, cal.get(Calendar.HOUR_OF_DAY));
74          assertEquals("Wrong minute", min, cal.get(Calendar.MINUTE));
75          assertEquals("Wrong seconds", sec, cal.get(Calendar.SECOND));
76          assertEquals("Wrong millis", millis, cal.get(Calendar.MILLISECOND));
77      }
