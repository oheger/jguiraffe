16  package net.sf.jguiraffe.di.impl.providers;
17  
18  import java.util.ArrayList;
19  import java.util.Collection;
20  import java.util.Collections;
21  import java.util.HashSet;
22  import java.util.Set;
23  
24  import net.sf.jguiraffe.di.Dependency;
25  import net.sf.jguiraffe.di.DependencyProvider;
26  
27  /**
28   * <p>
29   * An abstract base class for <code>BeanProvider</code> implementations that
30   * create beans derived from collections.
31   * </p>
32   * <p>
33   * This bean provider base implementation can be used when collection beans are
34   * to be created. What makes collections a bit special is the fact that their
35   * elements are also dependencies, which can reference other beans. So this bean
36   * provider has to ensure that all dependencies are registered correctly, so
37   * they can be resolved when the collection is created.
38   * </p>
39   * <p>
40   * This class implements the major part of the functionality required for
41   * creating collection beans. It is initialized with a collection of
42   * dependencies representing the elements of the final collection. These can be
43   * either constant dependencies or references to other beans. In the
44   * <code>getBean()</code> implementation a collection of the correct type is
45   * created, the element dependencies are resolved, and the resulting objects are
46   * added to the collection.
47   * </p>
48   * <p>
49   * A concrete subclass is responsible for creating a collection of a specific
50   * type. For this purpose the <code>createCollection()</code> method has to be
51   * defined. It is invoked by <code>getBean()</code> for obtaining a new
52   * instance of the collection class supported.
53   * </p>
54   * <p>
55   * Bean providers of this type are intended to be used together with a
56   * <code>{@link LifeCycleBeanProvider}</code> that controls the creation of
57   * new instances.
58   * </p>
59   * <p>
60   * Implementation note: Objects of this class are immutable and thus can be
61   * shared between multiple threads.
62   * </p>
63   *
64   * @author Oliver Heger
65   * @version $Id: CollectionBeanProvider.java 205 2012-01-29 18:29:57Z oheger $
66   */
67  public abstract class CollectionBeanProvider extends SimpleBeanProvider
68  {
69      /** Stores the dependencies of this bean provider. */
70      private final Collection<Dependency> dependencies;
71  
72      /**
73       * Creates a new instance of <code>CollectionBeanProvider</code> and
74       * initializes it with a collection with the dependencies of the elements.
75       *
76       * @param deps a collection with the element dependencies (must not be
77       *        <b>null</b>)
78       * @throws IllegalArgumentException if the collection is <b>null</b>
79       */
80      protected CollectionBeanProvider(Collection<Dependency> deps)
81      {
82          if (deps == null)
83          {
84              throw new IllegalArgumentException(
85                      "Dependency collection must not be null!");
86          }
87  
88          dependencies = new ArrayList<Dependency>(deps);
89      }
90  
91      /**
92       * Returns a collection with the dependencies of the collection elements.
93       * From these dependencies the collection will be populated. Note that the
94       * collection returned by this method is immutable.
95       *
96       * @return a collection with the dependencies of the elements
97       */
98      public Collection<Dependency> getElementDependencies()
99      {
100         return Collections.unmodifiableCollection(dependencies);
101     }
102 
103     /**
104      * Returns the bean managed by this provider. This implementation creates a
105      * new collection by calling <code>createCollection()</code>. Then the
106      * dependencies for the elements are resolved and added to the collection.
107      *
108      * @param dependencyProvider the dependency provider
109      * @return the bean managed by this provider
110      */
111     public Object getBean(DependencyProvider dependencyProvider)
112     {
113         Collection<Object> col = createCollection(dependencies.size());
114         for (Dependency dep : dependencies)
115         {
116             col.add(dependencyProvider.getDependentBean(dep));
117         }
118 
119         return col;
120     }
121 
122     /**
123      * Returns the class of the bean created by this provider. This
124      * implementation simply returns <code>Collection.class</code>. We expect
125      * that collection beans are not so specific that they are queried by class.
126      *
127      * @param dependencyProvider the dependency provider
128      * @return the class of the bean created by this provider
129      */
130     public Class<?> getBeanClass(DependencyProvider dependencyProvider)
131     {
132         return Collection.class;
133     }
134 
135     /**
136      * Returns a set with the dependencies of this bean provider. This
137      * implementation creates a new set from the collection with the
138      * dependencies for the elements.
139      *
140      * @return the dependencies of this bean provider
141      */
142     @Override
143     public Set<Dependency> getDependencies()
144     {
145         return new HashSet<Dependency>(dependencies);
146     }
147 
148     /**
149      * Creates the collection. This method is invoked whenever a new bean
150      * instance is to be created. Derived classes must here return a new
151      * instance of the collection supported.
152      *
153      * @param size the initial size of the new collection; here the number of
154      *        elements is passed in
155      * @return the collection
156      */
157     protected abstract Collection<Object> createCollection(int size);
158 }
