16  package net.sf.jguiraffe.gui.builder.action.tags;
17  
18  import net.sf.jguiraffe.gui.builder.action.FormActionException;
19  
20  /**
21   * <p>
22   * Definition of an interface for objects with container facilities.
23   * </p>
24   * <p>
25   * Some of the objects involved in the action builder process act as containers,
26   * e.g. menu bars or toolbars. For certain tags it is necessary to find these
27   * containers so that new elements can be added to them. This is done through
28   * this interface.
29   * </p>
30   * <p>
31   * The interface itself is very simple. It only provides access to a container
32   * object whose concrete type depends on the GUI library specific implementation
33   * of the
34   * <code>{@link net.sf.jguiraffe.gui.builder.action.ActionManager ActionManager}</code>
35   * interface. It is passed to the action manager, which can add further elements
36   * to it.
37   * </p>
38   *
39   * @author Oliver Heger
40   * @version $Id: ActionContainer.java 205 2012-01-29 18:29:57Z oheger $
41   */
42  public interface ActionContainer
43  {
44      /**
45       * Returns the container object. This method will be called when new
46       * elements need to be added to this container.
47       *
48       * @return the container object
49       */
50      Object getContainer();
51  
52      /**
53       * Adds a separator element to this container. This method will be invoked
54       * when a <code>&lt;separator&gt;</code> tag is executed. An
55       * implementation must ensure that the corresponding method of the current
56       * <code>{@link net.sf.jguiraffe.gui.builder.action.ActionManager ActionManager}</code>
57       * for adding the separator is called.
58       * @throws FormActionException if an error occurs
59       */
60      void addSeparator() throws FormActionException;
61  }
