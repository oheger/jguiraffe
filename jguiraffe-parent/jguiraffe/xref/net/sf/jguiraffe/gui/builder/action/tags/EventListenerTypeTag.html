16  package net.sf.jguiraffe.gui.builder.action.tags;
17  
18  import net.sf.jguiraffe.gui.builder.components.FormBuilderException;
19  import net.sf.jguiraffe.gui.builder.components.tags.FormBaseTag;
20  import net.sf.jguiraffe.gui.builder.event.FormListenerType;
21  
22  import org.apache.commons.jelly.JellyTagException;
23  import org.apache.commons.jelly.MissingAttributeException;
24  
25  /**
26   * <p>
27   * A specialized tag handler implementation that allows adding another event
28   * listener type to a {@link FormEventListenerTag}.
29   * </p>
30   * <p>
31   * Per default, tags derived from {@link FormEventListenerTag} support only a
32   * single event listener interface. The tag implemented by this handler class
33   * can be placed in the body of a {@link FormEventListenerTag} to add another
34   * listener interface. The event listener produced by the outer tag will then
35   * support this new listener interface, too. This is useful if multiple event
36   * types fired by a component should be mapped to a single action.
37   * </p>
38   * <p>
39   * This tag handler class can deal with both standard and non-standard events.
40   * For non-standard events the class of the event listener interface must be
41   * specified. The following attributes are supported:
42   * <table border="1">
43   * <tr>
44   * <th>Attribute</th>
45   * <th>Description</th>
46   * <th>Optional</th>
47   * </tr>
48   * <tr>
49   * <td valign="top">type</td>
50   * <td>Defines the name of the event listener type. This can be the name of a
51   * standard type (e.g. <em>change</em> or <em>mouse</em> - case is irrelevant)
52   * or a non-standard type. In the latter case the naming conventions used by the
53   * {@link net.sf.jguiraffe.gui.builder.event.FormEventManager} class apply.</td>
54   * <td valign="top">No</td>
55   * </tr>
56   * <tr>
57   * <td valign="top">listenerClass</td>
58   * <td>Here the class of the event listener interface can be provided. This is
59   * only needed for non-standard listener types. If the {@code type} attribute
60   * contains the name of a standard event type, this attribute is ignored.</td>
61   * <td valign="top">Yes</td>
62   * </tr>
63   * </table>
64   * </p>
65   * <p>
66   * The following example shows how this tag can be used to add an additional
67   * tree expansion listener to a change listener for a tree:
68   *
69   * <pre>
70   * &lt;f:tree name="myTree" model="myTreeModel"&gt;
71   *   &lt;a:changeListener actionName="myAction"&gt;
72   *     &lt;a:listenerType type="Expansion"
73   *       listenerClass="net.sf.jguiraffe.gui.builder.components.model.TreeExpansionLitener"/&gt;
74   *   &lt;/a:changeListener&gt;
75   * &lt;/f:tree&gt;
76   * </pre>
77   *
78   * </p>
79   *
80   * @author Oliver Heger
81   * @version $Id: EventListenerTypeTag.java 205 2012-01-29 18:29:57Z oheger $
82   */
83  public class EventListenerTypeTag extends FormBaseTag
84  {
85      /** The name of the listener type. */
86      private String type;
87  
88      /** The listener class. */
89      private Object listenerClass;
90  
91      /**
92       * Returns the event listener type as a string.
93       *
94       * @return the event listener type
95       */
96      public String getType()
97      {
98          return type;
99      }
100 
101     /**
102      * Set method of the {@code type} attribute.
103      *
104      * @param type the attribute's value
105      */
106     public void setType(String type)
107     {
108         this.type = type;
109     }
110 
111     /**
112      * Returns the event listener class as an object. This attribute accepts
113      * either a class object or the fully-qualified name of a class.
114      *
115      * @return the event listener class
116      */
117     public Object getListenerClass()
118     {
119         return listenerClass;
120     }
121 
122     /**
123      * Set method of the {@code listenerClass} attribute.
124      *
125      * @param listenerClass the attribute's value
126      */
127     public void setListenerClass(Object listenerClass)
128     {
129         this.listenerClass = listenerClass;
130     }
131 
132     /**
133      * Executes this tag. This implementation checks whether the tag is
134      * contained in the body of a {@link FormEventListenerTag}. If this is the
135      * case, an additional event listener type is added to the parent tag.
136      * Otherwise, an exception is thrown.
137      *
138      * @throws JellyTagException if the tag is used incorrectly
139      * @throws FormBuilderException if an error occurs
140      */
141     @Override
142     protected void process() throws JellyTagException, FormBuilderException
143     {
144         if (!(getParent() instanceof FormEventListenerTag))
145         {
146             throw new JellyTagException(
147                     "This tag must be nested inside a FormEventListenerTag!");
148         }
149         if (getType() == null)
150         {
151             throw new MissingAttributeException("type");
152         }
153 
154         FormEventListenerTag parent = (FormEventListenerTag) getParent();
155         FormListenerType listenerType = FormListenerType.fromString(getType());
156         if (listenerType == null)
157         {
158             if (getListenerClass() == null)
159             {
160                 throw new JellyTagException("An event listener class must be "
161                         + "provided for non-standard event type " + getType());
162             }
163             parent.addListenerType(getType(),
164                     convertToClass(getListenerClass()));
165         }
166         else
167         {
168             parent.addListenerType(listenerType);
169         }
170     }
171 }
