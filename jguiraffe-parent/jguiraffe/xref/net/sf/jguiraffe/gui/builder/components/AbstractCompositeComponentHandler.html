16  package net.sf.jguiraffe.gui.builder.components;
17  
18  import java.util.ArrayList;
19  import java.util.Iterator;
20  import java.util.LinkedHashMap;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.Set;
24  
25  import net.sf.jguiraffe.gui.forms.ComponentHandler;
26  
27  import org.apache.commons.lang.ObjectUtils;
28  
29  /**
30   * <p>
31   * An abstract base class for concrete implementations of the
32   * {@link CompositeComponentHandler} interface.
33   * </p>
34   * <p>
35   * This base class already provides functionality for managing a list of child
36   * {@link ComponentHandler} objects. It also has base implementations for most
37   * of the methods defined by the {@link CompositeComponentHandler} interface.
38   * Concrete sub classes mainly have to implement the methods for accessing the
39   * data of this handler. Here the idea is that the data of the child component
40   * handlers must be converted into an object that is the data of this composite
41   * handler. For the base implementations of the other interface methods refer to
42   * the method documentation.
43   * </p>
44   * <p>
45   * Implementation note: This class is not thread-safe.
46   * </p>
47   *
48   * @author Oliver Heger
49   * @version $Id: AbstractCompositeComponentHandler.java 205 2012-01-29 18:29:57Z oheger $
50   * @param <T> the type of the data of this component handler
51   * @param <S> the type of the child handlers
52   */
53  public abstract class AbstractCompositeComponentHandler<T, S> implements
54          CompositeComponentHandler<T, S>
55  {
56      /** A map for storing the child handlers and their names. */
57      private final Map<String, ComponentHandler<S>> childHandlers;
58  
59      /** A list of the child handlers for accessing a child by its index. */
60      private final List<ComponentHandler<S>> childHandlerList;
61  
62      /** The data class. */
63      private final Class<T> dataClass;
64  
65      /**
66       * Creates a new instance of {@code AbstractCompositeComponentHandler} and
67       * initializes it with the data type class.
68       *
69       * @param dataType the type of the data managed by this handler
70       */
71      protected AbstractCompositeComponentHandler(Class<T> dataType)
72      {
73          dataClass = dataType;
74          childHandlers = new LinkedHashMap<String, ComponentHandler<S>>();
75          childHandlerList = new ArrayList<ComponentHandler<S>>();
76      }
77  
78      /**
79       * Adds the specified child {@code ComponentHandler} to this {@code
80       * CompositeComponentHandler}.
81       *
82       * @param name the name of the child handler
83       * @param handler the child handler to be added (must not be <b>null</b>)
84       * @throws IllegalArgumentException if the handler is <b>null</b> or a child
85       *         handler with this name already exists
86       */
87      public void addHandler(String name, ComponentHandler<S> handler)
88      {
89          if (handler == null)
90          {
91              throw new IllegalArgumentException(
92                      "Child handler must not be null!");
93          }
94          if (childHandlers.containsKey(name))
95          {
96              throw new IllegalArgumentException(
97                      "There is already a child handler with this name: " + name);
98          }
99  
100         childHandlers.put(name, handler);
101         childHandlerList.add(handler);
102     }
103 
104     /**
105      * Returns the component this handler is associated with. This base
106      * implementation always returns <b>null</b> because there is no underlying
107      * component.
108      *
109      * @return the associated component
110      */
111     public Object getComponent()
112     {
113         return null;
114     }
115 
116     /**
117      * Returns the outer component this handler is associated with. This base
118      * implementation always returns <b>null</b> because this handler is not
119      * associated with a concrete component.
120      *
121      * @return the outer component
122      */
123     public Object getOuterComponent()
124     {
125         return null;
126     }
127 
128     /**
129      * Returns the data type of this component handler. This base implementation
130      * returns the class that was passed to the constructor.
131      *
132      * @return the data type of this handler
133      */
134     public Class<?> getType()
135     {
136         return dataClass;
137     }
138 
139     /**
140      * Returns a flag whether this {@code ComponentHandler} is enabled. This
141      * implementation returns <b>true</b> if and only if all child handlers are
142      * enabled.
143      *
144      * @return a flag whether this {@code ComponentHandler} is enabled
145      */
146     public boolean isEnabled()
147     {
148         for (ComponentHandler<?> ch : childHandlerList)
149         {
150             if (!ch.isEnabled())
151             {
152                 return false;
153             }
154         }
155 
156         return true;
157     }
158 
159     /**
160      * Changes the enabled state of this {@code ComponentHandler}. This
161      * implementation calls {@code setEnabled()} on all child handlers passing
162      * in the argument.
163      *
164      * @param f the new enabled state
165      */
166     public void setEnabled(boolean f)
167     {
168         for (ComponentHandler<?> ch : childHandlerList)
169         {
170             ch.setEnabled(f);
171         }
172     }
173 
174     /**
175      * Returns the number of child {@code ComponentHandler}s that have been
176      * added to this {@code CompositeComponentHandler}.
177      *
178      * @return the number of child handlers
179      */
180     public final int getChildHandlerCount()
181     {
182         return childHandlerList.size();
183     }
184 
185     /**
186      * Returns a list with all child handlers managed by this {@code
187      * CompositeComponentHandler}. Derived classes can call this method to
188      * access the child handlers. Note: the list must not be modified.
189      *
190      * @return a list with all child handlers
191      */
192     protected List<ComponentHandler<S>> getChildHandlers()
193     {
194         return childHandlerList;
195     }
196 
197     /**
198      * Returns a set with the names of the child handlers that have been added
199      * to this handler. The order of the names in this set corresponds to the
200      * order in which the child handlers have been added. Note: the set must not
201      * be modified.
202      *
203      * @return a set with the names of the child handlers
204      */
205     protected Set<String> getChildHandlerNames()
206     {
207         return childHandlers.keySet();
208     }
209 
210     /**
211      * Returns the child handler with the specified name. If the handler cannot
212      * be found, <b>null</b> is returned.
213      *
214      * @param name the name of the child handler in question
215      * @return the corresponding child {@code ComponentHandler} or <b>null</b>
216      */
217     protected ComponentHandler<S> getChildHandler(String name)
218     {
219         return childHandlers.get(name);
220     }
221 
222     /**
223      * Returns the index for the child handler with the given name. Indices are
224      * 0-based. If the name cannot be resolved, this method returns -1.
225      *
226      * @param name the name of the child handler in question
227      * @return the index of this child handler
228      */
229     protected int getChildHandlerIndex(String name)
230     {
231         int index = 0;
232         for (String handlerName : childHandlers.keySet())
233         {
234             if (ObjectUtils.equals(name, handlerName))
235             {
236                 return index;
237             }
238             index++;
239         }
240 
241         return -1;
242     }
243 
244     /**
245      * Returns the name of the child handler at the specified index. The index
246      * can run from 0 to the number of child handlers minus 1. The handlers are
247      * indexed in the order they have been added to this {@code
248      * CompositeComponentHandler}. If the index passed in is invalid, this
249      * method returns <b>null</b>.
250      *
251      * @param index the index of the desired child handler
252      * @return the corresponding child {@code ComponentHandler} or <b>null</b>
253      */
254     protected String getChildHandlerNameAt(int index)
255     {
256         if (index < 0 || index >= getChildHandlerCount())
257         {
258             return null;
259         }
260 
261         Iterator<String> it = childHandlers.keySet().iterator();
262         for (int i = 0; i < index; i++)
263         {
264             it.next();
265         }
266         return it.next();
267     }
