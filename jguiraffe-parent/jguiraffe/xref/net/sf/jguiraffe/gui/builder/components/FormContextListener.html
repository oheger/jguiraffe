16  package net.sf.jguiraffe.gui.builder.components;
17  
18  import net.sf.jguiraffe.gui.forms.Form;
19  
20  /**
21   * <p>
22   * An event listener interface to be implemented by components that are
23   * interested in the creation of new form contexts during a builder operation.
24   * </p>
25   * <p>
26   * Some complex components (e.g. tables) require the creation of sub forms
27   * during their construction. Typically, such sub forms require special
28   * treatment. Therefore, it may be important for affected components (like a
29   * concrete {@link ComponentManager} implementation) to receive a notification
30   * when a form context is created or closed. This interface defines a set of
31   * callback methods allowing such notifications to be passed.
32   * </p>
33   * <p>
34   * Objects implementing this interface can register themselves at a
35   * {@link ComponentBuilderData} object. They are then notified about changes of
36   * the current form context during a builder operation.
37   * </p>
38   *
39   * @author Oliver Heger
40   * @version $Id$
41   * @since 1.3
42   */
43  public interface FormContextListener
44  {
45      /**
46       * Notifies this object that a new form context was created. This new form
47       * context becomes the current context; so all newly created components are
48       * added to this form. The source object responsible for the context
49       * creation is also passed to this method - it may be <b>null</b> if no
50       * information about the source is available. In most cases, this source
51       * will be a tag.
52       *
53       * @param form the {@code Form} object associated with the new context
54       * @param source the source which created the new form context (if known)
55       */
56      void formContextCreated(Form form, Object source);
57  
58      /**
59       * Notifies this object that the current form context was closed. Each call
60       * to {@link #formContextCreated(Form, Object)} is followed eventually by an
61       * invocation of this method.
62       *
63       * @param form the {@code Form} object associated with the new context
64       * @param source the source responsible for this form context (if known)
65       */
66      void formContextClosed(Form form, Object source);
67  }
