16  package net.sf.jguiraffe.gui.builder.components.model;
17  
18  import net.sf.jguiraffe.gui.builder.event.FormEventListener;
19  
20  /**
21   * <p>
22   * An event listener interface to be implemented by objects that need to be
23   * notified before the expansion state of a tree's node changes.
24   * </p>
25   * <p>
26   * This interface is very similar in usage and purpose as the
27   * <code>{@link TreeExpansionListener}</code> interface. There are two main
28   * differences:
29   * <ul>
30   * <li>Event listeners are notified before the affected node is actually
31   * expanded or collapsed. This makes it possible for instance to implement a
32   * dynamic loading mechanism: The first time a node is expanded the listener can
33   * check whether further data is available. It can then add the data found to
34   * the tree.</li>
35   * <li>Event listeners have the opportunity to veto against the expand or
36   * collapse operation. This can be done by throwing an exception in the listener
37   * method.</li>
38   * </ul>
39   * </p>
40   *
41   * @author Oliver Heger
42   * @version $Id: TreePreExpansionListener.java 205 2012-01-29 18:29:57Z oheger $
43   */
44  public interface TreePreExpansionListener extends FormEventListener
45  {
46      /**
47       * A node of a tree is about to change its expansion state. The passed in
48       * event contains all information available about the objects involved. The
49       * listener can permit this operation or, by throwing an exception, abort
50       * it.
51       *
52       * @param event the event
53       * @throws TreeExpandVetoException thrown to indicate that this operation
54       *         should be aborted
55       */
56      void beforeExpansionStateChange(TreeExpansionEvent event)
57              throws TreeExpandVetoException;
58  }
