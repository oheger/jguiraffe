18  import net.sf.jguiraffe.di.ConversionHelper;
19  import net.sf.jguiraffe.gui.builder.di.DIBuilderData;
20  import net.sf.jguiraffe.gui.builder.di.tags.ClassDescData;

22  import org.apache.commons.beanutils.Converter;
23  import org.apache.commons.jelly.JellyTagException;

25  /**
26   * <p>
27   * A specialized tag handler class for adding custom data type converters in a
28   * builder script.
29   * </p>
30   * <p>
31   * The <em>dependency injection framework</em> supports data type converters
32   * that are automatically invoked when methods on beans are invoked or
33   * properties are set to convert data to the target types required by the method
34   * signatures. There are multiple ways to extent the default converters provided
35   * by the framework. Using this tag it is possible to define custom type
36   * converters directly in a builder script. This is pretty convenient because it
37   * does not require any specific glue code when invoking the builder.
38   * </p>
39   * <p>
40   * The basic idea is that custom data type converters are defined as regular
41   * beans using variants of the {@code <di:bean>} tag. Then this tag is used to
42   * register these beans as converters at the current builder context. Because
43   * {@code ConverterTag} extends {@link UseBeanBaseTag} multiple ways for
44   * referencing beans are supported; it is even possible to create new bean
45   * instances by calling their standard constructor.
46   * </p>
47   * <p>
48   * The scope of the data type converters defined by this tag should determine
49   * the scripts in which to place the tag. If the converters are very basic and
50   * used in many builder scripts of the application, they should be already
51   * registered in the script defining the main window - i.e. the first builder
52   * script executed by the application. There is a kind of an inheritance
53   * mechanism: data type converters registered for a root {@code BeanStore} are
54   * also available in builder scripts defining children of this root store.
55   * Specialized converters that are only required for a single builder script can
56   * be defined locally in this script.
57   * </p>
58   * <p>
59   * Of course, all attributes supported by the base class are also available for
60   * {@code ConverterTag}. The following table lists the additional attributes
61   * introduced by this class:
62   * </p>
63   * <p>
64   * <table border="1">
65   * <tr>
66   * <th>Attribute</th>
67   * <th>Description</th>
68   * <th>Optional</th>
69   * </tr>
70   * <tr>
71   * <td valign="top">converterTargetClass</td>
72   * <td>With this attribute the target class of the converter can be specified
73   * (either as a {@code java.lang.Class} object or as a string that will be
74   * converted to a class.</td>
75   * <td valign="top">Yes</td>
76   * </tr>
77   * <tr>
78   * <td valign="top">converterTargetClassName</td>
79   * <td>Using this attribute the name of the target class can be specified. While
80   * the {@code targetClass} attribute requires that the class specified can be
81   * loaded using the default class loader, this attribute - together with the
82   * {@code targetClassLoader} attribute - supports loading the class from an
83   * alternative class loader.</td>
84   * <td valign="top">Yes</td>
85   * </tr>
86   * <tr>
87   * <td valign="top">converterTargetClassLoader</td>
88   * <td>This attribute is only evaluated if the {@code targetClassName} attribute
89   * is set. In this case it references the class loader to be used for resolving
90   * the class name. A class loader with the given name is queried from the
91   * current {@link net.sf.jguiraffe.di.ClassLoaderProvider ClassLoaderProvider}.</td>
92   * <td valign="top">Yes</td>
93   * </tr>
94   * <tr>
95   * <td valign="top">isBaseClassConverter</td>
96   * <td>Determines whether the converter should be registered as a normal or a
97   * base class converter. If this attribute has an arbitrary value, it is
98   * registered as base class converter.</td>
99   * <td valign="top">Yes</td>
100  * </tr>
101  * </table>
102  * </p>
103  * <p>
104  * The target class must have been specified in one of the supported ways,
105  * otherwise an exception is thrown. The following example shows a typical usage
106  * scenario for this tag:
107  *
108  * <pre>
109  * &lt;!-- Definition of the bean for the converter.--&gt;
110  * &lt;di:bean name="converterBean" beanClass="com.mypackage.MyConverter"&gt;
111  *   ...
112  * &lt;/di:bean&gt;
113  *
114  * &lt;!-- Register the bean as converter.--&gt;
115  * &lt;f:converter beanName="converterBean"
116  *   converterTargetClass="com.mypackage.MyDataClass"/&gt;
117  * </pre>
118  *
119  * After this declaration bean definitions can be placed in the Jelly script
120  * with properties or method parameters of the type
121  * {@code com.mypackage.MyDataClass}. The newly registered converter will be
122  * automatically called to perform the data conversion to this type.
123  * </p>
124  *
125  * @author Oliver Heger
126  * @version $Id: ConverterTag.java 205 2012-01-29 18:29:57Z oheger $
127  */
128 public class ConverterTag extends UseBeanBaseTag
129 {
130     /** Constant for the name of the converter target class attribute. */
131     protected static final String ATTR_CONVERTER_TARGET_CLASS =
132             "converterTargetClass";
133 
134     /** Constant for the name of the converter target class name attribute. */
135     protected static final String ATTR_CONVERTER_TARGET_CLASS_NAME =
136             "converterTargetClassName";
137 
138     /** Constant for the name of the converter target class loader attribute. */
139     protected static final String ATTR_CONVERTER_TARGET_CLASS_LOADER =
140             "converterTargetClassLoader";
141 
142     /** Constant for the name of the base class converter attribute. */
143     protected static final String ATTR_BASE_CLASS_CONVERTER =
144             "isBaseClassConverter";
145 
146     /**
147      * Creates a new instance of {@code ConverterTag}.
148      */
149     public ConverterTag()
150     {
151         super(null, Converter.class);
152         addIgnoreProperties(ATTR_BASE_CLASS_CONVERTER,
153                 ATTR_CONVERTER_TARGET_CLASS,
154                 ATTR_CONVERTER_TARGET_CLASS_LOADER,
155                 ATTR_CONVERTER_TARGET_CLASS_NAME);
156     }
157 
158     /**
159      * Processes the bean created by this tag. This implementation registers the
160      * specified bean as a converter at the central {@link ConversionHelper}
161      * object. A target class for the converter must have been provided,
162      * otherwise an exception is thrown.
163      *
164      * @param bean the newly created bean
165      * @return a flag whether the bean could be processed
166      * @throws JellyTagException if an error occurs
167      */
168     @Override
169     protected boolean passResults(Object bean) throws JellyTagException
170     {
171         ClassDescData cdd = createTargetClassDescription();
172         Class<?> convTargetClass = fetchTargetClass(cdd);
173         boolean baseClassConv =
174                 getAttributeStr(ATTR_BASE_CLASS_CONVERTER) != null;
175 
176         if (baseClassConv)
177         {
178             fetchConversionHelper().registerBaseClassConverter(
179                     (Converter) bean, convTargetClass);
180         }
181         else
182         {
183             fetchConversionHelper().registerConverter((Converter) bean,
184                     convTargetClass);
185         }
186 
187         return true;
188     }
189 
190     /**
191      * Creates a {@link ClassDescData} object for the target class of the
192      * converter. This method evaluates the several attributes that can be used
193      * to specify the converter's target class. From these attributes the
194      * {@link ClassDescData} object is populated.
195      *
196      * @return the initialized {@link ClassDescData} object
197      */
198     protected ClassDescData createTargetClassDescription()
199     {
200         ClassDescData cdd = new ClassDescData();
201         cdd.setClassLoaderName(getAttributeStr(ATTR_CONVERTER_TARGET_CLASS_LOADER));
202         cdd.setTargetClassName(getAttributeStr(ATTR_CONVERTER_TARGET_CLASS_NAME));
203         cdd.setTargetClass(evaluateTargetClassAttribute());
204         return cdd;
205     }
206 
207     /**
208      * Obtains the target class of the converter from the specified
209      * {@link ClassDescData} object. This method is called by
210      * {@link #passResults(Object)}.
211      *
212      * @param cdd the description object populated from the attributes of this
213      *        tag
214      * @return the target class of the converter
215      * @throws JellyTagException if the class cannot be resolved
216      */
217     protected Class<?> fetchTargetClass(ClassDescData cdd)
218             throws JellyTagException
219     {
220         DIBuilderData diData = fetchDIBuilderData();
221         return cdd.resolveClass(diData.getClassLoaderProvider());
222     }
223 
224     /**
225      * Helper method for obtaining the current {@link DIBuilderData} object from
226      * the Jelly context.
227      *
228      * @return the {@link DIBuilderData} object
229      */
230     private DIBuilderData fetchDIBuilderData()
231     {
232         return DIBuilderData.get(getContext());
233     }
234 
235     /**
236      * Returns the target class that was specified using the target class
237      * attribute (if any). This implementation returns the value of this
238      * attribute and converts it if necessary to a Class object. Result is
239      * <b>null</b> if the attribute was not specified.
240      *
241      * @return the class obtained from the target class attribute
242      */
243     private Class<?> evaluateTargetClassAttribute()
244     {
245         Object clsAttr = getAttributes().get(ATTR_CONVERTER_TARGET_CLASS);
246         return fetchConversionHelper().convert(Class.class, clsAttr);
247     }
248 
249     /**
250      * Helper method for retrieving the current {@link ConversionHelper} object
251      * from the Jelly context.
252      *
253      * @return the conversion helper object
254      */
255     private ConversionHelper fetchConversionHelper()
256     {
257         DIBuilderData diData = fetchDIBuilderData();
258         return diData.getInvocationHelper().getConversionHelper();
259     }
260 }
