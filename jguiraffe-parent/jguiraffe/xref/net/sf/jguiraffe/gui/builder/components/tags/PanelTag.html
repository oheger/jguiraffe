28  /**
29   * <p>
30   * A specific container tag implementation that creates a panel.
31   * </p>
32   * <p>
33   * This container tag will create the most simple container. The following table

35   * simple components.

39   * <tr>

43   * </tr>
44   * <tr>
45   * <td valign="top">text</td>
46   * <td>Defines a text for the panel. This text will be displayed as a caption

48   * </td valign="top">yes</td>
49   * </tr>
50   * <tr>
51   * <td valing="top">textres</td>

65   * definitions in a format supported by the

102 public class PanelTag extends ContainerTag
103 {

105     private TextData textData;

251     }

261     }
262 
263     /**
264      * Creates the concrete container widget. This implementation creates a
265      * panel using the component manager object.
266      *
267      * @param manager the manager
268      * @param create the create flag
269      * @param components a collection with the container's children
270      * @return the newly created container
271      * @throws FormBuilderException if an error occurs
272      * @throws JellyTagException if the tag is used incorrectly
273      */
274     @Override
275     protected Object createContainer(ComponentManager manager, boolean create,
276             Collection<Object[]> components) throws FormBuilderException,
277             JellyTagException
278     {
279         if (create)
280         {
281             // Initialize some internal fields
282             if (StringUtils.isNotEmpty(textFontRef))
283             {
284                 Object font = getContext().getVariable(textFontRef);
285                 if (font == null)
286                 {
287                     throw new FormBuilderException(
288                             "Cannot find font with name " + textFontRef);
289                 }
290                 setTitleFont(font);
291             }
292 
293             color = ColorHelper.resolveColor(textColor);
294         }
295 
296         return manager.createPanel(this, create);
297     }
