16  package net.sf.jguiraffe.gui.builder.components.tags;

18  import net.sf.jguiraffe.gui.layout.CellAlignment;
19  import net.sf.jguiraffe.gui.layout.CellConstraints;
20  
21  import org.apache.commons.jelly.JellyContext;
22  import org.apache.commons.jelly.JellyTagException;
23  import org.apache.commons.jelly.MissingAttributeException;
24  import org.apache.commons.lang.StringUtils;
25  
26  /**
27   * <p>
28   * An abstract base class for tags that define cell constraints of a percent
29   * layout.
30   * </p>
31   * <p>
32   * Sub classes of this tag handler class can appear in the body of a
33   * {@link PercentLayoutTag} tag to define constraints for rows or
34   * columns. Each of these tags fully defines one
35   * {@link net.sf.jguiraffe.gui.layout.CellConstraints CellConstraints}
36   * object. The constraints (as strings) are specified using the
37   * <code>constr</code> attribute. The text value of this attribute must be
38   * understandable by the <code>CellConstraints</code> class. This is an
39   * alternative for defining all cell constraints in a single string.
40   * </p>
41   * <p>
42   * This base class implements the major functionality. Sub classes have to deal
43   * with creating the correct constraints object and passing it to the tag
44   * defining the percent layout.
45   * </p>
46   *
47   * @author Oliver Heger
48   * @version $Id: PercentCellConstraintsTag.java 205 2012-01-29 18:29:57Z oheger $
49   */
50  public abstract class PercentCellConstraintsTag extends FormBaseTag
51  {
52      /** Stores the value of the constr attribute. */
53      private String constr;
54  
55      /**
56       * Returns the constraints as string.
57       *
58       * @return the constraints
59       */
60      public String getConstr()
61      {
62          return constr;
63      }
64  
65      /**
66       * Setter method for the constr attribute.
67       *
68       * @param constr the attribute value
69       */
70      public void setConstr(String constr)
71      {
72          this.constr = constr;
73      }
74  
75      /**
76       * Executes this tag. Calls the <code>createConstraints()</code> and
77       * <code>addConstraints()</code> methods to create and initialize the
78       * constraints object.
79       *
80       * @throws JellyTagException if no parent tag can be found
81       */
82      @Override
83      protected void process() throws JellyTagException
84      {
85          PercentLayoutTag parent =
86                  (PercentLayoutTag) findAncestorWithClass(PercentLayoutTag.class);
87          if (parent == null)
88          {
89              throw new JellyTagException(
90                      "PercentCellConstraintsTag must be nested inside a PercentLayout tag!");
91          }
92  
93          if (StringUtils.isEmpty(getConstr()))
94          {
95              throw new MissingAttributeException("constr");
96          }
97  
98          addConstraints(parent, createConstraints());
99      }
100 
101     /**
102      * Returns the {@code CellConstraints.Builder} for creating {@code
103      * CellConstraints} objects with the given name. Constraints builders are
104      * stored in the Jelly context, one for each type of constraints (rows and
105      * columns). They are created on demand and then shared between all tags
106      * that need to create constraints objects. Note: As the Jelly context is
107      * confined to a single thread, this is no problem, and no synchronization
108      * is needed.
109      *
110      * @param context the Jelly context
111      * @param name the name of the variable which stores the builder
112      * @param defAlignment the default alignment of the builder
113      * @return the {@code CellConstraints.Builder}
114      */
115     protected static CellConstraints.Builder getConstraintsBuilder(
116             JellyContext context, String name, CellAlignment defAlignment)
117     {
118         CellConstraints.Builder builder = (CellConstraints.Builder) context
119                 .getVariable(name);
120         if (builder == null)
121         {
122             builder = new CellConstraints.Builder();
123             builder.setDefaultAlignment(defAlignment);
124             context.setVariable(name, builder);
125         }
126 
127         return builder;
128     }
129 
130     /**
131      * Creates the constraints object for the string defined by the user. This
132      * method must be defined in a concrete sub class to either return a column
133      * or a row constraints object.
134      *
135      * @return the constraints
136      */
137     protected abstract CellConstraints createConstraints();
138 
139     /**
140      * Passes the newly created constraints object to the corresponding percent
141      * layout tag. This method must be defined by concrete sub classes to call
142      * the correct setter method of the parent tag.
143      *
144      * @param parent the percent layout tag
145      * @param c the constraints object
146      */
147     protected abstract void addConstraints(PercentLayoutTag parent,
148             CellConstraints c);
149 }
