32  /**
33   * <p>
34   * A tag that creates a tree component.
35   * </p>
36   * <p>
37   * With this tag a typical tree view for displaying data organized in a
38   * hierarchical manner can be created. Per default a tree view is not directly
39   * used for entering data into a form. Often only the selection of a tree is
40   * relevant, e.g. for identifying an object to be edited. Therefore the
41   * component handler created by this tag (a
42   * {@link net.sf.jguiraffe.gui.builder.components.model.TreeHandler TreeHandler}
43   * by the way) is not added to the form per default. This can be changed by
44   * setting the <code>noField</code> attribute to <b>false</b>. In this respect
45   * this tag works analogously to the tag for creating tables.
46   * </p>
47   * <p>
48   * The content of the tree is provided by a model which must be specified to
49   * this tag. The <em>JGUIraffe</em> framework uses
50   * <code>HierarchicalConfiguration</code> objects as tree models. With the
51   * <code>model</code> attribute the name of the model variable can be set. The
52   * tag will lookup this variable in the current <code>BeanContext</code>, i.e.
53   * the whole power of the dependency injection framework is available for
54   * defining the model data.
55   * </p>
56   * <p>
57   * The following table lists the attributes supported by the
58   * <code>TreeTag</code> tag handler class:
59   * <table border="1">
60   * <tr>
61   * <th>Attribute</th>
62   * <th>Description</th>
63   * <th>Optional</th>
64   * </tr>
65   * <tr>
66   * <td valign="top">model</td>
67   * <td>With this attribute the model of the tree is specified. Here the name of
68   * a variable that can be found in the current <code>BeanContext</code> must be
69   * provided. This variable must be a <code>HierarchicalConfiguration</code>
70   * object.</td>
71   * <td valign="top">No</td>
72   * </tr>
73   * <tr>
74   * <td valign="top">editable</td>
75   * <td>This boolean property determines whether the tree is read-only or can be
76   * edited. The default value is <b>false</b>. If set to <b>true</b>, the nodes
77   * of the tree can be edited, which will also update the tree's model.</td>
78   * <td valign="top">Yes</td>
79   * </tr>
80   * <tr>
81   * <td valign="top">multiSelection</td>
82   * <td>This boolean property indicates whether the tree should support
83   * multi-selection, i.e. multiple nodes can be selected at the same time. If the
84   * attribute is missing, <b>false</b> is the default value.</td>
85   * <td valign="top">Yes</td>
86   * </tr>
87   * <tr>
88   * <td valign="top">rootVisible</td>
89   * <td>With this boolean attribute the presentation of the tree's root node can
90   * be determined. If set to <b>true</b> (which is the default), the root node is
91   * visible. Setting this attribute to <b>false</b> the root node won't be drawn.
92   * </td>
93   * <td valign="top">Yes</td>
94   * </tr>
95   * <tr>
96   * <td valign="top">iconHandler</td>
97   * <td>Using this attribute a custom {@link TreeIconHandler} can be provided. A
98   * bean with the name specified here will be looked up in the current bean
99   * context. It must implement the {@link TreeIconHandler} interface. This bean
100  * will then be used for determining the icons to be displayed for the single
101  * tree nodes. If this attribute is undefined, a default icon handler will be
102  * used.</td>
103  * <td valign="top">Yes</td>
104  * </tr>
105  * <tr>
106  * <td valing="top">scrollWidth</td>
107  * <td>Here the preferred width of the scroll pane enclosing the tree can be
108  * specified as a number with unit (e.g. &quot;1.5cm&quot;). If specified, the
109  * scroll pane will have exactly this preferred width. Otherwise, the width is
110  * determined by the preferred width of the tree.</td>
111  * <td valign="top">yes</td>
112  * </tr>
113  * <tr>
114  * <td valing="top">scrollHeight</td>
115  * <td>Here the preferred height of the scroll pane enclosing the tree can be
116  * specified as a number with unit (e.g. &quot;10dlu&quot;). If specified, the
117  * scroll pane will have exactly this preferred height. Otherwise, the height is
118  * determined by the preferred height of the tree.</td>
119  * <td valign="top">yes</td>
120  * </tr>
121  * </table>
122  * </p>
123  * <p>
124  * With {@link TreeIconTag} tags placed in the body of this tag the icons
125  * displayed by the tree for its nodes can be specified. You can either override
126  * the default icons for leaf and (expanded or collapsed) branch nodes or
127  * specify a custom {@link TreeIconHandler} object using the
128  * <code>iconHandler</code> attribute that is responsible for drawing the icons.
129  * </p>
130  *
131  * @author Oliver Heger
132  * @version $Id: TreeTag.java 205 2012-01-29 18:29:57Z oheger $
133  */

150     /**
151      * Constant for the default icon handler. This object is used when no custom
152      * handler is set.
153      */
154     private static final TreeIconHandler DEF_ICON_HANDLER = new TreeIconHandler()
155     {
156         /*
157          * Returns the name of the icon to use for the specified node. This
158          * implementation only evaluates the expanded and leaf flags and
159          * returns one of three default icon names (one of the <code>ICON_XXX</code> constants).
160          */
161         public String getIconName(ConfigurationNode node, boolean expanded,
162                 boolean leaf)
163         {
164             if (leaf)
165             {
166                 return ICON_LEAF;
167             }
168             else
169             {
170                 return expanded ? ICON_BRANCH_EXPANDED : ICON_BRANCH_COLLAPSED;
171             }
172         }
173     };

390     /**
391      * Adds an icon to this tree. The icon is given a unique name, under which
392      * it can be accessed by the tree's <code>{@link TreeIconHandler}</code>.
393      * The default icon handler (which will be used when no custom handler is
394      * specified using the <code>iconHandler</code> attribute) uses the
395      * following names for icons:
396      * <ul>
397      * <li>{@value #ICON_BRANCH_EXPANDED} for expanded branch nodes</li>
398      * <li>{@value #ICON_BRANCH_COLLAPSED} for collapsed branch nodes and</li>
399      * <li>{@value #ICON_LEAF} for leaf nodes</li>
400      * </ul>
401      * By adding icons with these names the default icons of the tree can be
402      * changed. However, if a custom icon handler is used, it is up to this
403      * object which icons it supports.
404      *
405      * @param name the name of the icon (must not be <b>null</b>)
406      * @param icon the icon (must not be <b>null</b>)
407      * @throws IllegalArgumentException if a required parameter is missing
408      */
409     public void addIcon(String name, Object icon)
410     {
411         if (name == null)
412         {
413             throw new IllegalArgumentException("Icon name must not be null");
414         }
415         if (icon == null)
416         {
417             throw new IllegalArgumentException("Icon must not be null!");
418         }
419 
420         icons.put(name, icon);
421     }

453     /**
454      * Performs pre-processing before the body of this tag is executed. This
455      * implementation tries to resolve the icon handler if one is specified.
456      *
457      * @throws FormBuilderException if an error occurs
458      * @throws JellyTagException if the tag is incorrectly used
459      * @throws net.sf.jguiraffe.di.InjectionException if a bean cannot be resolved
460      */
461     @Override
462     protected void processBeforeBody() throws JellyTagException,
463             FormBuilderException
464     {
465         super.processBeforeBody();
466 
467         iconHandlerImplementation =
468                 (getIconHandler() != null) ? resolveIconHandler(getIconHandler())
469                         : createDefaultIconHandler();
470     }

472     /**
473      * Creates the component handler for the tree component. This implementation
474      * will check whether a tree model is defined and obtain it from the bean
475      * context if necessary. Then it will delegate to the
476      * <code>ComponentManager</code> for actually creating the component.
477      *
478      * @param manager the component manager
479      * @param create the create flag
480      * @return the component handler for the newly created component
481      * @throws FormBuilderException if an error occurs
482      * @throws JellyTagException if the tag is used incorrectly
483      * @throws net.sf.jguiraffe.di.InjectionException if the model bean cannot
484      *         be resolved
485      */
486     @Override
487     protected ComponentHandler<?> createComponentHandler(
488             ComponentManager manager, boolean create)
489             throws FormBuilderException, JellyTagException
490     {
491         if (create)
492         {
493             preferredScrollWidth = convertToNumberWithUnit(scrollWidth,
494                     NumberWithUnit.ZERO);
495             preferredScrollHeight = convertToNumberWithUnit(scrollHeight,
496                     NumberWithUnit.ZERO);
497         }
498         else
499         {
500             initTreeModel();
501         }
502 
503         return manager.createTree(this, create);
504     }
