16  package net.sf.jguiraffe.gui.builder.components.tags.table;
17  
18  /**
19   * <p>
20   * Definition of an interface for a component which can recalibrate itself based
21   * on the column widths of a table.
22   * </p>
23   * <p>
24   * This interface is used to access functionality provided by
25   * {@link TableColumnWidthController} without having to reference the full
26   * object. So if a client just needs this recalibration, it can use this
27   * interface.
28   * </p>
29   *
30   * @author Oliver Heger
31   * @version $Id$
32   * @since 1.3
33   */
34  public interface TableColumnRecalibrator
35  {
36      /**
37       * Recalibrates the internally stored column sizes. This method is intended
38       * to be called if there is an external change in the sizes of the columns
39       * managed by this object, for instance if the user manually changed a
40       * column width. In this case all current column widths have to be passed to
41       * this method. The method then adjusts the sizes of the columns with fixed
42       * widths and re-calculates the percent values of the other columns.
43       *
44       * @param columnSizes an array with the new sizes of the managed columns
45       * @throws IllegalArgumentException if the array passed to this method is
46       *         <b>null</b> or does not have the expected number of elements
47       */
48      void recalibrate(int[] columnSizes);
49  }
