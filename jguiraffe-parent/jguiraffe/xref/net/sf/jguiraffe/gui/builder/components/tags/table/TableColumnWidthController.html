21  /**
22   * <p>
23   * A helper class that provides functionality for managing the widths of the
24   * columns of a table.
25   * </p>
26   * <p>
27   * An instance of this class is created and initialized by {@link TableTag}. It
28   * can be used by platform-specific table implementations. It helps storing the
29   * column widths of a table and provides some helper methods for calculating
30   * widths if columns with a relative width are involved. Because this
31   * functionality is not platform-specific it is collected in this helper class.
32   * </p>
33   * <p>
34   * Implementation note: This class is not thread-safe.
35   * </p>
36   *
37   * @author Oliver Heger
38   * @version $Id: TableColumnWidthController.java 205 2012-01-29 18:29:57Z oheger $
39   */
40  public final class TableColumnWidthController implements
41          TableColumnRecalibrator, TableColumnWidthCalculator
42  {
43      /** Constant for percent calculations. */
44      private static final double PERCENT = 100;
45  
46      /** An array with the original fixed widths of the columns. */
47      private final NumberWithUnit[] originalWidths;
48  
49      /** An array with the current fixed widths. */
50      private final int[] fixedWidths;
51  
52      /** Stores the current percent values. */
53      private final double[] percentValues;
54  
55      /** The number of columns with a percent width. */
56      private final int numberOfColumnsWithPercentWidth;
57  
58      /**
59       * Creates a new instance of {@code TableColumnWidthController}. This
60       * constructor is used internally. Clients create instances using the static
61       * factory method.
62       *
63       * @param orgWidths an array with the original width definitions for columns
64       *        with fixed width
65       * @param percents the current percent widths for all columns
66       * @param percentCnt the number of columns with a percent width
67       */
68      private TableColumnWidthController(NumberWithUnit[] orgWidths,
69              double[] percents, int percentCnt)
70      {
71          originalWidths = orgWidths;
72          percentValues = percents;
73          numberOfColumnsWithPercentWidth = percentCnt;
74          fixedWidths = new int[orgWidths.length];
75      }
76  
77      /**
78       * Creates a new instance of {@link TableColumnWidthController} and
79       * initializes it with the information provided by the given {@code
80       * TableTag}. This factory method first checks whether the widths of the
81       * columns managed by the table tag are valid; if not, an exception is
82       * thrown. Then an instance is created and initialized with the column
83       * widths.
84       *
85       * @param tt the {@code TableTag} (must not be <b>null</b>)
86       * @return a new instance of this class
87       * @throws FormBuilderException if the widths of the table's columns are
88       *         invalid
89       * @throws IllegalArgumentException if the tag is <b>null</b>
90       */
91      public static TableColumnWidthController newInstance(TableTag tt)
92              throws FormBuilderException
93      {
94          if (tt == null)
95          {
96              throw new IllegalArgumentException("TableTag must not be null!");
97          }
98  
99          NumberWithUnit[] widths = new NumberWithUnit[tt.getColumnCount()];
100         int sumPercents = 0;
101         int percentCnt = 0;
102         int undefPercentCnt = 0;
103 
104         for (int i = 0; i < widths.length; i++)
105         {
106             TableColumnTag colTag = tt.getColumn(i);
107             checkColumnWidthSpec(colTag);
108             if (isPercentWidth(colTag))
109             {
110                 sumPercents += colTag.getPercentWidth();
111                 percentCnt++;
112                 if (colTag.getPercentWidth() == 0)
113                 {
114                     undefPercentCnt++;
115                 }
116             }
117             else
118             {
119                 widths[i] = colTag.getColumnWidth();
120             }
121         }
122 
123         return new TableColumnWidthController(widths,
124                 calculateInitialPercentValues(tt, percentCnt, sumPercents,
125                         undefPercentCnt), percentCnt);
126     }

216     /**
217      * {@inheritDoc} This implementation evaluates the current widths of columns
218      * with fixed or percent widths.
219      */
220     public int[] calculateWidths(int totalSize)
221     {
222         int[] widths = new int[getColumnCount()];
223         int sum = 0;
224 
225         // handle columns with a fixed width
226         for (int i = 0; i < widths.length; i++)
227         {
228             if (!isPercentWidth(i))
229             {
230                 widths[i] = fixedWidths[i];
231                 sum += fixedWidths[i];
232             }
233         }
234 
235         // handle columns with a percent width
236         int remaining = totalSize - sum;
237         if (remaining > 0)
238         {
239             for (int i = 0; i < widths.length; i++)
240             {
241                 if (isPercentWidth(i))
242                 {
243                     widths[i] = roundInt(remaining * percentValues[i]);
244                 }
245             }
246         }
247 
248         return widths;
249     }

251     /**
252      * {@inheritDoc} This implementation adjusts the current width values for
253      * columns with a fixed width and recalculates the percent values for the
254      * other columns.
255      */
256     public void recalibrate(int[] columnSizes)
257     {
258         if (columnSizes == null)
259         {
260             throw new IllegalArgumentException(
261                     "Array with column sizes must not be null!");
262         }
263         if (columnSizes.length != getColumnCount())
264         {
265             throw new IllegalArgumentException(
266                     "Wrong number of elements in columns array!");
267         }
268 
269         int totalSize = sumUpTotalSize(columnSizes);
270         int fixedSize = recalibrateFixedWidths(columnSizes);
271         double remainingSize = totalSize - fixedSize;
272 
273         for (int i = 0; i < columnSizes.length; i++)
274         {
275             if (isPercentWidth(i))
276             {
277                 percentValues[i] = (remainingSize == 0) ? 0
278                         : columnSizes[i] / remainingSize;
279             }
280         }
281     }

283     /**
284      * Performs the recalibration of columns with a fixed width. This method
285      * also sums up the width of all columns with a fixed width.
286      *
287      * @param columnSizes the array with the column sizes
288      * @return the sum of the widths of the columns with a fixed width
289      */
290     private int recalibrateFixedWidths(int[] columnSizes)
291     {
292         int fixedWidth = 0;
293 
294         for (int i = 0; i < columnSizes.length; i++)
295         {
296             if (!isPercentWidth(i))
297             {
298                 fixedWidths[i] = columnSizes[i];
299                 fixedWidth += columnSizes[i];
300             }
301         }
302 
303         return fixedWidth;
304     }

306     /**
307      * Validates the width specification for the specified column tag. Throws an
308      * exception if the specification is invalid, i.e. if both a fixed and a
309      * relative width is specified or if the percent value is invalid.
310      *
311      * @param colTag the column tag to be checked
312      * @throws FormBuilderException if the width specification is invalid
313      */
314     private static void checkColumnWidthSpec(TableColumnTag colTag)
315             throws FormBuilderException
316     {
317         if (colTag.getColumnWidth() != null)
318         {
319             if (colTag.getPercentWidth() != 0)
320             {
321                 throw new FormBuilderException(
322                         "Column has both a fixed and a relative width!");
323             }
324         }
325         else
326         {
327             if (colTag.getPercentWidth() < 0 || colTag.getPercentWidth() > PERCENT)
328             {
329                 throw new FormBuilderException(String.format(
330                         "Invalid percent width %d!", colTag.getPercentWidth()));
331             }
332         }
333     }

343     {
344         return colTag.getColumnWidth() == null;

347     /**
348      * Calculates the percent values for all columns. This method is needed if
349      * the sum of percent values is less than 100. In this case the difference
350      * to 100 is determined, and the remaining number is added to all columns
351      * with an undefined percent value in equal parts.
352      *
353      * @param tt the table tag
354      * @param percentCnt the number of columns with a percent value
355      * @param sumPercents the sum of the percent values
356      * @param undefPercentCnt the number of columns with an undefined percent
357      *        value
358      * @return an array with all percent values
359      * @throws FormBuilderException if the sum of percent values is invalid
360      */
361     private static double[] calculateInitialPercentValues(TableTag tt,
362             int percentCnt, int sumPercents, int undefPercentCnt)
363             throws FormBuilderException
364     {
365         if (sumPercents > PERCENT)
366         {
367             throw new FormBuilderException(
368                     "Sum of percent values is greater than 100: " + sumPercents);
369         }
370         double[] percents = new double[tt.getColumnCount()];
371 
372         if (undefPercentCnt > 0)
373         {
374             double delta = (PERCENT - sumPercents) / undefPercentCnt;
375             for (int i = 0; i < percents.length; i++)
376             {
377                 TableColumnTag colTag = tt.getColumn(i);
378                 if (isPercentWidth(colTag))
379                 {
380                     double percentWidth =
381                             (colTag.getPercentWidth() != 0) ? colTag
382                                     .getPercentWidth() : delta;
383                     percents[i] = percentWidth / PERCENT;
384                 }
385             }
386         }
387 
388         else if (percentCnt > 0)
389         {
390             double delta = (PERCENT - sumPercents) / percentCnt / PERCENT;
391             for (int i = 0; i < percents.length; i++)
392             {
393                 TableColumnTag colTag = tt.getColumn(i);
394                 if (isPercentWidth(colTag))
395                 {
396                     percents[i] = colTag.getPercentWidth() / PERCENT + delta;
397                 }
398             }
399         }
400 
401         return percents;
402     }

404     /**
405      * Helper method for calculating the sum of the given array with column
406      * sizes.
407      *
408      * @param columnSizes an array with column sizes
409      * @return the sum of the sizes
410      */
411     private static int sumUpTotalSize(int[] columnSizes)
412     {
413         int totalSize = 0;
414         for (int columnSize : columnSizes)
415         {
416             totalSize += columnSize;
417         }
418 
419         return totalSize;
420     }
