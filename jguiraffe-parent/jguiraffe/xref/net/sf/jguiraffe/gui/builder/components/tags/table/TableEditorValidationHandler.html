16  package net.sf.jguiraffe.gui.builder.components.tags.table;
17  
18  import net.sf.jguiraffe.gui.forms.Form;
19  import net.sf.jguiraffe.gui.forms.FormValidatorResults;
20  
21  /**
22   * <p>
23   * Definition of an interface to be implemented by objects interested in
24   * validation events triggered by a column editor.
25   * </p>
26   * <p>
27   * For the single columns of a table special editors or validators can be
28   * defined. If validation fails, the user should somehow be notified to get a
29   * feedback what he or she has done wrong. This is the purpose of this
30   * interface: Its single method is invoked whenever user input entered into a
31   * table gets validated (also if validation is successful). An implementation
32   * can react on validation errors in a suitable manner.
33   * </p>
34   * <p>
35   * Platform-specific default implementations of this interface are available and
36   * are constructed for table components automatically. These implementations
37   * display validation errors in a message box if validation fails. Only if a
38   * different behavior is desired, custom implementations need to be created.
39   * </p>
40   *
41   * @author Oliver Heger
42   * @version $Id: TableEditorValidationHandler.java 205 2012-01-29 18:29:57Z oheger $
43   */
44  public interface TableEditorValidationHandler
45  {
46      /**
47       * Validation has been performed for a column editor of a table. An
48       * implementation can now react on the results of this validation. The
49       * passed in parameters provide all available information about the affected
50       * column and the <code>Form</code> object used for editing. An
51       * appropriate default behavior would be to display the validation errors
52       * (if any). But an implementation is not limited to this action: It can
53       * even manipulate the entered values. To achieve this the form can be
54       * directly manipulated. Then <b>true</b> must be returned (which means
55       * that the form's fields should again be queried). A return value of
56       * <b>false</b> means that no further steps need to be performed.
57       *
58       * @param table the table component
59       * @param editForm the editor form used for this table
60       * @param tableTag the tag describing the affected table
61       * @param results the object with the validation results
62       * @param row the index of the affected row
63       * @param col the index of the affected column
64       * @return a flag whether the entered values should be read again from the
65       * form's fields; an implementation should return <b>true</b> if it has
66       * manipulated the content of the fields
67       */
68      boolean validationPerformed(Object table, Form editForm,
69              TableTag tableTag, FormValidatorResults results, int row, int col);
70  }
