34  /**
35   * <p>
36   * A helper class for dealing with {@code Form} objects related to table
37   * components.
38   * </p>
39   * <p>
40   * The implementation of tables for different UI platforms in <em>JGUIraffe</em>
41   * requires some common functionality for dealing with the single properties of
42   * model beans. When defining a table {@code Form} instances are created
43   * allowing the manipulation of properties via the API provided by this class.
44   * This also includes support for data type transformation and validation.
45   * Nevertheless, there is still some boiler-plate code necessary to interact
46   * with these forms, e.g. to retrieve the values to be displayed in a table
47   * column or to write back data the user has changed.
48   * </p>
49   * <p>
50   * This class aims at providing this functionality in a central place so that
51   * platform-specific implementations of table components can be simplified.
52   * Before version 1.3 of this library this code was mainly contained in the
53   * Swing-specific table implementation. It is now refactored so that it can be
54   * reused by other implementations, too.
55   * </p>
56   * <p>
57   * This class uses a form-based approach for accessing the table model data of a
58   * specific row. This means that if a new row is selected for being rendered or
59   * edited, the data of the model is loaded into the row render and editor forms.
60   * From there it can be read (and even updated) using the typical API offered by
61   * the {@code Form} class.
62   * </p>
63   *
64   * @author Oliver Heger
65   * @version $Id$
66   * @since 1.3
67   */

250     /**
251      * Sets the value for the given column in the current row. This method can
252      * be used for modifying the value of a cell if no special edit form is
253      * specified for this column. (If there is an edit form, this method
254      * performs no action.) The value is written into the edit form, and
255      * validation is performed. The table's validation handler is invoked with
256      * the results of the validation. Depending on this, the table is updated or
257      * the changes are discarded. Note: Before this method can be used,
258      * {@code selectCurrentRow()} must have been called first.
259      *
260      * @param table the current table component
261      * @param col the index of the column in question
262      * @param value the value to be set for this cell
263      */
264     public void setColumnValue(Object table, int col, Object value)
265     {
266         if (!hasEditor(col))
267         {
268             ComponentHandler ch =
269                     getTableTag().getRowEditForm()
270                             .getField(getColumn(col).getName())
271                             .getComponentHandler();
272             /* This causes an unchecked warning, but we cannot do much
273              * about it. If the data type is wrong, a CCE is thrown.
274              */
275             ch.setData(value);
276             validateColumn(table, col);
277         }
278     }

280     /**
281      * Validates a column in the current row. This method can be called after
282      * the user has edited a cell in the table. It performs a validation on all
283      * fields displayed in this table. With the results of this validation the
284      * {@code TableEditorValidationHandler} is invoked. This object also
285      * determines how to handle validation errors. If everything goes well, the
286      * data from the edit form is copied into the model bean for the current
287      * row.
288      *
289      * @param table the current table component
290      * @param col the index of the column in question
291      * @return a flag whether validation was successful
292      */
293     public boolean validateColumn(Object table, int col)
294     {
295         Set<String> fields = getEditFields(col);
296         FormValidatorResults vres =
297                 getTableTag().getRowEditForm().validateFields(fields);
298 
299         boolean forceValues;
300         if (getTableTag().getEditorValidationHandler().validationPerformed(
301                 table, getTableTag().getRowEditForm(), getTableTag(), vres,
302                 currentRow, col))
303         {
304             // The validation handler has manipulated the field values;
305             // fetch them again
306             getTableTag().getRowEditForm().validateFields(fields);
307             forceValues = true;
308         }
309         else
310         {
311             forceValues = false;
312         }
313 
314         if (forceValues || vres.isValid())
315         {
316             getTableTag().getRowEditForm().readFields(getModelBean(currentRow),
317                     fields);
318             return true;
319         }
320         return false;
321     }

463     /**
464      * Installs special transformers and validators for the field associated
465      * with the specified column based on its {@code ColumnClass}. This method
466      * checks whether a logic type is defined for this column, if it does not
467      * have a renderer, and if no transformers and validators are defined. If
468      * all these conditions are met, corresponding transformers and validators
469      * are created for the column's logic type and installed at the field
470      * handlers of the row forms.
471      *
472      * @param col the column index
473      * @return a flag whether transformers were installed
474      */
475     public boolean installTransformersForColumnType(int col)
476     {
477         ColumnClass columnClass = getLogicDataClass(col);
478         if (columnClass != null && !hasRenderer(col))
479         {
480             FieldHandler field = getRenderField(col);
481             if (hasNoTransformer(field))
482             {
483                 TransformerWrapper readTransformer =
484                         getTransformerFactory().getReadTransformer(
485                                 getTableTag(), columnClass);
486                 TransformerWrapper writeTransformer =
487                         getTransformerFactory().getWriteTransformer(
488                                 getTableTag(), columnClass);
489                 ValidatorWrapper validator =
490                         getTransformerFactory().getValidator(getTableTag(),
491                                 columnClass);
492                 installTransformer(field, readTransformer, writeTransformer,
493                         validator);
494 
495                 field =
496                         getTableTag().getRowEditForm().getField(
497                                 getColumnFieldName(col));
498                 installTransformer(field, readTransformer, writeTransformer,
499                         validator);
500                 return true;
501             }
502         }
503 
504         return false;
505     }

507     /**
508      * Determines the fixed size columns of the represented table and
509      * initializes their widths in the {@code TableColumnWidthController} of the
510      * current table.
511      *
512      * @param sizeHandler the {@code UnitSizeHandler}
513      * @param container the enclosing container object
514      * @return the total width of all columns with a fixed column width
515      * @throws FormBuilderException if an error occurs
516      */
517     public int calculateFixedColumnWidths(UnitSizeHandler sizeHandler,
518             Object container) throws FormBuilderException
519     {
520         TableColumnWidthController widthController =
521                 getTableTag().getColumnWidthController();
522         int totalWidth = 0;
523 
524         for (int i = 0; i < getColumnCount(); i++)
525         {
526             NumberWithUnit columnWidth = getColumn(i).getColumnWidth();
527             if (columnWidth != null)
528             {
529                 int width = columnWidth.toPixel(sizeHandler, container, false);
530                 widthController.setFixedWidth(i, width);
531                 totalWidth += width;
532             }
533         }
534 
535         return totalWidth;
536     }

621      * is set.

624      * @return <b>true</b> if not transformer or validator is defined,

628     {
629         return getFieldHandlerFactory().getReadTransformerReference(field)
630                 .getTransformer() == DummyWrapper.INSTANCE
631                 && getFieldHandlerFactory().getWriteTransformerReference(field)
632                         .getTransformer() == DummyWrapper.INSTANCE
633                 && getFieldHandlerFactory().getValidatorReference(field)
634                         .getValidator() == DummyWrapper.INSTANCE;

637     /**
638      * Helper method for installing the specified transformers and validators in
639      * the references of the given field handler.
640      *
641      * @param field the {@code FieldHandler}
642      * @param readTransformer the read transformer
643      * @param writeTransformer the write transformer
644      * @param validator the validator
645      */
646     private void installTransformer(FieldHandler field,
647             TransformerWrapper readTransformer,
648             TransformerWrapper writeTransformer, ValidatorWrapper validator)
649     {
650         getFieldHandlerFactory().getReadTransformerReference(field)
651                 .setTransformer(readTransformer);
652         getFieldHandlerFactory().getWriteTransformerReference(field)
653                 .setTransformer(writeTransformer);
654         getFieldHandlerFactory().getValidatorReference(field).setValidator(
655                 validator);

658     /**
659      * Creates the data model for the managed table. This implementation tries
660      * to directly reuse the passed in model collection, so that it can be
661      * updated conveniently by the application. However, we have to ensure that
662      * the collection serving as data model can be directly accessed by index.
663      * So if this is not the case, a copy from the data collection is created.
664      *
665      * @param tabTag the {@code TableTag}
666      * @return the data model for the table
667      */
668     private static List<Object> createModel(TableTag tabTag)
669     {
670         if (tabTag.getTableModel() instanceof List
671                 && tabTag.getTableModel() instanceof RandomAccess)
672         {
673             // This is not a safe cast, however, as we do not add new objects,
674             // no class cast exceptions are produced.
675             @SuppressWarnings("unchecked")
676             List<Object> model = (List<Object>) tabTag.getTableModel();
677             return model;
678         }
679         return new ArrayList<Object>(tabTag.getTableModel());
680     }
