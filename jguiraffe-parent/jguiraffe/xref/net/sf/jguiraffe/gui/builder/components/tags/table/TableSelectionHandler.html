16  package net.sf.jguiraffe.gui.builder.components.tags.table;
17  
18  /**
19   * <p>
20   * Definition of an interface that allows changing the way a selected or focused
21   * cell in the table is displayed.
22   * </p>
23   * <p>
24   * If custom editors or renderers are installed for columns of a table the
25   * components used for this purpose must reflect the selected and/or focused
26   * state of the represented cell. The platform specific table implementations
27   * will provide defaults for doing this. If an application needs a very specific
28   * way of marking cells as selected or focused, it can create a specialized
29   * implementation of this interface.
30   * </p>
31   *
32   * @author Oliver Heger
33   * @version $Id: TableSelectionHandler.java 205 2012-01-29 18:29:57Z oheger $
34   */
35  public interface TableSelectionHandler
36  {
37      /**
38       * Prepares the renderer or editor component before it is displayed. This
39       * method is called whenever a custom renderer or editor component is
40       * requested. An implementation can initialize this component in an
41       * arbitrary way based on the passed in parameters.
42       *
43       * @param table the table component
44       * @param tableTag the tag representing the affected table
45       * @param component the component that is used as custom renderer or editor
46       * @param selected a flag whether the cell is selected
47       * @param hasFocus a flag whether the cell has the focus
48       * @param row the index of the affected row
49       * @param col the index of the affected column
50       */
51      void prepareComponent(Object table, TableTag tableTag, Object component,
52              boolean selected, boolean hasFocus, int row, int col);
53  }
