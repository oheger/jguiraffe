16  package net.sf.jguiraffe.gui.builder.di.tags;

18  import net.sf.jguiraffe.di.BeanProvider;
19  import net.sf.jguiraffe.gui.builder.di.DIBuilderData;
20  
21  import org.apache.commons.jelly.JellyTagException;
22  import org.apache.commons.jelly.MissingAttributeException;
23  import org.apache.commons.jelly.TagSupport;
24  import org.apache.commons.jelly.XMLOutput;

26  /**
27   * <p>
28   * An abstract base class for tag handler implementations that create bean
29   * providers.
30   * </p>
31   * <p>
32   * This class provides some common functionality related to the creation and
33   * management of <code>{@link BeanProvider}</code> objects. The features
34   * implemented here include the following:
35   * <ul>
36   * <li>A <code>store</code> attribute is implemented for selecting the bean
37   * store, in which to store the provider created by this tag. Alternatively the
38   * tag can be nested inside a <code>{@link BeanStoreTag}</code>.</li>
39   * <li>Functionality is provided for passing a bean provider to a bean store.
40   * Both standard and anonymous bean providers can be stored.</li>
41   * <li>A framework of methods is defined that are called during the execution of
42   * the tag. While most of these methods are fully functional, derived classes
43   * can override some for adding specific functionality.</li>
44   * <li>An abstract <code>createBeanProvider()</code> method is declared. Here
45   * concrete subclasses have to implement the creation of their bean provider.</li>
46   * </ul>
47   * </p>
48   *
49   * @author Oliver Heger
50   * @version $Id: AbstractBeanTag.java 205 2012-01-29 18:29:57Z oheger $
51   */
52  public abstract class AbstractBeanTag extends TagSupport
53  {
54      /** Stores a reference to an enclosing bean store tag. */
55      private BeanStoreTag beanStoreTag;
56  
57      /** The target dependency tag if this is an anonymous bean declaration. */
58      private DependencyTag targetDependency;
59  
60      /** Stores the name of the bean store to use. */
61      private String store;
62  
63      /**
64       * Returns the name of the managed bean definition. This name is used when
65       * the <code>BeanProvider</code> created by this tag is stored in a
66       * <code>BeanStore</code>. In this base implementation the name is only
67       * queried to distinguish whether this is a normal or an anonymous bean
68       * definition. <code>getName()</code> always returns <b>null</b> here; if
69       * derived classes support setting a name, they must override this method.
70       *
71       * @return the name of the bean definition
72       */
73      public String getName()
74      {
75          return null;
76      }
77  
78      /**
79       * Returns the name of the <code>BeanStore</code>, in which to place the
80       * managed bean definition.
81       *
82       * @return the name of the bean store
83       */
84      public String getStore()
85      {
86          return store;
87      }
88  
89      /**
90       * Set method of the store attribute.
91       *
92       * @param store the attribute's value
93       */
94      public void setStore(String store)
95      {
96          this.store = store;
97      }
98  
99      /**
100      * Returns a reference to the enclosing <code>BeanStoreTag</code>. This can
101      * be <b>null</b> if there is no such tag.
102      *
103      * @return a reference to the enclosing bean store tag
104      */
105     public BeanStoreTag getBeanStoreTag()
106     {
107         return beanStoreTag;
108     }
109 
110     /**
111      * Returns the target dependency. If this is a valid anonymous bean
112      * declaration, there is a target dependency tag, on which to set the
113      * reference to the declared bean.
114      *
115      * @return the target dependency tag
116      */
117     public DependencyTag getTargetDependency()
118     {
119         return targetDependency;
120     }
121 
122     /**
123      * Returns a flag whether this tag declares an anonymous bean. This is a
124      * bean without a name that is only visible in the context it is declared.
125      *
126      * @return a flag if this is an anonymous bean
127      */
128     public boolean isAnonymous()
129     {
130         return getTargetDependency() != null;
131     }
132 
133     /**
134      * The main method of this tag. Delegates to the specific processing
135      * methods.
136      *
137      * @param output the output object
138      * @throws JellyTagException if an error occurs or the tag is incorrectly
139      *         used
140      */
141     public void doTag(XMLOutput output) throws JellyTagException
142     {
143         setBeanStoreTag((BeanStoreTag) findAncestorWithClass(BeanStoreTag.class));
144         processBeforeBody();
145         invokeBody(output);
146         process();
147     }
148 
149     /**
150      * Sets the reference to the enclosing <code>BeanStoreTag</code>. This can
151      * be <b>null</b> if this tag is not placed in the body of a bean store tag.
152      *
153      * @param beanStoreTag the enclosing <code>BeanStoreTag</code>
154      */
155     protected void setBeanStoreTag(BeanStoreTag beanStoreTag)
156     {
157         this.beanStoreTag = beanStoreTag;
158     }
159 
160     /**
161      * Performs pre-processing before the body of this tag is evaluated. This
162      * method is called by the <code>doTag()</code> method. It checks whether
163      * this tag is nested inside a <code>{@link DependencyTag}</code>, which
164      * will become the target dependency, i.e. if this is an anonymous bean
165      * declaration, the target dependency will be initialized to point to this
166      * bean.
167      *
168      * @throws JellyTagException if no name is defined for this tag, but no
169      *         target dependency can be obtained or the target dependency is
170      *         already initialized
171      */
172     protected void processBeforeBody() throws JellyTagException
173     {
174         targetDependency = null;
175         if (getName() == null)
176         {
177             // Is this an anonymous tag?
178             if (getParent() instanceof DependencyTag)
179             {
180                 DependencyTag depParent = (DependencyTag) getParent();
181                 if (depParent.getRefName() == null)
182                 {
183                     targetDependency = depParent;
184                 }
185                 else
186                 {
187                     // This tag already has a dependency!
188                     throw new JellyTagException(
189                             "Cannot create anonymous bean: "
190                                     + "Parent tag is already defined!");
191                 }
192             }
193 
194             if (targetDependency == null)
195             {
196                 throw new MissingAttributeException("name");
197             }
198         }
199     }
200 
201     /**
202      * Processes this tag after its body has been evaluated. Creates the bean
203      * provider and adds it to the correct bean store.
204      *
205      * @throws JellyTagException if the tag is incorrectly used
206      */
207     protected void process() throws JellyTagException
208     {
209         store(createBeanProvider());
210     }
211 
212     /**
213      * Stores the newly created <code>BeanProvider</code> in the correct bean
214      * store. This method is called by <code>process()</code> when a provider
215      * could be successfully created.
216      *
217      * @param provider the newly created bean provider
218      * @throws JellyTagException if an error occurs
219      */
220     protected void store(BeanProvider provider) throws JellyTagException
221     {
222         String storeName = getStore();
223         if (storeName == null && getBeanStoreTag() != null)
224         {
225             storeName = getBeanStoreTag().getName();
226         }
227 
228         if (isAnonymous())
229         {
230             String refName = DIBuilderData.get(getContext())
231                     .addAnonymousBeanProvider(storeName, provider);
232             getTargetDependency().setRefName(refName);
233         }
234         else
235         {
236             DIBuilderData.get(getContext()).addBeanProvider(storeName,
237                     getName(), provider);
238         }
239     }
240 
241     /**
242      * Creates the bean provider defined by this tag. This method is invoked by
243      * <code>process()</code>. A concrete subclass has to implement it and
244      * return a <code>BeanProvider</code>, which will be stored in the selected
245      * bean store.
246      *
247      * @return the bean provider defined by this tag
248      * @throws JellyTagException in case of an error
249      */
250     protected abstract BeanProvider createBeanProvider()
251             throws JellyTagException;
252 }
