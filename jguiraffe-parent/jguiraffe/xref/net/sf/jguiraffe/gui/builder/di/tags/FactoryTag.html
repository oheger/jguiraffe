16  package net.sf.jguiraffe.gui.builder.di.tags;

18  import net.sf.jguiraffe.di.impl.Invokable;
19  import net.sf.jguiraffe.di.impl.MethodInvocation;
20  import net.sf.jguiraffe.di.impl.providers.MethodInvocationBeanProvider;

22  import org.apache.commons.jelly.JellyTagException;
23  import org.apache.commons.jelly.XMLOutput;

25  /**
26   * <p>
27   * A tag for defining a <em>bean factory</em> that is used as creation
28   * <code>BeanProvider</code> for a {@link BeanTag}.
29   * </p>
30   * <p>
31   * This tag handler class creates a
32   * {@link MethodInvocationBeanProvider} and passes it to its
33   * enclosing <code>BeanTag</code>. This way it is possible to use a kind of
34   * factory for creating beans. The factory is defined by an optional dependency
35   * to another bean and a required <code>MethodInvocation</code> defining the
36   * method to be invoked on the factory; if no dependency is specified, the
37   * <code>MethodInvocation</code> must refer to a static method. The method to
38   * be invoked must be defined by a {@link MethodInvocationTag} in
39   * the body of this tag. The optional dependency is specified through the
40   * attributes of this tag:
41   * </p>
42   * <p>
43   * <table border="1">
44   * <tr>
45   * <th>Attribute</th>
46   * <th>Description</th>
47   * <th>Optional</th>
48   * </tr>
49   * <tr>
50   * <td valign="top">refName</td>
51   * <td>Specifies a dependency to another bean. On this bean the factory method
52   * will be invoked.</td>
53   * <td valign="top">yes</td>
54   * </tr>
55   * <tr>
56   * <td valign="top">refClass</td>
57   * <td>Specifies a dependency to another bean by its class. On this bean the
58   * factory method will be invoked.</td>
59   * <td valign="top">yes</td>
60   * </tr>
61   * <tr>
62   * <td valign="top">refClassName</td>
63   * <td>Specifies a dependency to another bean by its class name. On this bean
64   * the factory method will be invoked.</td>
65   * <td valign="top">yes</td>
66   * </tr>
67   * <tr>
68   * <td valign="top">refClassLoader</td>
69   * <td>With this attribute a symbolic name for the class loader to be used can
70   * be specified. It is evaluated only if the <code>refClassName</code>
71   * attribute was set. In this case the class loader specified here will be used
72   * for resolving the class name.</td>
73   * <td valign="top">yes</td>
74   * </tr>
75   * <tr>
76   * <td valign="top">value</td>
77   * <td>If the factory object is a constant object, this attribute can be used.
78   * It allows to directly specify the value.</td>
79   * <td valign="top">yes</td>
80   * </tr>
81   * <tr>
82   * <td valign="top">valueClass</td>
83   * <td>If a constant value is to be used for the factory object, it may be
84   * necessary to perform some type conversion. With this attribute the type of
85   * the factory can be specified. The value will then be converted to this type.</td>
86   * <td valign="top">yes</td>
87   * </tr>
88   * <tr>
89   * <td valign="top">valueClassName</td>
90   * <td>Like <code>valueClass</code>, but the name of the property's data
91   * type class is specified.</td>
92   * <td valign="top">yes</td>
93   * </tr>
94   * <tr>
95   * <td valign="top">valueClassLoader</td>
96   * <td>If the data type class of the value is specified by its name only, with
97   * this attribute the class loader can be determined for resolving the class.</td>
98   * <td valign="top">yes</td>
99   * </tr>
100  * </table>
101  * </p>
102  *
103  * @author Oliver Heger
104  * @version $Id: FactoryTag.java 205 2012-01-29 18:29:57Z oheger $
105  */
106 public class FactoryTag extends DependencyTag implements InvokableSupport
107 {
108     /** Stores a reference to the enclosing bean tag. */
109     private BeanTag beanTag;
110 
111     /** A flag whether an <code>Invokable</code> object has been added. */
112     private boolean invokableAdded;
113 
114     /**
115      * Returns a reference to the enclosing <code>BeanTag</code>.
116      *
117      * @return the enclosing bean tag
118      */
119     public BeanTag getBeanTag()
120     {
121         return beanTag;
122     }
123 
124     /**
125      * The main method of this tag. Delegates to the processing methods.
126      *
127      * @param output the output object
128      * @throws JellyTagException if an error occurs
129      */
130     public void doTag(XMLOutput output) throws JellyTagException
131     {
132         if (getParent() instanceof BeanTag)
133         {
134             setBeanTag((BeanTag) getParent());
135         }
136 
137         processBeforeBody();
138         invokeBody(output);
139         process();
140     }
141 
142     /**
143      * Adds an <code>Invokable</code> to this object. This implementation
144      * expects that a single {@link MethodInvocation} object is passed, which
145      * will be used for creating a {@link MethodInvocationBeanProvider}.
146      *
147      * @param inv the <code>Invokable</code> to be added
148      * @throws JellyTagException if the passed in object is not accepted
149      */
150     public void addInvokable(Invokable inv) throws JellyTagException
151     {
152         if (!(inv instanceof MethodInvocation))
153         {
154             throw new JellyTagException(
155                     "Only MethodInvocation objects are supported!");
156         }
157         if (invokableAdded)
158         {
159             throw new JellyTagException("Only a single Invokable can be added!");
160         }
161 
162         assert getBeanTag() != null : "No bean tag is set!";
163         if (getBeanTag().getBeanClassDesc() == null)
164         {
165             throw new JellyTagException(
166                     "A bean class must be defined for the bean tag "
167                             + "when using a factory tag!");
168         }
169 
170         MethodInvocationBeanProvider creator = new MethodInvocationBeanProvider(
171                 hasDependency() ? getDependency() : null,
172                 (MethodInvocation) inv, getBeanTag().getBeanClassDesc());
173         getBeanTag().setBeanCreator(creator);
174         invokableAdded = true;
175     }
176 
177     /**
178      * Initializes the reference to the enclosing <code>BeanTag</code>. This
179      * method is called by the tag's main method if a <code>BeanTag</code> can
180      * be found.
181      *
182      * @param beanTag the enclosing <code>BeanTag</code>
183      */
184     protected void setBeanTag(BeanTag beanTag)
185     {
186         this.beanTag = beanTag;
187     }
188 
189     /**
190      * Performs some pre-processing before the tag's body is evaluated.
191      * Registers the tag as an <code>InvokableSupport</code> object.
192      *
193      * @throws JellyTagException if the tag is incorrectly used
194      */
195     protected void processBeforeBody() throws JellyTagException
196     {
197         if (getBeanTag() == null)
198         {
199             throw new JellyTagException(
200                     "This tag must be nested inside a BeanTag!");
201         }
202         invokableAdded = false;
203         InvocationData.get(getContext()).registerInvokableSupport(this);
204     }
205 
206     /**
207      * The main processing method. This method is invoked after the tag's body
208      * has been processed. It performs some validity checks.
209      *
210      * @throws JellyTagException if the tag is incorrectly used
211      */
212     protected void process() throws JellyTagException
213     {
214         InvocationData.get(getContext()).unregisterInvokableSupport();
215         if (!invokableAdded)
216         {
217             throw new JellyTagException("No MethodInvocation object was added!");
218         }
219     }
220 }
