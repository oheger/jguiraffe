18  import java.util.Stack;
19  
20  import net.sf.jguiraffe.di.Dependency;
21  import net.sf.jguiraffe.di.impl.ChainedInvocation;
22  import net.sf.jguiraffe.di.impl.Invokable;

24  import org.apache.commons.jelly.JellyContext;
25  import org.apache.commons.jelly.JellyTagException;

27  /**
28   * <p>
29   * An internally used helper class for dealing with invocation objects.
30   * </p>
31   * <p>
32   * Invocation objects (e.g. <code>MethodInvocation</code> or
33   * <code>ConstructorInvocation</code>) play an important role in the dependency
34   * injection framework. Many bean providers support them for performing creation
35   * and initialization of beans. The tags that create such objects need a generic
36   * way of passing their invocation objects to the correct receivers. This task
37   * is handled by this class.
38   * </p>
39   * <p>
40   * An instance of this class lives in the Jelly context (or is automatically
41   * created when it is accessed for the first time). Tasks that support
42   * invocation objects have to register itself at this instance before they
43   * execute their body. This can either be done by passing an
44   * <code>InvokableSupport</code> object to the
45   * <code>registerInvokableSupport()</code> method or an instance of
46   * {@link ChainedInvocation}. In the latter case some enhanced functionality is
47   * available, e.g. obtaining dependencies to local variables or setting a result
48   * and a source variable.
49   * </p>
50   *
51   * @author Oliver Heger
52   * @version $Id: InvocationData.java 205 2012-01-29 18:29:57Z oheger $
53   */
54  final class InvocationData
55  {
56      /**
57       * Constant for the key, under which an instance is stored in the Jelly
58       * context.
59       */
60      private static final String CTX_KEY = InvocationData.class.getName();
61  
62      /** A stack with the registered support objects. */
63      private Stack<SupportData> supportObjects;
64  
65      /**
66       * Creates a new instance of <code>InvocationData</code>. For obtaining
67       * an instance use the static <code>get()</code> method.
68       */
69      private InvocationData()
70      {
71          supportObjects = new Stack<SupportData>();
72      }
73  
74      /**
75       * Returns the object of this class that can be found in the specified Jelly
76       * context. If the context does not contain such an instance, a new instance
77       * is created, stored in the context, and returned.
78       *
79       * @param context the Jelly context
80       * @return the instance for this context
81       */
82      public static InvocationData get(JellyContext context)
83      {
84          InvocationData result = (InvocationData) context.getVariable(CTX_KEY);
85          if (result == null)
86          {
87              result = new InvocationData();
88              context.setVariable(CTX_KEY, result);
89          }
90          return result;
91      }
92  
93      /**
94       * Registers the specified <code>InvokableSupport</code> object at this
95       * instance. This means that following invocations of the
96       * <code>addInvokable()</code> method are routed to this object. This
97       * method should be called by tag handler classes that deal with
98       * <code>Invokable</code> objects at the beginning of their processing. It
99       * can be called multiple times for dealing with arbitrary nested
100      * hierarchies of <code>InvokableSupport</code> objects.
101      *
102      * @param invSupport the support object to register
103      * @throws IllegalArgumentException if the object to register is undefined
104      */
105     public void registerInvokableSupport(InvokableSupport invSupport)
106     {
107         if (invSupport == null)
108         {
109             throw new IllegalArgumentException(
110                     "Support object must not be null!");
111         }
112         supportObjects.push(new InvokableSupportData(invSupport));
113     }
114 
115     /**
116      * Registers the specified <code>ChainedInvocation</code> object at this
117      * instance. <code>Invokable</code> objects added after this call will now
118      * be passed to this chain. This method is very similar to the overloaded
119      * variant. However for <code>ChainedInvocation</code> objects some
120      * special features are available.
121      *
122      * @param chain the chain object to register
123      * @throws IllegalArgumentException if the object to register is undefined
124      * @see #registerInvokableSupport(InvokableSupport)
125      */
126     public void registerInvokableSupport(ChainedInvocation chain)
127     {
128         if (chain == null)
129         {
130             throw new IllegalArgumentException("Chain must not be null!");
131         }
132         supportObjects.push(new ChainSupportData(chain));
133     }
134 
135     /**
136      * Unregisters an <code>InvokableSupport</code> object. This method must
137      * be called for each invocation of the
138      * <code>registerInvokableSupport()</code> method at the end of the
139      * responsible tag handler class. (Internally a stack of these registrations
140      * is maintained, so it is not necessary to specify the concrete object to
141      * be unregistered).
142      *
143      * @throws IllegalStateException if nothing can be unregistered
144      */
145     public void unregisterInvokableSupport()
146     {
147         if (supportObjects.isEmpty())
148         {
149             throw new IllegalStateException("Nothing to unregister!");
150         }
151         supportObjects.pop();
152     }
153 
154     /**
155      * Adds an <code>Invokable</code> object. This method is invoked by tag
156      * handler classes that create <code>Invokable</code> objects. It ensures
157      * that the provided object is passed to the correct receiver. This might
158      * cause an exception if the <code>Invokable</code> object does not fit
159      * into the current context.
160      *
161      * @param inv the <code>Invokable</code> object to add
162      * @param result the name of the result variable (can be <b>null</b>)
163      * @param source the name of the variable, which contains the target object
164      * for this invocation
165      * @throws JellyTagException if the passed in object is rejected by its
166      * receiver for some reasons
167      * @see ChainedInvocation#addInvokable(Invokable, String, String)
168      */
169     public void addInvokable(Invokable inv, String result, String source)
170             throws JellyTagException
171     {
172         getCurrentSupportData().addInvokable(inv, result, source);
173     }
174 
175     /**
176      * Returns a <code>Dependency</code> for a local variable of a
177      * <code>{@link ChainedInvocation}</code>. This method is available only
178      * if the current <code>InvokableSupport</code> target object is a
179      * <code>ChainedInvocation</code>.
180      *
181      * @param varName the name of the variable for the dependency
182      * @return the dependency to this local variable
183      * @throws JellyTagException if this method cannot be called in the current
184      * context
185      */
186     public Dependency getVariableDependency(String varName)
187             throws JellyTagException
188     {
189         return getCurrentSupportData().getVariableDependency(varName);
190     }
191 
192     /**
193      * Obtains the currently active support object. If there is none, an
194      * exception will be thrown.
195      *
196      * @return the active support object
197      * @throws JellyTagException if this method cannot be called in the current
198      * context
199      */
200     private SupportData getCurrentSupportData() throws JellyTagException
201     {
202         if (supportObjects.isEmpty())
203         {
204             throw new JellyTagException(
205                     "No InvokableSupport object registered!");
206         }
207         return supportObjects.peek();
208     }
209 
210     /**
211      * An internally used class for managing a support object. There will be
212      * different concrete implementations for the different types of support
213      * objects.
214      */
215     private abstract static class SupportData
216     {
217         /**
218          * Adds an Invokable object to the managed support object if possible.
219          *
220          * @param inv the object to add
221          * @param result the name of the result variable
222          * @param source the name of the source variable
223          * @throws JellyTagException if the object cannot be added
224          */
225         public abstract void addInvokable(Invokable inv, String result,
226                 String source) throws JellyTagException;
227 
228         /**
229          * Returns a dependency to a local variable if possible.
230          *
231          * @param name the name of the variable
232          * @return the dependency
233          * @throws JellyTagException if local variables are not supported
234          */
235         public abstract Dependency getVariableDependency(String name)
236                 throws JellyTagException;
237     }
238 
239     /**
240      * A concrete <code>SupportData</code> implementation that deals with
241      * arbitrary <code>InvokableSupport</code> objects.
242      */
243     private static class InvokableSupportData extends SupportData
244     {
245         /** Stores the support object. */
246         private InvokableSupport support;
247 
248         /**
249          * Creates a new instance of <code>InvokableSupportData</code> and
250          * initializes it with the support object.
251          *
252          * @param sup the support object
253          */
254         public InvokableSupportData(InvokableSupport sup)
255         {
256             support = sup;
257         }
258 
259         /**
260          * Adds the invokable to the managed support objects. Names for result
261          * or source variables are not allowed and cause an exception.
262          *
263          * @param inv the object to add
264          * @param result the name of the result variable
265          * @param source the name of the source variable
266          * @throws JellyTagException if the object cannot be added
267          */
268         @Override
269         public void addInvokable(Invokable inv, String result, String source)
270                 throws JellyTagException
271         {
272             if (result != null || source != null)
273             {
274                 throw new JellyTagException(
275                         "Variable names are not allowed outside a chain context!");
276             }
277             support.addInvokable(inv);
278         }
279 
280         /**
281          * Requests a dependency to a local variable. This implementation always
282          * throws an exception because there is no chain in the context.
283          *
284          * @param name the name of the variable
285          * @return the dependency
286          * @throws JellyTagException if local variables are not supported
287          */
288         @Override
289         public Dependency getVariableDependency(String name)
290                 throws JellyTagException
291         {
292             throw new JellyTagException("Local variables are not supported!");
293         }
294     }
295 
296     /**
297      * A concrete <code>SupportData</code> implementation that deals with
298      * <code>ChainedInvocation</code> objects.
299      */
300     private static class ChainSupportData extends SupportData
301     {
302         /** Stores the chain object. */
303         private ChainedInvocation chain;
304 
305         /**
306          * Creates a new instance of <code>ChainSupportData</code> and
307          * initializes it with the chain object.
308          *
309          * @param c the chain
310          */
311         public ChainSupportData(ChainedInvocation c)
312         {
313             chain = c;
314         }
315 
316         /**
317          * Adds an Invokable to the managed chain. Variable names are fully
318          * supported.
319          *
320          * @param inv the object to add
321          * @param result the name of the result variable
322          * @param source the name of the source variable
323          * @throws JellyTagException if the object cannot be added
324          */
325         @Override
326         public void addInvokable(Invokable inv, String result, String source)
327                 throws JellyTagException
328         {
329             chain.addInvokable(inv, result, source);
330         }
331 
332         /**
333          * Requests a dependency to a local variable. This request is delegated
334          * to the chain.
335          *
336          * @param name the name of the variable
337          * @return the dependency
338          * @throws JellyTagException if local variables are not supported
339          */
340         @Override
341         public Dependency getVariableDependency(String name)
342                 throws JellyTagException
343         {
344             return chain.getChainDependency(name);
345         }
346     }
