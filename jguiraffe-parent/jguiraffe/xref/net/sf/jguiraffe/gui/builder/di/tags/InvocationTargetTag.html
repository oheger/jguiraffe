16  package net.sf.jguiraffe.gui.builder.di.tags;
17  
18  import org.apache.commons.jelly.JellyTagException;
19  import org.apache.commons.jelly.XMLOutput;
20  
21  /**
22   * <p>
23   * A specialized tag handler implementation for defining the target object of a
24   * method invocation.
25   * </p>
26   * <p>
27   * When a {@link MethodInvocationTag} is used in the initializer script of a
28   * bean declaration the method is typically invoked on the managed object which
29   * is the result of the bean declaration. Using {@code InvocationTargetTag} it
30   * is possible to define a different target, e.g. another bean defined in the
31   * current bean context. A use case would be a factory bean which is called to
32   * create objects of a specific type. The following example fragment shows how
33   * the tag can be used in an initializer script of a bean declaration:
34   * </p>
35   *
36   * <pre>
37   * &lt;di:bean name="initializer" singleton="false"&gt;
38   *   &lt;di:methodInvocation method="create" result="obj"&gt;
39   *     &lt;di:param parameterClass="java.lang.String" refName="strConst"/&gt;
40   *     &lt;di:invocationTarget refName="factory"/&gt;
41   *   &lt;/di:methodInvocation&gt;
42   *   ...
43   * &lt;/di:bean&gt;
44   * </pre>
45   * <p>
46   * The tag defines no attributes in addition to the ones inherited from its base
47   * class. So basically the various options for defining the dependency are
48   * available. Refer to the documentation of {@code DependencyTag} or
49   * {@link ParameterTag}. The tag must occur in the body of a
50   * {@link MethodInvocationTag}.
51   * </p>
52   *
53   * @author Oliver Heger
54   * @version $Id: InvocationTargetTag.java 207 2012-02-09 07:30:13Z oheger $
55   * @since 1.1
56   */
57  public class InvocationTargetTag extends DependencyTag
58  {
59      /**
60       * {@inheritDoc} This implementation determines the parent tag which must be
61       * of type {@link MethodInvocationTag}. This tag is passed the dependency
62       * defined here.
63       */
64      public void doTag(XMLOutput output) throws JellyTagException
65      {
66          if (!(getParent() instanceof MethodInvocationTag))
67          {
68              throw new JellyTagException("InvocationTargetTag must be nested in"
69                      + " a MethodInvocationTag!");
70          }
71  
72          invokeBody(output);
73          ((MethodInvocationTag) getParent())
74                  .setTargetDependency(getDependency());
75      }
76  }
