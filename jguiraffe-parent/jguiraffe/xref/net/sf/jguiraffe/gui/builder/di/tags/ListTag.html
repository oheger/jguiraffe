18  import net.sf.jguiraffe.di.BeanProvider;
19  import net.sf.jguiraffe.di.impl.providers.ListBeanProvider;

21  import org.apache.commons.jelly.JellyTagException;
22  
23  /**
24   * <p>
25   * A specialized <code>CollectionTag</code> implementation for creating lists.
26   * </p>
27   * <p>
28   * With this tag handler class list objects can be created that can be passed to
29   * other tags supporting a value (the tag can be placed in the body of each tag
30   * handler class derived from {@link DependencyTag}). Alternatively, the
31   * {@code name} attribute can be specified; in this case the list bean can be
32   * directly queried from the current bean store using this name. The content of
33   * the list is defined by {@link ElementTag} tags in the body of this tag. It is
34   * possible to define the type of the list elements as attributes. Alternatively
35   * the <code>ElementTag</code> tags can have a class definition, which will
36   * override the one set at this tag.
37   * </p>
38   * <p>
39   * The following example fragment shows how this tag can be used to define a
40   * list that will be passed as an argument of a constructor invocation:
41   *
42   * <pre>
43   *   &lt;constructor&gt;
44   *     &lt;param&gt;
45   *       &lt;list elementClassName=&quot;java.lang.Integer&quot;&gt;
46   *         &lt;element value=&quot;1&quot;/&gt;
47   *         &lt;element value=&quot;2&quot;/&gt;
48   *         &lt;element value=&quot;3&quot;/&gt;
49   *       &lt;/list&gt;
50   *     &lt;/param&gt;
51   *   &lt;/constructor&gt;
52   * </pre>
53   *
54   * </p>
55   * <p>
56   * <code>ListTag</code> supports the following attributes:
57   * <table border="1">
58   * <tr>
59   * <th>Attribute</th>
60   * <th>Description</th>
61   * <th>Optional</th>
62   * </tr>
63   * <tr>
64   * <td valign="top">name</td>
65   * <td>Using this attribute a name for the list bean can be specified. The bean
66   * can then be queried from the current bean store by this name. If no name is
67   * provided, an anonymous bean is created; in this case the tag must be nested
68   * in the body of a {@link DependencyTag}.</td>
69   * <td valign="top">yes</td>
70   * </tr>
71   * <tr>
72   * <td valign="top">elementClass</td>
73   * <td>Here the class of the elements contained can be specified. When elements
74   * are added, the tag will try to convert the objects to this class if
75   * necessary. If an {@link ElementTag} in the body of this tag defines a value
76   * class, this class will be used thus overriding the element class specified at
77   * the collection level. If neither the collection tag nor the element tag
78   * specifies a class, no type conversion will be performed.</td>
79   * <td valign="top">yes</td>
80   * </tr>
81   * <tr>
82   * <td valign="top">elementClassName</td>
83   * <td>This attribute has the same effect as <code>elementClass</code>, but the
84   * name of the element class is specified rather than the class object itself.</td>
85   * <td valign="top">yes</td>
86   * </tr>
87   * <tr>
88   * <td valign="top">elementClassLoader</td>
89   * <td>If the name of the element class is specified, with this attribute a
90   * class loader for resolving the class can be defined. The name specified here
91   * will be passed to the current {@link net.sf.jguiraffe.di.ClassLoaderProvider
92   * ClassLoaderProvider} for obtaining the desired class loader.</td>
93   * <td valign="top">yes</td>
94   * </tr>
95   * </table>
96   * </p>
97   *
98   * @author Oliver Heger
99   * @version $Id: ListTag.java 205 2012-01-29 18:29:57Z oheger $
100  */
101 public class ListTag extends CollectionTag
102 {
103     /**
104      * Creates the bean provider representing the collection managed by this
105      * tag. This implementation will create a {@link ListBeanProvider} object.
106      *
107      * @return the bean provider produced by this tag
108      * @throws JellyTagException if an error occurs
109      */
110     @Override
111     protected BeanProvider createBeanProvider() throws JellyTagException
112     {
113         return new ListBeanProvider(getElementDependencies());
114     }
115 }
