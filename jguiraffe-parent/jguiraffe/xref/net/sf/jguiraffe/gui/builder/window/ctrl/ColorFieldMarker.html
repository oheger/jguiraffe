27  /**
28   * <p>
29   * A specialized {@link FormControllerValidationListener} implementation that
30   * can change the colors of fields depending on their validation status.
31   * </p>
32   * <p>
33   * An instance of this class can be registered at a {@link FormController} as
34   * {@link FormControllerValidationListener} and
35   * {@link FormControllerFieldStatusListener}. Whenever an event is received from
36   * the {@link FormController} the current status of the affected field(s) is
37   * checked. The instance can be configured with multiple colors corresponding to
38   * the possible field states:
39   * <ul>
40   * <li>not visited, invalid</li>
41   * <li>visited, invalid</li>
42   * <li>not visited, validation warning</li>
43   * <li>visited, validation warning</li>
44   * </ul>
45   * The foreground and background colors of the field(s) are set to the
46   * corresponding colors.
47   * </p>
48   * <p>
49   * Valid fields do not change their colors. If colors for a certain field status
50   * are undefined, the default colors are used, too. For instance, if fields for
51   * which a validation warning exists should not be highlighted in a special way,
52   * just leave the properties for these colors undefined. This class implements a
53   * very simple means for marking invalid input fields. If the background color
54   * of a text field is changed to, say, red, it will be pretty obvious that
55   * something is wrong with this field.
56   * </p>
57   * <p>
58   * Instances of this class can be associated with a single form controller only
59   * and should not be reused (because they keep an internal state about the input
60   * fields involved). If they are declared in a bean definition file of the
61   * dependency injection framework, the {@code singleton} attribute should be set
62   * to <b>false</b>.
63   *
64   * @author Oliver Heger
65   * @version $Id: ColorFieldMarker.java 205 2012-01-29 18:29:57Z oheger $
66   */

349      * A validation operation was performed by the {@code FormController}. This
350      * implementation checks which fields need to be updated and sets the
351      * corresponding new colors.

353      * @param event the validation event
354      */
355     public void validationPerformed(FormControllerValidationEvent event)
356     {
357         for (String field : event.getValidationResults().getFieldNames())
358         {
359             updateField(event.getFormController(), field, event

436     /**
437      * Updates the colors for the given field. This method is called whenever an
438      * event is received from the {@code FormController} that indicates a status
439      * change of a field. This implementation checks whether the status of the
440      * given field has actually changed. If this is the case, the colors for the
441      * new status are obtained and set.
442      *
443      * @param controller the {@code FormController}
444      * @param field the name of the field affected by the change
445      * @param validationResults the current {@code FormValidatorResults} object
446      * @param visited the visited state of the current field
447      * @throws FormBuilderRuntimeException if the field name cannot be resolved
448      */
449     protected void updateField(FormController controller, String field,
450             FormValidatorResults validationResults, boolean visited)
451     {
452         FieldValidationStatus status = FieldValidationStatus.getStatus(
453                 validationResults.getResultsFor(field), visited);
454         if (status != getLastValidationStatus(field))
455         {
456             // status has changed
457             setLastValidationStatus(field, status);
458 
459             Color orgFg = getWidgetForegroundColor(controller, field);
460             Color orgBg = getWidgetBackgroundColor(controller, field);
461             Color newFg = getForegroundColor(status);
462             Color newBg = getBackgroundColor(status);
463 
464             if (newFg == null)
465             {
466                 newFg = orgFg;
467             }
468             if (newBg == null)
469             {
470                 newBg = orgBg;
471             }
472 
473             WidgetHandler wh = controller.getComponentBuilderData()
474                     .getWidgetHandler(field);
475             wh.setForegroundColor(newFg);
476             wh.setBackgroundColor(newBg);
477         }
478     }

482      *
483      * @param status the {@code FieldValidationStatus}
484      * @param idx the index of the color (background or foreground)
485      * @return the corresponding color or <b>null</b> if undefined

491     }

493     /**
494      * Helper method for setting one of the colors for marking input fields.
495      *
496      * @param status the {@code FieldValidationStatus}
497      * @param idx the index of the color (background or foreground)
498      * @param col the new color
499      */
500     private void setColor(FieldValidationStatus status, int idx, Color col)
501     {
502         Color[] cols = markColors.get(status);
503         if (cols == null)
504         {
505             cols = new Color[2];
506             markColors.put(status, cols);
507         }
508 
509         cols[idx] = col;
510     }

512     /**
513      * Helper method for obtaining the color of a widget. The original widget
514      * colors are retrieved once and then cached.
515      *
516      * @param controller the {@code FormController}
517      * @param field the name of the field in question
518      * @param idx the index of the color (foreground or background)
519      * @return the corresponding color
520      * @throws FormBuilderRuntimeException if the widget cannot be resolved
521      */
522     private Color getWidgetColor(FormController controller, String field,
523             int idx)
524     {
525         Color[] widgetColors = orgColors.get(field);
526 
527         if (widgetColors == null)
528         {
529             WidgetHandler wh = controller.getComponentBuilderData()
530                     .getWidgetHandler(field);
531             if (wh == null)
532             {
533                 throw new FormBuilderRuntimeException(
534                         "Widget cannot be found: " + field);
535             }
536 
537             widgetColors = new Color[2];
538             widgetColors[IDX_BG] = wh.getBackgroundColor();
539             widgetColors[IDX_FG] = wh.getForegroundColor();
540             orgColors.put(field, widgetColors);
541         }
542 
543         return widgetColors[idx];
544     }
