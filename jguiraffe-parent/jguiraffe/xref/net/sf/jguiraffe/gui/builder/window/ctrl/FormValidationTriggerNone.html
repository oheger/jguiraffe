16  package net.sf.jguiraffe.gui.builder.window.ctrl;

18  /**
19   * <p>
20   * A specialized implementation of a <code>FormValidationTrigger</code> that
21   * does no triggering at all.
22   * </p>
23   * <p>
24   * This is a very simplistic implementation of the
25   * <code>FormValidationTrigger</code> interface. It literally does nothing,
26   * i.e. no user event will cause an additional validation. This means that the
27   * form is validated once when the window opens, and then every time the user
28   * hits the OK button. (At these points in a form's life-cycle a validation is
29   * always performed.)
30   * </p>
31   * <p>
32   * If this validation trigger implementation is used, the user gets feedback
33   * about the validity of the data entered only when the form is to be committed.
34   * If then some fields contain invalid data, a message box is displayed, and the
35   * form cannot be closed. This is not very spectacular. On the other hand, the
36   * user is not overwhelmed by validation error messages while still editing the
37   * form. So especially for classic data gathering applications this trigger
38   * strategy is appropriate.
39   * </p>
40   * <p>
41   * Applications that want to show a more interactive behavior can choose a
42   * different trigger strategy, e.g. one that triggers a validation every time
43   * the user leaves an input field.
44   * </p>
45   *
46   * @author Oliver Heger
47   * @version $Id: FormValidationTriggerNone.java 205 2012-01-29 18:29:57Z oheger $
48   */
49  public class FormValidationTriggerNone implements FormValidationTrigger
50  {
51      /**
52       * Initializes this trigger and gives it the chance to register itself at
53       * the specified form controller. This implementation does not do any
54       * registration. It will never trigger an additional validation.
55       *
56       * @param controller the form controller
57       */
58      public void initTrigger(FormController controller)
59      {
60      }
61  }
