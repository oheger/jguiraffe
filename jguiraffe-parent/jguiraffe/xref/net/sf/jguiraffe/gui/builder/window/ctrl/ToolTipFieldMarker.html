16  package net.sf.jguiraffe.gui.builder.window.ctrl;
17  
18  import java.util.Collections;
19  import java.util.HashSet;
20  import java.util.Set;
21  
22  import net.sf.jguiraffe.gui.builder.components.ToolTipManager;
23  import net.sf.jguiraffe.gui.forms.FormValidationMessageFormat;
24  import net.sf.jguiraffe.transform.ValidationResult;
25  
26  import org.apache.commons.lang.StringUtils;
27  
28  /**
29   * <p>
30   * A specialized {@code FormControllerValidationListener} implementation that is
31   * able to adapt the tool tips of input elements for which validation messages
32   * are available.
33   * </p>
34   * <p>
35   * An instance of this class can be registered as validation listener at a
36   * {@link FormController}. Whenever a validation event is received the instance
37   * iterates over the validation results for all input fields. The texts for
38   * error and warning messages are extracted and combined. Then they are set as
39   * additional tool tip for the corresponding input element. This is an easy but
40   * powerful means to give feedback about validation results to the user: the
41   * user just have to move the mouse over a control, and all related validation
42   * messages are displayed.
43   * </p>
44   * <p>
45   * The tool tip texts produced by this class can be configured by associating an
46   * instance with a {@link FormValidationMessageFormat} object. This object knows
47   * how to translate validation messages to text. It is invoked for each field
48   * with validation messages, and the text produced by it is set as additional
49   * tool tip text.
50   * </p>
51   * <p>
52   * Implementation note: This class is not thread-safe. An instance can be
53   * associated with a single {@code FormController} only.
54   * </p>
55   *
56   * @author Oliver Heger
57   * @version $Id: ToolTipFieldMarker.java 205 2012-01-29 18:29:57Z oheger $
58   */
59  public class ToolTipFieldMarker implements FormControllerValidationListener
60  {
61      /** The object for formatting validation messages. */
62      private final FormValidationMessageFormat format;
63  
64      /** Stores the names of the fields for which tool tips have been created. */
65      private Set<String> changedFields;
66  
67      /**
68       * Creates a new instance of {@code ToolTipFieldMarker} and initializes it
69       * with the given {@code FormValidationMessageFormat} object.
70       *
71       * @param messageFormat the {@code FormValidationMessageFormat} object (must
72       *        not be <b>null</b>)
73       * @throws IllegalArgumentException if the format object is <b>null</b>
74       */
75      public ToolTipFieldMarker(FormValidationMessageFormat messageFormat)
76      {
77          if (messageFormat == null)
78          {
79              throw new IllegalArgumentException(
80                      "Format object must not be null!");
81          }
82  
83          format = messageFormat;
84          changedFields = Collections.emptySet();
85      }
86  
87      /**
88       * A validation operation has been performed. This implementation determines
89       * the fields for which validation messages exist. It creates corresponding
90       * tool tips for the messages and sets them.
91       *
92       * @param event the validation event
93       */
94      public void validationPerformed(FormControllerValidationEvent event)
95      {
96          // stores the names of the fields affected by this operation
97          Set<String> fields = new HashSet<String>();
98          ToolTipManager manager = event.getFormController()
99                  .getComponentBuilderData().getToolTipManager();
100 
101         // process all fields with validation messages
102         for (String field : event.getValidationResults().getFieldNames())
103         {
104             ValidationResult vres = event.getValidationResults().getResultsFor(
105                     field);
106             if (!vres.getValidationMessages().isEmpty())
107             {
108                 String tip = format.formatField(event.getValidationResults(),
109                         event.getFormController().getForm(), field);
110                 if (StringUtils.isNotEmpty(tip))
111                 {
112                     manager.setAdditionalToolTip(field, tip);
113                     fields.add(field);
114                 }
115             }
116         }
117 
118         // reset all fields that are now valid
119         for (String field : changedFields)
120         {
121             if (!fields.contains(field))
122             {
123                 manager.setAdditionalToolTip(field, null);
124             }
125         }
126 
127         changedFields = fields;
128     }
129 }
