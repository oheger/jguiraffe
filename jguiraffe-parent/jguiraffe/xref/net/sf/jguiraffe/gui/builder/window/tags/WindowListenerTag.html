16  package net.sf.jguiraffe.gui.builder.window.tags;
17  
18  import java.util.EventListener;
19  
20  import net.sf.jguiraffe.gui.builder.action.tags.EventListenerTag;
21  import net.sf.jguiraffe.gui.builder.components.FormBuilderException;
22  import net.sf.jguiraffe.gui.builder.window.WindowListener;
23  
24  import org.apache.commons.jelly.JellyTagException;
25  
26  /**
27   * <p>
28   * A specialized event listener tag that can be used to register window
29   * listeners.
30   * </p>
31   * <p>
32   * With this tag it is possible to link events triggered by a window to an
33   * action. The name of the action to be invoked must be specified as parameter.
34   * Optionally an event filter can be specified; the action is then only invoked
35   * if the window event is matched by the filter. There are no additional
36   * attributes other than the ones defined by the base class.
37   * </p>
38   * <p>
39   * Tags of this type must be nested inside a window tag or alternatively specify
40   * the {@code targetBean} attribute. The listener is registered at the window
41   * created by this window tag or at the target bean.
42   * </p>
43   *
44   * @author Oliver Heger
45   * @version $Id: WindowListenerTag.java 205 2012-01-29 18:29:57Z oheger $
46   */
47  public class WindowListenerTag extends EventListenerTag
48  {
49      /** Constant for the name of the listener type supported by this tag. */
50      private static final String LISTENER_TYPE_NAME = "Window";
51  
52      /**
53       * Creates a new instance of {@code WindowListenerTag}. Sets the initial
54       * listener type.
55       */
56      public WindowListenerTag()
57      {
58          addListenerType(LISTENER_TYPE_NAME, WindowListener.class);
59      }
60  
61      /**
62       * {@inheritDoc} This implementation checks whether this tag is nested
63       * inside a {@link WindowBaseTag}. If this is the case, the window created
64       * by this tag is obtained, and the listener is registered at this window.
65       */
66      @Override
67      protected boolean registerListener(EventListener listener)
68              throws JellyTagException, FormBuilderException
69      {
70          WindowBaseTag parent = (WindowBaseTag) findAncestorWithClass(WindowBaseTag.class);
71          if (parent != null)
72          {
73              addBeanRegistrationCallbacks(listener, null, parent.getWindow());
74              return true;
75          }
76  
77          return false;
78      }
79  }
