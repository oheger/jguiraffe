257     /**
258      * Helper method for obtaining a method.
259      *
260      * @param targetClass the target class (default: ReflectionTestClass)
261      * @param name the method name (default: the overloaded initialize method)
262      * @param signature a string with the signature
263      * @return the method
264      */
265     private static Method fetchMethod(Class<?> targetClass, String name,
266             String signature)
267     {
268         try
269         {
270             Class<?>[] classes = parameterTypesFromSignature(signature);
271             Class<?> target =
272                     (targetClass != null) ? targetClass
273                             : ReflectionTestClass.class;
274             String method = (name != null) ? name : METHOD_INIT;
275             return target.getMethod(method, classes);
276         }
277         catch (Exception ex)
278         {
279             // handle all reflection exception the same way
280             fail("Unexpected exception: " + ex);
281             return null;
282         }
283     }
284 
285     /**
286      * Helper method for obtaining a constructor.
287      *
288      * @param <T> the type of the involved class
289      * @param targetClass the target class (default: ReflectionTestClass)
290      * @param signature a string with the signature
291      * @return the constructor found
292      */
293     private static <T> Constructor<T> fetchConstructor(Class<T> targetClass,
294             String signature)
295     {
296         try
297         {
298             Class<?>[] classes = parameterTypesFromSignature(signature);
299             Class<?> target =
300                     (targetClass != null) ? targetClass
301                             : ReflectionTestClass.class;
302             @SuppressWarnings("unchecked")
303             Constructor<T> result =
304                     (Constructor<T>) target.getConstructor(classes);
305             return result;
306         }
307         catch (Exception ex)
308         {
309             // handle all reflection exception the same way
310             fail("Unexpected exception: " + ex);
311             return null;
312         }
313     }

315     /**
316      * Helper method for extracting the parameter types from a signature string.
317      * The signature is a comma-separated list of class names. Abbreviations
318      * related to the test class are supported.
319      *
320      * @param signature the signature string
321      * @return an array with the corresponding parameter types
322      * @throws ClassNotFoundException if a class cannot be found
323      */
324     private static Class<?>[] parameterTypesFromSignature(String signature)
325             throws ClassNotFoundException
326     {
327         String[] classNames = signature.split(",");
328         Class<?>[] classes = new Class<?>[classNames.length];
329         int idx = 0;
330         for (String cn : classNames)
331         {
332             Class<?> cls;
333             if ("Mode".equals(cn))
334             {
335                 cls = ReflectionTestClass.Mode.class;
336             }
337             else if ("String".equals(cn))
338             {
339                 cls = String.class;
340             }
341             else
342             {
343                 cls = ClassUtils.getClass(cn);
344             }
345             classes[idx++] = cls;
346         }
347         return classes;
348     }
