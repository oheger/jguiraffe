38  /**
39   * An abstract base class for testing {@code Invocation} objects. This
40   * class defines some tests for the basic functionality provided by the
41   * {@code Invocation} base class. These tests are run on concrete
42   * implementations. Some useful constants are also provided.
43   *
44   * @author Oliver Heger
45   * @version $Id: AbstractInvocationTest.java 207 2012-02-09 07:30:13Z oheger $
46   */
47  public abstract class AbstractInvocationTest
48  {
49      /** An array with the parameter types for the invocation. */
50      protected static final ClassDescription[] PARAM_TYPES =
51      { ClassDescription.getInstance(String.class),
52              ClassDescription.getInstance(Integer.TYPE) };
53  
54      /** An array with the current parameter values. */
55      protected static final Object[] PARAM_VALUES =
56      { "ATestString", 42 };
57  
58      /** Constant for the class description of the test class. */
59      protected static final ClassDescription TARGET_CLASS = ClassDescription
60              .getInstance(ReflectionTestClass.class);
61  
62      /** An array with the parameter dependencies. */
63      protected Dependency[] paramDeps;
64  
65      /**
66       * Sets up the test environment. This implementation creates the array with
67       * the dependencies.
68       */
69      @Before
70      public void setUp() throws Exception
71      {
72          paramDeps = new Dependency[PARAM_VALUES.length];
73          for (int i = 0; i < PARAM_VALUES.length; i++)
74          {
75              paramDeps[i] = ConstantBeanProvider.getInstance(PARAM_VALUES[i]);
76          }
77      }

79      /**
80       * Creates a mock object for the dependency provider and initializes it for
81       * the given parameter dependencies. The mock object is initialized to
82       * expect calls for querying the values of the dependencies and to return
83       * the invocation helper.
84       *
85       * @param parameters an array with the parameter dependencies
86       * @param replay a flag whether the provider mock should be replayed
87       * @return the initialized mock object
88       */
89      protected DependencyProvider setUpDependencyProvider(
90              Dependency[] parameters, boolean replay)
91      {
92          DependencyProvider mock = EasyMock.createMock(DependencyProvider.class);
93          for (int i = 0; i < paramDeps.length; i++)
94          {
95              EasyMock.expect(mock.getDependentBean(parameters[i])).andReturn(
96                      PARAM_VALUES[i]);
97          }
98          EasyMock.expect(mock.getInvocationHelper()).andReturn(
99                  new InvocationHelper());
100         if (replay)
101         {
102             EasyMock.replay(mock);
103         }
104         return mock;
105     }

327     /**
328      * Tests obtaining the resolved parameters.
329      */
330     @Test
331     public void testGetResolvedParameters()
332     {
333         DependencyProvider mockDepProvider = EasyMock
334                 .createMock(DependencyProvider.class);
335         Dependency[] deps = new Dependency[PARAM_VALUES.length];
336         for (int i = 0; i < PARAM_VALUES.length; i++)
337         {
338             deps[i] = EasyMock.createMock(Dependency.class);
339             EasyMock.expect(mockDepProvider.getDependentBean(deps[i]))
340                     .andReturn(PARAM_VALUES[i]);
341             EasyMock.replay(deps[i]);
342         }
343         EasyMock.replay(mockDepProvider);
344         Invocation inv = createInvocation(TARGET_CLASS, PARAM_TYPES, deps);
345         Object[] values = inv.getResolvedParameters(mockDepProvider);
346         assertEquals("Wrong length of values array", deps.length, values.length);
347         for (int i = 0; i < values.length; i++)
348         {
349             assertEquals("Wrong value of parameter " + i, PARAM_VALUES[i],
350                     values[i]);
351             EasyMock.verify(deps[i]);
352         }
353         EasyMock.verify(mockDepProvider);
354     }

372     /**
373      * Tests querying the parameter classes.
374      */
375     @Test
376     public void testGetParameterClasses()
377     {
378         DependencyProvider mockDepProvider = EasyMock
379                 .createMock(DependencyProvider.class);
380         mockDepProvider.loadClass(getClass().getName(), "testLoader");
381         EasyMock.expectLastCall().andReturn(getClass());
382         EasyMock.replay(mockDepProvider);
383         ClassDescription[] descs = new ClassDescription[PARAM_TYPES.length + 1];
384         System.arraycopy(PARAM_TYPES, 0, descs, 0, PARAM_TYPES.length);
385         descs[descs.length - 1] = ClassDescription.getInstance(getClass()
386                 .getName(), "testLoader");
387         Dependency[] deps = new Dependency[descs.length];
388         System.arraycopy(paramDeps, 0, deps, 0, paramDeps.length);
389         deps[deps.length - 1] = ConstantBeanProvider.getInstance(this);
390         Invocation inv = createInvocation(TARGET_CLASS, descs, deps);
391         Class<?>[] classes = inv.getParameterClasses(mockDepProvider);
392         assertEquals("Wrong length of classes", descs.length, classes.length);
393         for (int i = 0; i < PARAM_TYPES.length; i++)
394         {
395             assertEquals("Wrong parameter class at " + i, PARAM_TYPES[i]
396                     .getTargetClassName(), classes[i].getName());
397         }
398         assertEquals("Wrong test parameter class", getClass(),
399                 classes[classes.length - 1]);
400         EasyMock.verify(mockDepProvider);
401     }
