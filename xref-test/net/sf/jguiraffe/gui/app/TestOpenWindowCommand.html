89      /**
90       * Tests a successful execution of the command.
91       */
92      @Test
93      public void testExecute() throws Exception
94      {
95          ApplicationContext appCtx = EasyMock
96                  .createMock(ApplicationContext.class);
97          Builder builder = EasyMock.createMock(Builder.class);
98          Window wnd = EasyMock.createMock(Window.class);
99          Application app = new Application();
100         ApplicationBuilderData builderData = new ApplicationBuilderData();
101         EasyMock.expect(appCtx.newBuilder()).andReturn(builder);
102         EasyMock.expect(appCtx.initBuilderData()).andReturn(builderData);
103         EasyMock.expect(builder.buildWindow(locator, builderData)).andReturn(
104                 wnd);
105         EasyMock.replay(appCtx, builder, wnd);
106         app.setApplicationContext(appCtx);
107         OpenWindowCommandTestImpl cmd = new OpenWindowCommandTestImpl(locator);
108         cmd.setApplication(app);
109         cmd.execute();
110         assertEquals("Wrong window", wnd, cmd.getWindow());
111         EasyMock.verify(locator, appCtx, builder, wnd);
112     }

114     /**

117     @Test

145     /**
146      * A test open window command implementation for testing whether the command
147      * is correctly executed.
148      */
149     private static class OpenWindowCommandTestImpl extends OpenWindowCommand
150     {
151         /** The builder data passed to prepareBuilderData(). */
152         ApplicationBuilderData prepareBuilderData;
153 
154         /** The mock window to be returned by getWindow(). */
155         Window mockWindow;
156 
157         public OpenWindowCommandTestImpl(Locator loc)
158         {
159             super(loc);
160         }
161 
162         /**
163          * Records this invocation.
164          */
165         @Override
166         protected void prepareBuilderData(ApplicationBuilderData builderData)
167         {
168             assertNull("Too many invocations", prepareBuilderData);
169             prepareBuilderData = builderData;
170             super.prepareBuilderData(builderData);
171         }
172 
173         /**
174          * Either returns the mock window or calls the super method.
175          */
176         @Override
177         Window getWindow()
178         {
179             return (mockWindow != null) ? mockWindow : super.getWindow();
180         }
181     }
