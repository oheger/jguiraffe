100     /**
101      * Tests whether the registration of the listener is performed correctly.
102      *
103      * @param expectedComponent the expected component name
104      */
105     private void checkRegistration(String expectedComponent)
106             throws JellyTagException, FormBuilderException
107     {
108         assertTrue("Wrong result", tag.passResults(bean));
109         assertNull("Event manager already invoked",
110                 eventManager.addListenerComponent);
111         builderData.invokeCallBacks();
112         assertEquals("Wrong component name", expectedComponent,
113                 eventManager.addListenerComponent);
114         assertEquals("Wrong event type", TYPE, eventManager.addListenerType);
115     }

393     /**
394      * A specialized event manager implementation used for testing whether the
395      * expected listener registration method is called.
396      */
397     private static class FormEventManagerTestImpl extends FormEventManager
398     {
399         /** The component name passed to addEventListener(). */
400         String addListenerComponent;
401 
402         /** The bean passed to addEventListenerToObject(). */
403         Object addListenerTarget;
404 
405         /** The listener type passed to addEventListener(). */
406         String addListenerType;
407 
408         /** The result to be returned by addEventListener(). */
409         int addListenerResult = 1;
410 
411         /** The result to be returned by addEventListenerToObject(). */
412         boolean addListenerObjResult = true;
413 
414         public FormEventManagerTestImpl()
415         {
416             super(new PlatformEventManagerImpl());
417         }
418 
419         /**
420          * Records this invocation.
421          */
422         @Override
423         public int addEventListener(String componentName, String listenerType,
424                 EventListener l)
425         {
426             assertEquals("Wrong listener bean", bean, l);
427             addListenerComponent = componentName;
428             addListenerType = listenerType;
429             return addListenerResult;
430         }
431 
432         /**
433          * Records this invocation.
434          */
435         @Override
436         public boolean addEventListenerToObject(Object target,
437                 String listenerType, EventListener l)
438         {
439             assertEquals("Wrong listener bean", bean, l);
440             addListenerTarget = target;
441             addListenerType = listenerType;
442             return addListenerObjResult;
443         }
444     }
