16  package net.sf.jguiraffe.gui.builder.action.tags;
17  
18  import net.sf.jguiraffe.gui.builder.action.ActionManagerImpl;
19  import net.sf.jguiraffe.gui.builder.action.PopupMenuHandler;
20  import net.sf.jguiraffe.gui.layout.NumberWithUnit;
21  
22  import org.easymock.EasyMock;
23  
24  /**
25   * A test class for PopupHandlerTag that executes a Jelly script. This tests
26   * whether the tag can be correctly accessed in a Jelly script.
27   *
28   * @author Oliver Heger
29   * @version $Id: TestPopupHandlerTagScript.java 205 2012-01-29 18:29:57Z oheger $
30   */
31  public class TestPopupHandlerTagScript extends AbstractActionTagTest
32  {
33      /** The name of the test script. */
34      private static final String SCRIPT = "popup";
35  
36      /** The name of the handler bean. */
37      private static final String BEAN_HANDLER = "menuHandler";
38  
39      /**
40       * Tests executing the tag in a script.
41       */
42      public void testExecuteTag() throws Exception
43      {
44          PopupMenuHandler handler = EasyMock.createMock(PopupMenuHandler.class);
45          EasyMock.replay(handler);
46          context.setVariable(BEAN_HANDLER, handler);
47          executeScript(SCRIPT);
48          builderData.invokeCallBacks();
49          checkResult("Container: ROOT { TEXTAREA [ NAME = text SCROLLWIDTH = "
50                  + NumberWithUnit.ZERO + " SCROLLHEIGHT = "
51                  + NumberWithUnit.ZERO + " ]"
52                  + ActionManagerImpl.popupHandlerText(handler, builderData)
53                  + " }");
54      }
55  }
