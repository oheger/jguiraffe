463     /**
464      * A helper class for storing an expectation about a widget. This class
465      * stores either the name or the component for which a widget handler is
466      * expected to be queried. The widget handler to be returned is also stored.
467      */
468     private static class WidgetExpectation
469     {
470         /** The component. */
471         private final Object component;
472 
473         /** The name of the component. */
474         private final String componentName;
475 
476         /** The widget handler to return. */
477         private final WidgetHandler handler;
478 
479         /**
480          * Creates a new instance of {@code WidgetExpectation} and initializes
481          * it.
482          *
483          * @param comp the component
484          * @param name the name of the component
485          * @param h the widget handler
486          */
487         public WidgetExpectation(Object comp, String name, WidgetHandler h)
488         {
489             component = comp;
490             componentName = name;
491             handler = h;
492         }
493 
494         /**
495          * Processes a query for a widget handler for the specified component.
496          * Throws an exception if a different component is passed.
497          *
498          * @param comp the component
499          * @return the widget handler
500          */
501         public WidgetHandler queryHandler(Object comp)
502         {
503             assertEquals("Wrong component", component, comp);
504             return handler;
505         }
506 
507         /**
508          * Processes a query for a component for the specified component name.
509          * Throws an exception if a different name is passed.
510          *
511          * @param name the component name
512          * @return the component
513          */
514         public Object queryComponent(String name)
515         {
516             assertEquals("Wrong component name", componentName, name);
517             return component;
518         }
519     }

521     /**
522      * A test implementation of ComponentBuilderData that allows specifying
523      * expectations about the widget handlers to be queried.
524      */
525     private static class ComponentBuilderDataTestImpl extends
526             ComponentBuilderData
527     {
528         /** The list with the expectations. */
529         private final LinkedList<WidgetExpectation> expectations = new LinkedList<WidgetExpectation>();
530 
531         /**
532          * Adds an expectation for the specified component that should return
533          * the given handler.
534          *
535          * @param comp the expected component
536          * @param handler the widget handler
537          */
538         public void addExpectation(Object comp, WidgetHandler handler)
539         {
540             expectations.add(new WidgetExpectation(comp, null, handler));
541         }
542 
543         /**
544          * Adds an expectation for the specified component name that should
545          * return the given component.
546          *
547          * @param name the expected component name
548          * @param comp the corresponding component
549          */
550         public void addComponentExpectation(String name, Object comp)
551         {
552             expectations.add(new WidgetExpectation(comp, name, null));
553         }
554 
555         /**
556          * Returns the widget handler for the given component. Checks whether a
557          * corresponding expectation exists.
558          */
559         @Override
560         public WidgetHandler getWidgetHandlerForComponent(Object component)
561         {
562             return nextExpectation().queryHandler(component);
563         }
564 
565         /**
566          * Returns the component for the given name. Checks whether a
567          * corresponding expectation exists.
568          */
569         @Override
570         public Object getComponent(String name)
571         {
572             return nextExpectation().queryComponent(name);
573         }
574 
575         /**
576          * Returns the next expectation.
577          *
578          * @return the next expectation
579          */
580         private WidgetExpectation nextExpectation()
581         {
582             assertFalse("Too many queries for widget handlers", expectations
583                     .isEmpty());
584             return expectations.removeFirst();
585         }
586 
587         /**
588          * Verifies whether all expectations have been processed.
589          */
590         public void verify()
591         {
592             assertTrue("Remaining expectations: " + expectations, expectations
593                     .isEmpty());
594         }
595     }
