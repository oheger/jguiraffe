68      /**
69       * Convenience function to convert the collection of radio button handlers
70       * to an array.
71       *
72       * @param handlers the collection of handlers
73       * @return the corresponding array with handlers
74       */
75      private static WidgetHandler[] radioArray(List<WidgetHandler> handlers)
76      {
77          return handlers.toArray(new WidgetHandler[0]);
78      }

80      /**
81       * Convenience function to return the first radio button handler in the
82       * group managed by the given handler.
83       *
84       * @param groupHandler the handler
85       * @return the handler of the first radio button
86       */
87      private static WidgetHandler firstRadio(
88              RadioGroupWidgetHandler groupHandler)
89      {
90          return groupHandler.getRadioButtons().get(0);
91      }

397     /**
398      * Tests that a tool tip for the group can be set and removed again later.
399      */
400     @Test
401     public void testGroupToolTipCanBeRemoved()
402     {
403         final String radioTip1 = "This is my radio";
404         final String radioTip2 = "This is my other radio";
405         final String groupTip = "First group tip";
406         WidgetHandler radio1 = EasyMock.createMock(WidgetHandler.class);
407         WidgetHandler radio2 = EasyMock.createMock(WidgetHandler.class);
408         EasyMock.expect(radio1.getToolTip()).andReturn(radioTip1).anyTimes();
409         EasyMock.expect(radio2.getToolTip()).andReturn(radioTip2).anyTimes();
410         radio1.setToolTip(radioTip1 + TIP_SEPARATOR + groupTip);
411         radio1.setToolTip(radioTip1);
412         radio2.setToolTip(radioTip2 + TIP_SEPARATOR + groupTip);
413         radio2.setToolTip(radioTip2);
414         RadioGroupWidgetHandler groupHandler = new RadioGroupWidgetHandler(
415                 BUTTON_GROUP, Arrays.asList(radio1, radio2), TIP_SEPARATOR);
416         replayGroup(groupHandler);
417 
418         groupHandler.setToolTip(groupTip);
419         groupHandler.setToolTip(null);
420         verifyGroup(groupHandler);
421     }
