73      /**
74       * Creates the test node hierarchy.
75       */
76      @BeforeClass
77      public static void setUpBeforeClass() throws Exception
78      {
79          root = new HierarchicalConfiguration.Node();
80          ConfigurationNode ndTabs = new DefaultConfigurationNode("tables");
81          root.addChild(ndTabs);
82          for (int i = 0; i < TABLE_NAMES.length; i++)
83          {
84              ConfigurationNode ndTab = new DefaultConfigurationNode("table",
85                      TABLE_NAMES[i]);
86              ndTabs.addChild(ndTab);
87              for (int j = 0; j < TABLE_FIELDS[i].length; j++)
88              {
89                  ConfigurationNode ndField = new DefaultConfigurationNode(
90                          CHILD_NAME, TABLE_FIELDS[i][j]);
91                  ndTab.addChild(ndField);
92              }
93          }
94      }

96      /**
97       * Resolves the specified key and returns the corresponding node.
98       *
99       * @param key the key
100      * @return the node for this key
101      */
102     private static ConfigurationNode nodeForKey(String key)
103     {
104         HierarchicalConfiguration config = new HierarchicalConfiguration();
105         config.setRootNode(root);
106         List<?> nodes = config.getExpressionEngine().query(root, key);
107         assertEquals("Not unique key", 1, nodes.size());
108         return (ConfigurationNode) nodes.get(0);
109     }

306     /**
307      * Helper method for testing whether a path object contains the expected
308      * nodes.
309      *
310      * @param expected a list with the expected nodes
311      * @param path the path to check
312      */
313     private void checkPathNodes(List<ConfigurationNode> expected,
314             TreeNodePath path)
315     {
316         List<ConfigurationNode> actual = path.getNodes();
317         assertEquals("Wrong size of path", expected.size(), actual.size());
318         for (int i = 0; i < expected.size(); i++)
319         {
320             assertSame("Wrong node at " + i, expected.get(i), actual.get(i));
321         }
322     }
