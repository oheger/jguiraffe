16  package net.sf.jguiraffe.gui.builder.components.tags;
17  
18  /**
19   * Test class for ProgressBarTag.
20   *
21   * @author Oliver Heger
22   * @version $Id: TestProgressBarTag.java 205 2012-01-29 18:29:57Z oheger $
23   */
24  public class TestProgressBarTag extends AbstractTagTest
25  {
26  	/** Constant for the name of the test script. */
27  	private static final String SCRIPT = "progressbar";
28  
29  	/** Constant for the name of the created element. */
30  	private static final String ELEM_NAME = "p1";
31  
32  	/** Constant for the prefix of the expected result. */
33  	private static final String RESULT_PREFIX = "Container: ROOT { "
34  			+ "PROGRESSBAR [ NAME = " + ELEM_NAME;
35  
36  	/** Constant for the suffix of the expected result. */
37  	private static final String RESULT_SUFFIX = " ] }";
38  
39  	/** Constant for the expected result. */
40  	private static final String RESULT_TEXT = RESULT_PREFIX
41  			+ " MIN = 1 MAX = 200 " + "VALUE = 17 ALLOWTEXT = true TEXT = progress"
42  			+ RESULT_SUFFIX;
43  
44  	/** Constant for the expected default result. */
45  	private static final String RESULT_DEFAULT = RESULT_PREFIX
46  			+ " MIN = 0 MAX = 100 ALLOWTEXT = false" + RESULT_SUFFIX;
47  
48  	/** Constant for the test default builder. */
49  	private static final String BUILDER_DEFAULT = "TEST_DEFAULT";
50  
51  	/** Constant for the test text builder. */
52  	private static final String BUILDER_TEXT = "TEST_TEXT";
53  
54  	/** Constant for the test text res builder. */
55  	private static final String BUILDER_TEXTRES = "TEST_TEXTRES";
56  
57  	/** Constant for the test form builder. */
58  	private static final String BUILDER_TESTFORM = "TEST_FORM";
59  
60  	/**
61       * Helper method for performing a test. This method executes the test script
62       * with the specified builder and checks the expected result. It will also
63       * check if a component handler was created and, optionally, whether the
64       * static text element was not added to the form.
65       *
66       * @param builderName the name of the builder
67       * @param expected the expected result
68       * @param checkForm a flag whether the form is to be checked
69       * @throws Exception if an error occurs
70       */
71  	private void check(String builderName, String expected, boolean checkForm)
72  			throws Exception
73  	{
74  		builderData.setBuilderName(builderName);
75  		checkScript(SCRIPT, expected);
76  		assertNotNull("No component handler was created", builderData
77  				.getComponentHandler(ELEM_NAME));
78  		if (checkForm)
79  		{
80  			assertNull("Element was added to form", builderData.getForm()
81  					.getField(ELEM_NAME));
82  		}
83  	}
84  
85  	/**
86       * Tests creating a progress bar with all default values.
87       */
88  	public void testCreateDefault() throws Exception
89  	{
90  		check(BUILDER_DEFAULT, RESULT_DEFAULT, true);
91  	}
92  
93  	/**
94       * Tests creating a fully defined progress bar with a progress text.
95       */
96  	public void testCreateWithText() throws Exception
97  	{
98  		check(BUILDER_TEXT, RESULT_TEXT, true);
99  	}
100 
101 	/**
102      * Tests creating a fully defined progress bar with a progress text defined
103      * as a resource ID.
104      */
105 	public void testCreateWithTextRes() throws Exception
106 	{
107 		check(BUILDER_TEXTRES, RESULT_TEXT, true);
108 	}
109 
110 	/**
111      * Tests creating a progress bar that will be added to a form.
112      */
113 	public void testCreateWithForm() throws Exception
114 	{
115 		check(BUILDER_TESTFORM, RESULT_DEFAULT, false);
116 		assertNotNull("Element was not added to form", builderData.getForm()
117 				.getField(ELEM_NAME));
118 	}
119 }
