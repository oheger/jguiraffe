34  /**
35   * Test class for {@code RadioButtonTag}. This test class tests {@code
36   * RadioButtonTag} directly. There is also another test class which executes a
37   * test script.
38   *
39   * @author Oliver Heger
40   * @version $Id: TestRadioButtonTag.java 205 2012-01-29 18:29:57Z oheger $
41   */
42  public class TestRadioButtonTag
43  {
44      /** Constant for the name of the radio group. */
45      private static final String GROUP = "testRadioGroup";
46  
47      /** Constant for the name of the radio button. */
48      private static final String NAME = "radio";
49  
50      /** The Jelly context. */
51      private JellyContext context;
52  
53      /** The tag to be tested. */
54      private RadioButtonTag tag;
55  
56      @Before
57      public void setUp() throws Exception
58      {
59          context = new JellyContext();
60          tag = new RadioButtonTag();
61          tag.setContext(context);
62          tag.setName(NAME);
63          ComponentBuilderData compData = new ComponentBuilderData();
64          compData.initializeForm(new TransformerContextImpl(),
65                  new BeanBindingStrategy());
66          compData.setComponentManager(new ComponentManagerImpl());
67          compData.put(context);
68      }
69  
70      /**
71       * Creates a radio group tag and sets it as parent of the test tag.
72       *
73       * @return the parent tag
74       */
75      private RadioGroupTag setUpParent()
76      {
77          RadioGroupTag parent = new RadioGroupTag();
78          try
79          {
80              parent.setContext(context);
81              parent.setName(GROUP);
82              ComponentGroup.createGroup(context, GROUP);
83              tag.setParent(parent);
84          }
85          catch (Exception ex)
86          {
87              fail("Exception when preparing parent tag: " + ex);
88          }
89          return parent;
90      }
91  
92      /**
93       * Tests the default value of the noField attribute.
94       */
95      @Test
96      public void testIsNoFieldDefault()
97      {
98          assertTrue("A field", tag.isNoField());
99      }
100 
101     /**
102      * Tests whether the radio button is added to the radio group if it is in
103      * the body of a group tag.
104      */
105     @Test
106     public void testProcessBeforeBodyAddGroup() throws FormBuilderException,
107             JellyTagException
108     {
109         setUpParent();
110         tag.processBeforeBody();
111         ComponentGroup group = ComponentGroup.fromContext(context, GROUP);
112         assertEquals("Wrong number of elements", 1, group.getComponentNames()
113                 .size());
114         assertTrue("Component not found", group.getComponentNames().contains(
115                 NAME));
116     }
117 
118     /**
119      * Tests whether the name attribute is already checked before processing of
120      * the body.
121      */
122     @Test(expected = MissingAttributeException.class)
123     public void testProcessBeforeBodyNoName() throws FormBuilderException,
124             JellyTagException
125     {
126         setUpParent();
127         tag.setName(null);
128         tag.processBeforeBody();
129     }
