16  package net.sf.jguiraffe.gui.builder.components.tags;

18  import net.sf.jguiraffe.gui.layout.NumberWithUnit;
19  
20  /**
21   * Test class for SplitterTag.
22   *
23   * @author Oliver Heger
24   * @version $Id: TestSplitterTag.java 205 2012-01-29 18:29:57Z oheger $
25   */
26  public class TestSplitterTag extends AbstractTagTest
27  {
28      /** Constant for the name of the test script. */
29      private static final String SCRIPT = "splitter";
30  
31      /** Constant for the prefix of the expected result string. */
32      private static final String PREFIX = "Container: ROOT { SPLITTER [ ORIENTATION = ";
33  
34      /** Constant for the text representation of a component. */
35      private static final String COMP = "TEXTAREA [ NAME = comp%d SCROLLWIDTH = "
36              + NumberWithUnit.ZERO
37              + " SCROLLHEIGHT = "
38              + NumberWithUnit.ZERO
39              + " ]";
40  
41      /** Constant for the simple builder. */
42      private static final String BUILDER_SIMPLE = "TEST_SIMPLE";
43  
44      /** Constant for the nested builder. */
45      private static final String BUILDER_NESTED = "TEST_NESTED";
46  
47      /** Constant for the error too few builder. */
48      private static final String BUILDER_ERR_TOO_FEW = "ERR_TOO_FEW";
49  
50      /** Constant for the error too many builder. */
51      private static final String BUILDER_ERR_TOO_MANY = "ERR_TOO_MANY";
52  
53      /** Constant for the error invalid size builder. */
54      private static final String BUILDER_ERR_SIZE = "ERR_SIZE";
55  
56      /** Constant for the error invalid resize weight builder. */
57      private static final String BUILDER_ERR_RESIZEWEIGHT = "ERR_RESIZEWEIGHT";
58  
59      /** Constant for the error invalid orientation builder. */
60      private static final String BUILDER_ERR_ORIENTATION = "ERR_ORIENTATION";
61  
62      /**
63       * Tests creating a simple splitter.
64       */
65      public void testSimpleSplitter() throws Exception
66      {
67          builderData.setBuilderName(BUILDER_SIMPLE);
68          checkScript(SCRIPT, PREFIX
69                  + "VERTICAL POS = 50 SIZE = 10 RESIZEWEIGHT = 0 ] { " + comp(1)
70                  + ", " + comp(2) + " } }");
71      }
72  
73      /**
74       * Tests a splitter nested within another splitter.
75       */
76      public void testNestedSplitter() throws Exception
77      {
78          builderData.setBuilderName(BUILDER_NESTED);
79          checkScript(SCRIPT, PREFIX
80                  + "VERTICAL POS = 50 SIZE = 10 RESIZEWEIGHT = 50 ] { "
81                  + comp(1) + ", "
82                  + "SPLITTER [ ORIENTATION = HORIZONTAL POS = 25 "
83                  + "RESIZEWEIGHT = 100 ] { " + comp(2) + ", " + comp(3)
84                  + " } } }");
85      }
86  
87      /**
88       * Tests a splitter with too few child components.
89       */
90      public void testTooFewComponents() throws Exception
91      {
92          errorScript(SCRIPT, BUILDER_ERR_TOO_FEW,
93                  "Could create splitter with too few components!");
94      }
95  
96      /**
97       * Tests a splitter with too many child components.
98       */
99      public void testTooManyComponents() throws Exception
100     {
101         errorScript(SCRIPT, BUILDER_ERR_TOO_MANY,
102                 "Could create splitter with too many components!");
103     }
104 
105     /**
106      * Tests a splitter with an invalid size attribute.
107      */
108     public void testInvalidSize() throws Exception
109     {
110         errorScript(SCRIPT, BUILDER_ERR_SIZE, "Invalid size was not detected!");
111     }
112 
113     /**
114      * Tests a splitter with an invalid resizeWeight attribute.
115      */
116     public void testInvalidResizeWeight() throws Exception
117     {
118         errorScript(SCRIPT, BUILDER_ERR_RESIZEWEIGHT,
119                 "Invalid resizeWeight was not detected!");
120     }
121 
122     /**
123      * Tests a splitter with an invalid orientation attribute.
124      */
125     public void testInvalidOrientation() throws Exception
126     {
127         errorScript(SCRIPT, BUILDER_ERR_ORIENTATION,
128                 "Invalid orientation not detected!");
129     }
130 
131     /**
132      * Creates the text representation of a component with the given index.
133      *
134      * @param nr the index
135      * @return the text representation for this component
136      */
137     private static String comp(int nr)
138     {
139         return String.format(COMP, nr);
140     }
