16  package net.sf.jguiraffe.gui.builder.components.tags;
17  
18  import net.sf.jguiraffe.gui.layout.NumberWithUnit;
19  
20  /**
21   * Test class for TextAreaTag.
22   *
23   * @author Oliver Heger
24   * @version $Id: TestTextAreaTag.java 205 2012-01-29 18:29:57Z oheger $
25   */
26  public class TestTextAreaTag extends AbstractTagTest
27  {
28      /** Constant for the name of the test script. */
29      private static final String SCRIPT = "textarea";
30  
31      /** Constant for the simple text area builder. */
32      private static final String BUILDER_SIMPLE = "BUILDER_SIMPLE";
33  
34      /** Constant for the text area columns and rows builder. */
35      private static final String BUILDER_COLS = "BUILDER_COLS";
36  
37      /** Constant for the text area maximum length builder. */
38      private static final String BUILDER_LEN = "BUILDER_LEN";
39  
40      /** Constant for the text area scroll size builder. */
41      private static final String BUILDER_SCROLL = "BUILDER_SCROLLSIZE";
42  
43      /** Constant for the name prefix of text area components. */
44      private static final String COMP_PREFIX = "desc";
45  
46      /** Constant for the format for the expected script. */
47      private static final String EXPECTED_FMT = "Container: ROOT { TEXTAREA [ "
48              + "NAME = " + COMP_PREFIX + "%d%s ] }";
49  
50      /** Constant for the default scroll sizes. */
51      private static final String SCROLL_SIZES = " SCROLLWIDTH = "
52              + NumberWithUnit.ZERO + " SCROLLHEIGHT = " + NumberWithUnit.ZERO;
53  
54      /**
55       * Helper method for executing a test script.
56       *
57       * @param builder the name of the builder
58       * @param idx the index of the component
59       * @param txtAreaContent the expected content of the text area declaration
60       * @throws Exception if an error occurs
61       */
62      private void checkScript(String builder, int idx, String txtAreaContent)
63              throws Exception
64      {
65          builderData.setBuilderName(builder);
66          checkScript(SCRIPT, String.format(EXPECTED_FMT, idx, txtAreaContent));
67          assertNotNull("Component handler not found", builderData
68                  .getFieldHandler(COMP_PREFIX + idx));
69      }
70  
71      /**
72       * Tests the creation of a simple text area.
73       */
74      public void testCreateTextAreaSimple() throws Exception
75      {
76          checkScript(BUILDER_SIMPLE, 1, SCROLL_SIZES);
77      }
78  
79      /**
80       * Tests whether columns and rows attributes are taken into account.
81       */
82      public void testCreateTextAreaColRows() throws Exception
83      {
84          checkScript(BUILDER_COLS, 2, " COLUMNS = 40 ROWS = 5 WRAP = YES"
85                  + SCROLL_SIZES);
86      }
87  
88      /**
89       * Tests whether maximum length and wrap attributes are taken into account.
90       */
91      public void testCreateTextAreaLen() throws Exception
92      {
93          checkScript(BUILDER_LEN, 3, " MAXLEN = 300" + SCROLL_SIZES);
94      }
95  
96      /**
97       * Tests whether the scroll size attributes are taken into account.
98       */
99      public void testCreateTextAreaScrollSize() throws Exception
100     {
101         checkScript(BUILDER_SCROLL, 4,
102                 " SCROLLWIDTH = NumberWithUnit [ 2.0in ]"
103                         + " SCROLLHEIGHT = NumberWithUnit [ 3.0cm ]");
104     }
105 }
