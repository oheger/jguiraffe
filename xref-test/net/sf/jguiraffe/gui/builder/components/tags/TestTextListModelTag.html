26  public class TestTextListModelTag extends AbstractTagTest

28      /** Constant for the name of the test script. */
29      private static final String SCRIPT = "textlistmodel";
30  
31      /** Constant for a variable that is referenced by an item tag. */
32      private static final String VAR_NAME = "intConst";
33  
34      /** Constant for a test value referenced by a list item tag. */
35      private static final Object ITEM_VALUE = 42;
36  
37      /**
38       * Helper method for executing a builder script.
39       *
40       * @throws Exception if an error occurs
41       */
42      private void executeBuilderOperation() throws Exception
43      {
44          context.setVariable(VAR_NAME, ITEM_VALUE);
45          executeScript(SCRIPT);
46      }

48      /**
49       * Tests creating some list models.
50       */
51      public void testCreateModels() throws Exception
52      {
53          executeBuilderOperation();
54          ListModel model = (ListModel) context.getVariable("model1");
55          assertNotNull("Model 1 not found", model);
56          assertEquals("Wrong number of items in model 1", 3, model.size());
57          assertEquals("Wrong value class", Integer.class, model.getType());
58          for (int i = 1; i <= model.size(); i++)
59          {
60              assertEquals("Wrong item in model 1 at " + i, "item" + i, model
61                      .getDisplayObject(i - 1));
62              assertEquals("Wrong value in model 1 at " + i, Integer
63                      .valueOf(i - 1), ListModelUtils.getValue(model, i - 1));
64          }
65  
66          model = (ListModel) context.getVariable("modelPasta");
67          assertNotNull("Model 2 not found", model);
68          assertEquals("Wrong number of items in model 2", 4, model.size());
69          assertEquals("Wrong item class in model 2", String.class, model
70                  .getType());
71          assertEquals("Wrong item in model 2 at 0", "Spaghetti", model
72                  .getDisplayObject(0));
73          assertEquals("Wrong item in model 2 at 2", "Penne", model
74                  .getDisplayObject(2));
75          assertEquals("Wrong item in model 2 at 3", "Taggliatelle", model
76                  .getDisplayObject(3));
77          assertEquals("Wrong value in model 2 at 1", "2", model
78                  .getValueObject(1));
79          assertEquals("Wrong value in model 2 at 3", String.valueOf(ITEM_VALUE),
80                  model.getValueObject(3));
81  
82          model = (ListModel) context.getVariable("modelMix");
83          assertNotNull("Model 3 not found", model);
84          assertEquals("Wrong number of elelements in model 3", 4, model.size());
85          assertEquals("Wrong item class in model 3", String.class, model
86                  .getType());
87          assertEquals("Wrong item int model 3 at 0", "onions", model
88                  .getDisplayObject(0));
89          assertNull("Got value in model 3 at 0", model.getValueObject(0));
90          assertEquals("Wrong item in model 3 at 1", "broccoli", model
91                  .getDisplayObject(1));
92          assertEquals("Wrong value in model 3 at 1", "x", model
93                  .getValueObject(1));
94          assertEquals("Wrong item in model 3 at 2", "extra cheese", model
95                  .getDisplayObject(2));
96          assertNull("Got value in model 3 at 2", model.getValueObject(2));
97          assertEquals("Wrong item in model 3 at 3", "tuna", model
98                  .getDisplayObject(3));
99          assertEquals("Wrong value in model 3 at 3", "y", model
100                 .getValueObject(3));
101 
102         model = (ListModel) context.getVariable("modelEmpty");
103         assertNotNull("Model 4 not found", model);
104         assertEquals("Got elements in empty model", 0, model.size());
105     }
