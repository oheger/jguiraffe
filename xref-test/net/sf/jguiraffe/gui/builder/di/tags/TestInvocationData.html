245     /**
246      * Tests registration and unregistration of support objects.
247      */
248     public void testUnregisterInvokableSupport() throws JellyTagException
249     {
250         InvokableSupport support = EasyMock.createMock(InvokableSupport.class);
251         Invokable inv1 = EasyMock.createMock(Invokable.class);
252         Invokable inv2 = EasyMock.createMock(Invokable.class);
253         ChainedInvocation chain = new ChainedInvocation();
254         support.addInvokable(inv1);
255         EasyMock.replay(inv1, inv2, support);
256         data.registerInvokableSupport(chain);
257         data.registerInvokableSupport(support);
258         data.addInvokable(inv1, null, null);
259         data.unregisterInvokableSupport();
260         data.addInvokable(inv2, "myResult", "mySource");
261         data.unregisterInvokableSupport();
262         assertEquals("Nothing added to chain", 1, chain.size());
263         EasyMock.verify(inv1, inv2, support);
264     }
