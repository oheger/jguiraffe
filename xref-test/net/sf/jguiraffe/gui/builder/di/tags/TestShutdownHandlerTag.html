25  import org.apache.commons.jelly.JellyContext;

30  /**
31   * Test class for {@link ShutdownHandlerTag}.
32   *
33   * @author Oliver Heger
34   * @version $Id: TestShutdownHandlerTag.java 208 2012-02-11 20:57:33Z oheger $
35   */
36  public class TestShutdownHandlerTag
37  {
38      /** The parent tag of the test tag. */
39      private BeanTag parent;
40  
41      /** The tag to be tested. */
42      private ShutdownHandlerTag tag;
43  
44      @Before
45      public void setUp() throws Exception
46      {
47          JellyContext context = new JellyContext();
48          parent = new BeanTag();
49          parent.setContext(context);
50          tag = new ShutdownHandlerTag();
51          tag.setContext(context);
52          tag.setParent(parent);
53      }
54  
55      /**
56       * Tests a tag with a wrong parent. This should cause an exception.
57       */
58      @Test(expected = JellyTagException.class)
59      public void testProcessBeforeBodyWrongParent() throws JellyTagException
60      {
61          tag.setParent(null);
62          tag.processBeforeBody();
63      }
64  
65      /**
66       * Tests processing the tag and registering invokables.
67       */
68      @Test
69      public void testProcess() throws JellyTagException
70      {
71          tag.processBeforeBody();
72          InvocationData.get(tag.getContext()).addInvokable(
73                  new SetPropertyInvocation("intProp", ConstantBeanProvider
74                          .getInstance(0)), null, null);
75          InvocationData.get(tag.getContext()).addInvokable(
76                  new SetPropertyInvocation("stringProp",
77                          ConstantBeanProvider.NULL), null, null);
78          tag.process();
79          ChainedInvocation inv = (ChainedInvocation) parent.getShutdownHandler();
80          assertEquals("Wrong number of invocations", 2, inv.size());
81      }
82  
83      /**
84       * Tests whether the invocation script is unregistered when the tag ends.
85       */
86      @Test
87      public void testProcessUnregisterInvokableSupport()
88              throws JellyTagException
89      {
90          tag.processBeforeBody();
91          tag.process();
92          try
93          {
94              InvocationData.get(tag.getContext()).addInvokable(
95                      HelperInvocations.NULL_INVOCATION, null, null);
96              fail("Could still add invocations!");
97          }
98          catch (JellyTagException jtex)
99          {
100             // ok
101         }
102     }
103 }
