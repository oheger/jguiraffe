480     /**
481      * Tests whether the root store is correctly created and linked into the
482      * hierarchy of bean stores.
483      */
484     @Test
485     public void testInitBuilderBeanContextHierarchy()
486     {
487         JellyContext ctx = prepareRootStoreTest();
488         MutableBeanStore rootStore =
489                 builder.initBuilderBeanContext(this, ctx,
490                         new InvocationHelper());
491         BeanContext beanContext =
492                 ComponentBuilderData.get(ctx).getBeanContext();
493         assertTrue("Wrong parent of root store",
494                 rootStore.getParent() instanceof SimpleBeanStoreImpl);
495         assertTrue("BeanContext not initialized",
496                 beanContext.getDefaultBeanStore() instanceof CombinedBeanStore);
497         assertEquals("Wrong parent of component data", getParentContext()
498                 .getDefaultBeanStore(), rootStore.getParent().getParent());
499         assertTrue("Wrong context class " + beanContext,
500                 beanContext instanceof BeanContextWrapper);
501         CombinedBeanStore cbs =
502                 (CombinedBeanStore) beanContext.getDefaultBeanStore();
503         assertEquals("Wrong number of child stores", 2, cbs.size());
504         assertTrue("Wrong class of 1st child store " + cbs.getChildStore(0),
505                 cbs.getChildStore(0) instanceof MutableBeanStore);
506         assertTrue("Wrong class of 2nd child store",
507                 cbs.getChildStore(1) instanceof JellyContextBeanStore);
508         assertEquals("Builder context has not been set", beanContext,
509                 getBuilderContext());
510     }

1414     /**
1415      * Helper method for comparing two maps. (We cannot use equals here as the
1416      * maps may be of different types.)
1417      *
1418      * @param <K> the type of the keys
1419      * @param <V> the type of the values
1420      * @param map1 the first map
1421      * @param map2 the second map
1422      */
