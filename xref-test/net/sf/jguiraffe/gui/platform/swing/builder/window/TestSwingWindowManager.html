445     /**
446      * Tests whether a key stroke for escape is registered if necessary.
447      */
448     @Test
449     public void testInitSwingWindowPropertiesCloseOnEsc()
450     {
451         Window parent = EasyMock.createMock(Window.class);
452         SwingWindow window = prepareInitPropertiesTest(parent);
453         JRootPane rootPane = new JRootPane();
454         EasyMock.expect(window.getRootPane()).andReturn(rootPane);
455         EasyMock.expect(window.close(false)).andReturn(Boolean.TRUE);
456         EasyMock.replay(window, parent);
457         WindowDataImpl data = createWindowData(false);
458         data.setCloseOnEsc(true);
459         manager.initSwingWindowProperties(window, data, parent);
460         Object obj = rootPane.getInputMap(JComponent.WHEN_IN_FOCUSED_WINDOW)
461                 .get(KeyStroke.getKeyStroke(KeyEvent.VK_ESCAPE, 0));
462         assertNotNull("No key stroke registered for ESC", obj);
463         Action action = rootPane.getActionMap().get(obj);
464         assertNotNull("No action registered", action);
465         action.actionPerformed(null);
466         EasyMock.verify(window, parent);
467     }

572      * Checks all attributes of the given internal frame.
573      *
574      * @param window the internal frame to check
575      * @param data the expected data
576      * @param center the expected center flag
577      */
578     private void checkInternalFrame(SwingWindow window, WindowDataImpl data,
579             boolean center)
580     {
581         assertTrue("Window is no internal frame",
582                 window instanceof InternalFrameAdapter);
583         checkWindow(window, data, center);
584         JInternalFrame iframe = (InternalFrameAdapter) window;
585         assertEquals("Wrong internal frame close operation",
586                 JInternalFrame.DO_NOTHING_ON_CLOSE,
587                 iframe.getDefaultCloseOperation());
588         assertEquals("Wrong menu", data.getMenuBar(), iframe.getJMenuBar());
589         assertEquals("Wrong resizable flag", data.isResizable(),
590                 iframe.isResizable());
591         assertEquals("Wrong closable flag", data.isClosable(),
592                 iframe.isClosable());
593         assertEquals("Wrong maximizable flag", data.isMaximizable(),
594                 iframe.isMaximizable());
595         assertEquals("Wrong iconifiable flag", data.isIconifiable(),
596                 iframe.isIconifiable());
597         assertEquals("Wrong icon", data.getIcon(), iframe.getFrameIcon());
598         assertTrue("Wrong parent window",
599                 builderData.getParentWindow() instanceof FrameAdapter);
600         JDesktopPane desktop = WindowHelper
601                 .findDesktopPane(((FrameAdapter) builderData.getParentWindow())
602                         .getFrame());
603         assertNotNull("No desktop pane", desktop);
604         assertEquals("Internal frame was not added to desktop pane", 1,
605                 desktop.getComponentCount());
606     }
