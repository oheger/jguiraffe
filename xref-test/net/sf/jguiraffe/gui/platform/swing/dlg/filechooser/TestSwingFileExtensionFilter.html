39      /**
40       * Tests whether the filter description is resolved correctly.
41       */
42      @Test
43      public void testDescriptionIsResolved()
44      {
45          final Object resID = "myResourceID";
46          final String description = "This is my filter";
47          ApplicationContext appCtx =
48                  EasyMock.createMock(ApplicationContext.class);
49          EasyMock.expect(appCtx.getResourceText(resID)).andReturn(description);
50          EasyMock.replay(appCtx);
51          FileExtensionFilter filter =
52                  new FileExtensionFilter(TextResource.fromResourceID(resID));
53  
54          SwingFileExtensionFilter swingFilter =
55                  SwingFileExtensionFilter.fromExtensionFilter(filter, appCtx);
56          assertEquals("Wrong description", description,
57                  swingFilter.getDescription());
58      }

63      @Test
64      public void testAcceptTrue()
65      {
66          File[] files = new File[] {
67                  new File("image1.jpg"), new File("image2.JPG"),
68                  new File("image3.jpeg"), new File("image.3.jpg"),
69                  new File("test/imageSub.JpEg")
70          };
71          FileExtensionFilter filter =
72                  new FileExtensionFilter(TextResource.UNDEFINED, "jpg", "jpeg");
73  
74          SwingFileExtensionFilter swingFilter =
75                  SwingFileExtensionFilter.fromExtensionFilter(filter,
76                          EasyMock.createMock(ApplicationContext.class));
