96      /**
97       * Creates a validation result indicating an error.
98       *
99       * @return the result object
100      */
101     private ValidationResult errorResult()
102     {
103         ValidationMessage vmsg = EasyMock.createMock(ValidationMessage.class);
104         EasyMock.expect(vmsg.getMessage()).andStubReturn(ERR_MSG + errCounter);
105         EasyMock.expect(vmsg.getLevel())
106                 .andReturn(ValidationMessageLevel.ERROR).anyTimes();
107         errCounter++;
108         EasyMock.replay(vmsg);
109         return new DefaultValidationResult.Builder().addValidationMessage(vmsg).build();
110     }

138     /**
139      * Tests a wrapped context.
140      *
141      * @param c the context to check
142      * @param ctx the expected parent context
143      */
144     private void checkWrappedContext(TransformerContext c,
145             TransformerContext ctx)
146     {
147         assertTrue("Wrong context returned: " + c,
148                 c instanceof TransformerContextPropertiesWrapper);
149         TransformerContextPropertiesWrapper cw = (TransformerContextPropertiesWrapper) c;
150         assertEquals("Wrong wrapped context", ctx, cw.getWrappedContext());
151         Map<String, Object> props = createTestProperties();
152         Map<String, Object> props2 = cw.properties();
153         for (String key : props.keySet())
154         {
155             assertEquals("Wrong property " + key, props.get(key), props2
156                     .get(key));
157         }
158         EasyMock.verify(ctx);
159     }

282     /**
283      * Tests the isValid() method in short evaluation mode.
284      */
285     @Test
286     public void testIsValidShortEvaluation()
287     {
288         TransformerContext ctx = setUpContext();
289         final int count = 10;
290         final int errIdx = 4;
291         initChildValidators(count);
292         for (int i = 0; i < errIdx; i++)
293         {
294             EasyMock.expect(children[i].isValid(TEST_OBJ, ctx)).andReturn(
295                     DefaultValidationResult.VALID);
296         }
297         EasyMock.expect(children[errIdx].isValid(TEST_OBJ, ctx)).andReturn(
298                 errorResult());
299         replay();
300         ValidationResult vr = validator.isValid(TEST_OBJ, ctx);
301         assertFalse("Wrong validation result", vr.isValid());
302         Collection<ValidationMessage> msgs = vr.getValidationMessages();
303         assertEquals("Wrong number of messages", 1, msgs.size());
304         ValidationMessage msg = msgs.iterator().next();
305         assertEquals("Wrong message", ERR_MSG + "0", msg.getMessage());
306         verify();
307         EasyMock.verify(ctx);
308     }

310     /**
311      * Tests the isValid() method when short evaluation is turned off.
312      */
313     @Test
314     public void testIsValidNoShortEvaluation()
315     {
316         TransformerContext ctx = setUpContext();
317         final Integer[] errIdx = {
318                 1, 5, 6, 10
319         };
320         final int count = 16;
321         initChildValidators(count);
322         validator.setShortEvaluation(false);
323         Set<Integer> errIndices = new HashSet<Integer>(Arrays.asList(errIdx));
324         for (int i = 0; i < count; i++)
325         {
326             ValidationResult vr = errIndices.contains(i) ? errorResult()
327                     : DefaultValidationResult.VALID;
328             EasyMock.expect(children[i].isValid(TEST_OBJ, ctx)).andReturn(vr);
329         }
330         replay();
331         ValidationResult vr = validator.isValid(TEST_OBJ, ctx);
332         assertFalse("Wrong validation result", vr.isValid());
333         Collection<ValidationMessage> msgs = vr.getValidationMessages();
334         assertEquals("Wrong number of messages", errIdx.length, msgs.size());
335         Iterator<ValidationMessage> it = msgs.iterator();
336         for (int i = 0; i < errIdx.length; i++)
337         {
338             ValidationMessage msg = it.next();
339             assertEquals("Wrong message at " + i, ERR_MSG + i, msg.getMessage());
340         }
341         verify();
342         EasyMock.verify(ctx);
343     }
