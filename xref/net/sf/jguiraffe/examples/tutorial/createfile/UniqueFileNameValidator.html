16  package net.sf.jguiraffe.examples.tutorial.createfile;
17  
18  import net.sf.jguiraffe.examples.tutorial.model.DirectoryData;
19  import net.sf.jguiraffe.examples.tutorial.model.FileData;
20  import net.sf.jguiraffe.transform.DefaultValidationResult;
21  import net.sf.jguiraffe.transform.TransformerContext;
22  import net.sf.jguiraffe.transform.ValidationMessage;
23  import net.sf.jguiraffe.transform.ValidationResult;
24  import net.sf.jguiraffe.transform.Validator;
25  
26  /**
27   * <p>
28   * A specialized {@link Validator} implementations for checking whether the name
29   * entered for a new file does not exist in the current directory.
30   * </p>
31   * <p>
32   * In the validation method this class obtains the data object with the content
33   * of the current directory (which is available as a typed property from the
34   * context). Then it checks whether the name entered by the user already exists
35   * in this directory. If this is the case, validation fails and an error message
36   * is produced.
37   * </p>
38   *
39   * @author Oliver Heger
40   * @version $Id: UniqueFileNameValidator.java 205 2012-01-29 18:29:57Z oheger $
41   */
42  public class UniqueFileNameValidator implements Validator
43  {
44      /** Constant for the error message produced by this validator. */
45      public static final String ERR_FILE_EXISTS = "ERR_FILE_EXISTS";
46  
47      /**
48       * Performs the validation.
49       *
50       * @param o the object to be validated (this is the file name; <b>null</b>
51       *        names are accepted)
52       * @param ctx the validation context
53       * @return the validation results
54       */
55      @Override
56      public ValidationResult isValid(Object o, TransformerContext ctx)
57      {
58          if (o != null) // null is okay
59          {
60              DirectoryData dirData = ctx.getTypedProperty(DirectoryData.class);
61              assert dirData != null : "No current directory!";
62              String fileName = o.toString();
63  
64              for (FileData fd : dirData.getContent())
65              {
66                  if (fileName.equalsIgnoreCase(fd.getName()))
67                  {
68                      ValidationMessage msg = ctx.getValidationMessageHandler()
69                              .getValidationMessage(ctx, ERR_FILE_EXISTS,
70                                      fileName);
71                      return new DefaultValidationResult.Builder()
72                              .addValidationMessage(msg).build();
73                  }
74              }
75          }
76  
77          return DefaultValidationResult.VALID;
78      }
79  }
