16  package net.sf.jguiraffe.examples.tutorial.viewset;
17  
18  import java.util.Map;
19  
20  import net.sf.jguiraffe.gui.forms.DefaultFormValidatorResults;
21  import net.sf.jguiraffe.gui.forms.Form;
22  import net.sf.jguiraffe.gui.forms.FormValidator;
23  import net.sf.jguiraffe.gui.forms.FormValidatorResults;
24  import net.sf.jguiraffe.transform.ValidationResult;
25  
26  /**
27   * <p>
28   * A {@code FormValidator} implementation for validating the dialog with view
29   * settings.
30   * </p>
31   * <p>
32   * This class performs specific validations that involve multiple fields and
33   * thus cannot be achieved by simple {@code Validator} implementations. The
34   * following validation rules are implemented:
35   * <ul>
36   * <li>If the check box for filtering file types is checked, at least one file
37   * type must have been selected.</li>
38   * <li>If the check box for filtering by file size is checked, a valid file
39   * minimum size must have been entered.</li>
40   * <li>If the check box for filtering by file date is checked, the from date
41   * and/or the to date must have been provided.</li>
42   * </ul>
43   * </p>
44   *
45   * @author Oliver Heger
46   * @version $Id: ViewSettingsFormValidator.java 205 2012-01-29 18:29:57Z oheger $
47   */
48  public class ViewSettingsFormValidator implements FormValidator
49  {
50      /** Constant for the field for the file types. */
51      private static final String FLD_FILETYPES = "fileTypes";
52  
53      /** Constant for the field for the minimum file size. */
54      private static final String FLD_MINFILESIZE = "minFileSize";
55  
56      /** Constant for the field with the from date. */
57      private static final String FLD_FROMDATE = "fileDateFrom";
58  
59      /** Constant for the error message "no file types". */
60      private static final String ERR_NOFILETYPES = "ERR_NOFILETYPES";
61  
62      /** Constant for the error message "no minimum file size". */
63      private static final String ERR_NOFILESIZE = "ERR_NOFILESIZE";
64  
65      /** Constant for the error message "no date range". */
66      private static final String ERR_NOFILEDATE = "ERR_NOFILEDATE";
67  
68      /** Constant for the error message "invalid date range". */
69      private static final String ERR_FILEDATEINV = "ERR_FILEDATEINV";
70  
71      /**
72       * Validates the specified {@code Form} instance.
73       *
74       * @param form the {@code Form}
75       * @return an object with the results of the validation
76       */
77      @Override
78      public FormValidatorResults isValid(Form form)
79      {
80          // Create a map with valid result objects as basis
81          Map<String, ValidationResult> resultMap = DefaultFormValidatorResults
82                  .validResultMapForForm(form);
83  
84          // Obtain the data entered into the form
85          ViewSettings settings = new ViewSettings();
86          form.readFields(settings);
87  
88          // Now perform the specific checks
89          if (settings.isFilterTypes())
90          {
91              if (settings.getFileTypes().length < 1)
92              {
93                  resultMap.put(FLD_FILETYPES, DefaultFormValidatorResults
94                          .createValidationErrorResult(form, ERR_NOFILETYPES));
95              }
96          }
97  
98          if (settings.isFilterSize())
99          {
100             if (settings.getMinFileSize() == null)
101             {
102                 resultMap.put(FLD_MINFILESIZE, DefaultFormValidatorResults
103                         .createValidationErrorResult(form, ERR_NOFILESIZE));
104             }
105         }
106 
107         if (settings.isFilterDate())
108         {
109             if (settings.getFileDateFrom() == null
110                     && settings.getFileDateTo() == null)
111             {
112                 resultMap.put(FLD_FROMDATE, DefaultFormValidatorResults
113                         .createValidationErrorResult(form, ERR_NOFILEDATE));
114             }
115             else
116             {
117                 if (settings.getFileDateFrom() != null
118                         && settings.getFileDateTo() != null
119                         && settings.getFileDateTo().before(
120                                 settings.getFileDateFrom()))
121                 {
122                     resultMap
123                             .put(FLD_FROMDATE, DefaultFormValidatorResults
124                                     .createValidationErrorResult(form,
125                                             ERR_FILEDATEINV));
126                 }
127             }
128         }
129 
130         // Return the validation results
131         return new DefaultFormValidatorResults(resultMap);
132     }
133 }
