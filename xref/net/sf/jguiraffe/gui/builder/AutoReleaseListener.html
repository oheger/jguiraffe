16  package net.sf.jguiraffe.gui.builder;

18  import net.sf.jguiraffe.gui.builder.window.WindowEvent;
19  import net.sf.jguiraffe.gui.builder.window.WindowListener;
20  
21  /**
22   * <p>
23   * A specialized {@code WindowListener} implementation for implementing the
24   * <em>auto release</em> mechanism supported by builders.
25   * </p>
26   * <p>
27   * The results produced by a {@link Builder} usually need to be released when
28   * they are no longer needed to free the resources used by them. This can be
29   * done manually by invoking the {@link Builder#release(BuilderData)} method
30   * with the {@link BuilderData} object used for specifying the parameters of the
31   * builder operation.
32   * </p>
33   * <p>
34   * When a window is generated by a builder script, there is an alternative: it
35   * is possible to let the framework perform the {@code release()} invocation
36   * automatically when the window is closed. This is often useful, for instance
37   * when a dialog window is created from a builder script: when the user closes
38   * the dialog all resources associated with it (including beans defined in the
39   * script) are no more needed and can be released. To enable this mechanism the
40   * {@code autoRelease} property of {@link BuilderData} has to be set to
41   * <b>true</b>.
42   * </p>
43   * <p>
44   * This class implements the functionality for automatically releasing a
45   * {@link BuilderData} object when a window (i.e. the main window produced by
46   * the builder script) is closed. It does not rely on a concrete {@link Builder}
47   * implementation and thus can be used with arbitrary implementations. Concrete
48   * implementations of the {@link Builder} interface only need to create an
49   * instance and register it at the main window of the builder results.
50   * </p>
51   * <p>
52   * Client code typically will not have to use this class directly. The framework
53   * will create and initialize and instance behind the scenes when a
54   * {@link BuilderData} object configured appropriately is passed to a
55   * {@link Builder}.
56   * </p>
57   *
58   * @author Oliver Heger
59   * @version $Id: AutoReleaseListener.java 205 2012-01-29 18:29:57Z oheger $
60   */
61  public class AutoReleaseListener implements WindowListener
62  {
63      /** Stores a reference to the builder data object to be released. */
64      private final BuilderData builderData;
65  
66      /** Stores the builder that must be called for releasing. */
67      private final Builder builder;
68  
69      /**
70       * Creates a new instance of {@code AutoReleaseListener} and sets the
71       * {@code BuilderData} object to be released automatically. This object must
72       * be fully initialized.
73       *
74       * @param data the {@code BuilderData} object (must not be <b>null</b>)
75       * @throws IllegalArgumentException if the data object is <b>null</b> or
76       *         properties required are not set
77       */
78      public AutoReleaseListener(BuilderData data)
79      {
80          if (data == null)
81          {
82              throw new IllegalArgumentException("BuilderData must not be null!");
83          }
84          builderData = data;
85          builder = builderData.getBuilder();
86          if (builder == null)
87          {
88              throw new IllegalArgumentException("Builder must not be null!");
89          }
90      }
91  
92      /**
93       * Window activated event. This is just a dummy implementation.
94       *
95       * @param event the event
96       */
97      public void windowActivated(WindowEvent event)
98      {
99      }
100 
101     /**
102      * The window was closed. This will cause the {@code BuilderData} object to
103      * be released.
104      *
105      * @param event the event
106      */
107     public void windowClosed(WindowEvent event)
108     {
109         builder.release(builderData);
110     }
111 
112     /**
113      * Window deactivated event. This is just a dummy implementation.
114      *
115      * @param event the event
116      */
117     public void windowDeactivated(WindowEvent event)
118     {
119     }
120 
121     /**
122      * Window deiconified event. This is just a dummy implementation.
123      *
124      * @param event the event
125      */
126     public void windowDeiconified(WindowEvent event)
127     {
128     }
129 
130     /**
131      * Window iconified event. This is just a dummy implementation.
132      *
133      * @param event the event
134      */
135     public void windowIconified(WindowEvent event)
136     {
137     }
138 
139     /**
140      * Window opened event. This is just a dummy implementation.
141      *
142      * @param event the event
143      */
144     public void windowOpened(WindowEvent event)
145     {
146     }
147 
148     /**
149      * Window closing event. This is just a dummy implementation.
150      *
151      * @param event the event
152      */
153     public void windowClosing(WindowEvent event)
154     {
155     }
