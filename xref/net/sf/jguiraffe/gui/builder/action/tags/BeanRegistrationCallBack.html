16  package net.sf.jguiraffe.gui.builder.action.tags;
17  
18  import java.util.EventListener;
19  
20  import net.sf.jguiraffe.gui.builder.components.ComponentBuilderData;
21  
22  /**
23   * <p>
24   * A specialized registration callback dealing with beans from the current
25   * {@code BeanContext}.
26   * </p>
27   * <p>
28   * This implementation treats the target name as the name of a bean. In its
29   * {@link #register(ComponentBuilderData, Object)} method it obtains this bean
30   * from the current {@code BeanContext}. Then it asks the event manager to
31   * register the listener at this bean. It is also possible to pass a parameter
32   * object when this callback is registered at the {@link ComponentBuilderData}
33   * object. In this case, this parameter object becomes the target of the
34   * registration.
35   * </p>
36   *
37   * @author Oliver Heger
38   * @version $Id: BeanRegistrationCallBack.java 205 2012-01-29 18:29:57Z oheger $
39   */
40  class BeanRegistrationCallBack extends RegistrationCallBackBase
41  {
42      /**
43       * Creates a new instance of {@code BeanRegistrationCallBack} and
44       * initializes it.
45       *
46       * @param beanName the name of the target bean from the {@code BeanContext}
47       * @param type the listener type
48       * @param l the listener
49       * @param ignore the ignore failure flag
50       */
51      public BeanRegistrationCallBack(String beanName, String type,
52              EventListener l, boolean ignore)
53      {
54          super(beanName, type, l, ignore);
55      }
56  
57      /**
58       * Performs the registration. This implementation obtains the bean from the
59       * {@code BeanContext}. Then it delegates to the {@code FormEventManager} to
60       * add the event listener. If a parameter object was passed, the listener is
61       * registered at this object.
62       *
63       * @param builderData the {@code ComponentBuilderData}
64       * @param params additional parameters passed to the callback
65       * @return a flag whether the registration was successful
66       * @throws Exception in case of an error
67       */
68      @Override
69      protected boolean register(ComponentBuilderData builderData, Object params)
70              throws Exception
71      {
72          Object bean = (params == null) ? builderData.getBeanContext().getBean(
73                  getTargetName()) : params;
74          return builderData.getEventManager().addEventListenerToObject(bean,
75                  getListenerType(), getListener());
76      }
77  }
