16  package net.sf.jguiraffe.gui.builder.action.tags;
17  
18  import net.sf.jguiraffe.gui.builder.components.tags.UseBeanBaseTag;
19  import net.sf.jguiraffe.gui.builder.event.filter.EventFilter;
20  
21  import org.apache.commons.jelly.JellyTagException;
22  
23  /**
24   * <p>
25   * A specialized tag handler class for creating event filters.
26   * </p>
27   * <p>
28   * With this tag handler class simple (i.e. non chained) event filters can be
29   * constructed. By specifying the <code>class</code> attribute the concrete
30   * class of the filter to be created is defined. Then all the properties
31   * supported by this specific filter class can be set in further attributes.
32   * </p>
33   * <p>
34   * The tag also checks if it is nested inside a {@link ChainedEventFilterTag}
35   * tag. If this is the case, the current filter is added as a child filter to
36   * this chained filter tag. Otherwise the variable <code>CURRENT_FILTER</code>
37   * is set to the active event filter, which can be queried by other tags.
38   * </p>
39   * <p>
40   * By inheriting from <code>UseBeanBaseTag</code> all of the features provided
41   * by this base class are available for this tag handler class, too.
42   * </p>
43   *
44   * @author Oliver Heger
45   * @version $Id: EventFilterTag.java 205 2012-01-29 18:29:57Z oheger $
46   */
47  public class EventFilterTag extends UseBeanBaseTag
48  {
49      /**
50       * Constant for the current filter variable. This variable will be set to
51       * the currently created event filter and can be evaluated by other tags.
52       */
53      public static final String CURRENT_FILTER = "CURRENT_FILTER";
54  
55      /**
56       * Creates a new instance of <code>EventFilterTag</code>.
57       */
58      public EventFilterTag()
59      {
60          super();
61          setBaseClass(EventFilter.class);
62      }
63  
64      /**
65       * Creates a new instance of <code>EventFilterTag</code> and sets the
66       * default class and the base class.
67       *
68       * @param defaultClass the default class
69       * @param baseClass the base class
70       */
71      public EventFilterTag(Class<?> defaultClass, Class<?> baseClass)
72      {
73          super(defaultClass, baseClass);
74      }
75  
76      /**
77       * Performs post processing on the bean. This implementation checks if this
78       * tag is nested inside a <code>ChainedEventFilter</code> tag. If this is
79       * the case, the filter is added as a child filter. Otherwise the filter is
80       * set in the Jelly context as the current filter.
81       *
82       * @param bean the new bean
83       * @return a flag whether the bean could be passed to a target
84       * @throws JellyTagException if an error occurs
85       */
86      @Override
87      protected boolean passResults(Object bean) throws JellyTagException
88      {
89          ChainedEventFilterTag parent =
90                  (ChainedEventFilterTag) findAncestorWithClass(ChainedEventFilterTag.class);
91          if (parent != null)
92          {
93              parent.addChildFilter((EventFilter) bean);
94          }
95          else
96          {
97              getContext().setVariable(CURRENT_FILTER, bean);
98          }
99  
100         return true;
101     }
102 }
