16  package net.sf.jguiraffe.gui.builder.action.tags;

18  import java.util.EventListener;
19  
20  import net.sf.jguiraffe.gui.builder.components.ComponentBuilderCallBack;
21  import net.sf.jguiraffe.gui.builder.components.tags.FormBaseTag;
22  import net.sf.jguiraffe.gui.builder.components.tags.InputComponentTag;
23  import net.sf.jguiraffe.gui.builder.components.tags.UseBeanBaseTag;
24  
25  import org.apache.commons.jelly.JellyTagException;
26  import org.apache.commons.jelly.MissingAttributeException;
27  
28  /**
29   * <p>
30   * A tag handler class for registering an event listener at a single or multiple
31   * components.
32   * </p>
33   * <p>
34   * This tag handler class has a similar purpose as {@link EventListenerTag}.
35   * However, it uses a different approach for connecting an event listener with a
36   * component. While {@link EventListenerTag} and its sub classes allow
37   * connecting actions to standard component events (e.g. action or change
38   * events), this class operates on plain event listener implementations. It can
39   * also be used for working with non-standard event listeners, i.e. event
40   * listener interfaces defined in the {@code
41   * net.sf.jguiraffe.gui.builder.components.model} package that do not extend the
42   * {@code FormEventListener} interface or even 3rd party or custom event
43   * listener interfaces.
44   * </p>
45   * <p>
46   * This tag handler class extends {@code UseBeanBaseTag} and thus allows the
47   * definition of a bean class in multiple ways. The bean specified using the
48   * tag's attributes must implement the appropriate event listener interface.
49   * With the mandatory {@code type} attribute the event listener type must be
50   * specified. This can be either one of the standard types ({@code ACTION},
51   * {@code CHANGE}, or {@code FOCUS}) or a non-standard event listener type. In
52   * the latter case the type name is derived from the name of the method for
53   * adding event listeners. As an example take the
54   * {@link net.sf.jguiraffe.gui.builder.components.model.TreeHandler} interface
55   * that defines the {@code addExpansionListener()} method for registering an
56   * event listener to be notified when a node of the tree is expanded or
57   * collapsed. The corresponding event type is <em>Expansion</em>, i.e. the
58   * string between <em>add</em> and <em>Listener</em>.
59   * </p>
60   * <p>
61   * The component(s) the event listener should be registered at can be specified
62   * in multiple ways:
63   * <ul>
64   * <li>This tag can be placed in the body of a tag defining a component. Then it
65   * registers the listener at the component handler of this component.</li>
66   * <li>The name of a single component can be specified in the {@code component}
67   * attribute. The listener will be registered at this component's {@code
68   * ComponentHandler}.</li>
69   * <li>Event handlers may also be registered at arbitrary beans accessible
70   * through the current {@code BeanContext}. To do this the {@code targetBean}
71   * attribute must be specified. The tag obtains this bean from the context and
72   * calls the appropriate {@code add()} method for adding the handler. Note that
73   * if the tag is used in this way really all beans of the current bean context
74   * can be accessed, e.g. the platform-specific components or the current window.
75   * </li>
76   * <li>It is also possible to register a listener at all components that support
77   * the specified event type. In this case the {@code multiple} attribute must be
78   * set to <b>true</b>, and no component name or bean name must be specified.
79   * (The additional {@code multiple} attribute was introduced to provide a better
80   * error diagnosis: If the {@code component} attribute is omitted accidently, an
81   * exception is thrown rather than silently registering the listener at all
82   * components.)</li>
83   * </ul>
84   * </p>
85   * <p>
86   * The following table lists the attributes supported by this tag handler class.
87   * Of course, all attributes defined by the {@link UseBeanBaseTag} class for
88   * specifying the bean are also supported.
89   * </p>
90   * <table border="1">
91   * <tr>
92   * <th>Attribute</th>
93   * <th>Meaning</th>
94   * <th>Optional</th>
95   * </tr>
96   * <tr>
97   * <td valign="top">component</td>
98   * <td>The name of the component the event listener is to be registered at.</td>
99   * <td valign="top">Yes</td>
100  * </tr>
101  * <tr>
102  * <td valign="top">targetBean</td>
103  * <td>With this attribute an arbitrary object available in the current {@code
104  * BeanContext} can be selected to which the event listener is added. This is a
105  * more generic form than using the {@code component} attribute. The bean
106  * specified here does not need to implement the {@code ComponentHandler}
107  * interface; as the event listener is added through reflection, it can be an
108  * arbitrary object defining an appropriate {@code addXXXListener()} method.
109  * <strong>Note:</strong> This attribute should not be used for registering
110  * standard event handlers (<em>ACTION</em>, <em>FOCUS</em>, and <em>CHANGE</em>
111  * ).</td>
112  * <td valign="top">Yes</td>
113  * </tr>
114  * <tr>
115  * <td valign="top">multiple</td>
116  * <td>If this attribute is specified (with an arbitrary value), the event
117  * listener will be registered at all compatible components. In this case the
118  * {@code component} attribute must be empty.</td>
119  * <td valign="top">Yes</td>
120  * </tr>
121  * <tr>
122  * <td valign="top">eventType</td>
123  * <td>Determines the type of the event listener. The bean specified by this tag
124  * must implement a compatible event listener interface.</td>
125  * <td valign="top">No</td>
126  * </tr>
127  * <tr>
128  * <td valign="top">ignoreFail</td>
129  * <td>With this attribute error handling can be controlled. Per default, the
130  * tag causes an error if no event listener can be registered. This can have
131  * various reasons, e.g. the name of the component is wrong, there is a typo in
132  * the event listener type, or there is no compatible component supporting the
133  * listener type. In most cases it is preferable to throw an exception in such a
134  * situation instead of ignoring the problem silently. There may be use cases
135  * though, where it is considered normal behavior that no fitting component for
136  * registering an event listener can be found. In this cases the {@code
137  * ignoreFail} attribute should be set - it can have an arbitrary value. If it
138  * is present, the tag will not throw an exception if the registration of the
139  * event listener is not possible.</td>
140  * <td valign="top">Yes</td>
141  * </tr>
142  * </table>
143  * </p>
144  * <p>
145  * The following fragment shows an example how this tag can be used for
146  * registering a listener for tree expansion events:
147  *
148  * <pre>
149  * &lt;f:tree name=&quot;myTree&quot; model=&quot;treeModel&quot;&gt;
150  * &lt;/f:tree&gt;
151  * &lt;!-- The event listener bean --&gt;
152  * &lt;di:bean name=&quot;treeListener&quot;
153  *   beanClass=&quot;com.mypackage.MyTreeListenerClass&quot;&gt;
154  * &lt;/di:bean&gt;
155  * &lt;!-- Register the event listener --&gt;
156  * &lt;a:eventListener component=&quot;myTree&quot; eventType=&quot;Expansion&quot;
157  *   beanName=&quot;treeListener&quot;/&gt;
158  * </pre>
159  *
160  * </p>
161  * <p>
162  * Note that it does not matter where in the builder script the event
163  * registration tags are placed as the actual registration is performed after
164  * all components have been created.
165  * </p>
166  *
167  * @author Oliver Heger
168  * @version $Id: EventRegistrationTag.java 205 2012-01-29 18:29:57Z oheger $
169  */
170 public class EventRegistrationTag extends UseBeanBaseTag
171 {
172     /** Constant for the component attribute. */
173     private static final String ATTR_COMPONENT = "component";
174 
175     /** Constant for the targetBean attribute. */
176     private static final String ATTR_TARGET_BEAN = "targetBean";
177 
178     /** Constant for the multiple attribute. */
179     private static final String ATTR_MULTIPLE = "multiple";
180 
181     /** Constant for the event type attribute. */
182     private static final String ATTR_EVENT_TYPE = "eventType";
183 
184     /** Constant for the ignore fail attribute. */
185     private static final String ATTR_IGNORE_FAIL = "ignoreFail";
186 
187     /**
188      * Creates a new instance of {@code EventRegistrationTag}.
189      */
190     public EventRegistrationTag()
191     {
192         super(null, EventListener.class);
193         addIgnoreProperties(ATTR_COMPONENT, ATTR_MULTIPLE, ATTR_EVENT_TYPE,
194                 ATTR_IGNORE_FAIL, ATTR_TARGET_BEAN);
195     }
196 
197     /**
198      * Performs the registration of the event listener. The actual registration
199      * is done by a callback object at the very end of the builder script. This
200      * ensures that all components have already been created.
201      *
202      * @param bean the bean object
203      * @return a flag whether the event listener could be registered
204      * @throws JellyTagException if an error occurs
205      */
206     @Override
207     protected boolean passResults(Object bean) throws JellyTagException
208     {
209         String eventType = getAttributeStr(ATTR_EVENT_TYPE);
210         if (eventType == null)
211         {
212             throw new MissingAttributeException(ATTR_EVENT_TYPE);
213         }
214 
215         String beanName = null;
216         String componentName = null;
217         if (getAttributes().get(ATTR_MULTIPLE) != null)
218         {
219             if (getAttributeStr(ATTR_COMPONENT) != null
220                     || getAttributeStr(ATTR_TARGET_BEAN) != null)
221             {
222                 throw new JellyTagException(
223                         "A component or bean name must not be specified "
224                                 + "if the multiple attribute is set!");
225             }
226         }
227 
228         else if (getAttributeStr(ATTR_TARGET_BEAN) != null)
229         {
230             if (getAttributeStr(ATTR_COMPONENT) != null)
231             {
232                 throw new JellyTagException(
233                         "Cannot specify both a component name and "
234                                 + "a bean name!");
235             }
236             beanName = getAttributeStr(ATTR_TARGET_BEAN);
237         }
238 
239         else
240         {
241             componentName = getAttributeStr(ATTR_COMPONENT);
242             if (componentName == null)
243             {
244                 if (getParent() instanceof InputComponentTag)
245                 {
246                     componentName = ((InputComponentTag) getParent()).getName();
247                 }
248             }
249             if (componentName == null)
250             {
251                 return false; // cannot pass listener to a component
252             }
253         }
254 
255         EventListener listener = (EventListener) bean;
256         boolean ignoreErr = getAttributes().get(ATTR_IGNORE_FAIL) != null;
257         ComponentBuilderCallBack cb = (beanName != null) ? new BeanRegistrationCallBack(
258                 beanName, eventType, listener, ignoreErr)
259                 : new ComponentRegistrationCallBack(componentName, eventType,
260                         listener, ignoreErr);
261         FormBaseTag.getBuilderData(getContext()).addCallBack(cb, null);
262         return true;
263     }
264 }
