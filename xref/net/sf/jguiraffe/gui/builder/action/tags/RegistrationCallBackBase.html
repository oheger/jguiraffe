16  package net.sf.jguiraffe.gui.builder.action.tags;
17  
18  import java.util.EventListener;
19  
20  import net.sf.jguiraffe.gui.builder.components.ComponentBuilderCallBack;
21  import net.sf.jguiraffe.gui.builder.components.ComponentBuilderData;
22  import net.sf.jguiraffe.gui.builder.components.FormBuilderException;
23  
24  import org.apache.commons.logging.Log;
25  import org.apache.commons.logging.LogFactory;
26  
27  /**
28   * <p>
29   * An abstract {@code ComponentBuilderCallBack} base implementation that
30   * registers an event listener at an object at the end of the processing of the
31   * current builder script.
32   * </p>
33   * <p>
34   * This class is used by tag handler classes for registering event listeners.
35   * There are concrete sub classes for registering listeners at graphical
36   * components and at beans.
37   * </p>
38   *
39   * @author Oliver Heger
40   * @version $Id: RegistrationCallBackBase.java 205 2012-01-29 18:29:57Z oheger $
41   */
42  abstract class RegistrationCallBackBase implements ComponentBuilderCallBack
43  {
44      /** The logger. */
45      protected final Log log = LogFactory.getLog(getClass());
46  
47      /** The name of the target of the registration. */
48      private final String targetName;
49  
50      /** The listener type. */
51      private final String listenerType;
52  
53      /** The event listener to register. */
54      private final EventListener listener;
55  
56      /** The ignore failure flag. */
57      private final boolean ignoreFailure;
58  
59      /**
60       * Creates a new instance of {@code RegistrationCallBackBase} and
61       * initializes it.
62       *
63       * @param name the name of the target of this registration
64       * @param type the listener type
65       * @param l the listener
66       * @param ignore the ignore failure flag
67       */
68      protected RegistrationCallBackBase(String name, String type,
69              EventListener l, boolean ignore)
70      {
71          targetName = name;
72          listenerType = type;
73          listener = l;
74          ignoreFailure = ignore;
75      }
76  
77      /**
78       * Executes this callback. Delegates to the
79       * {@link #register(ComponentBuilderData, Object)} method to actually
80       * register the event listener. If no registration can be performed and the
81       * ignore failure flag is not set, an exception is thrown.
82       *
83       * @param builderData the {@code ComponentBuilderData} object
84       * @param params additional parameters
85       * @throws FormBuilderException if registration is not possible
86       */
87      public void callBack(ComponentBuilderData builderData, Object params)
88              throws FormBuilderException
89      {
90          boolean success = false;
91  
92          try
93          {
94              if (register(builderData, params))
95              {
96                  success = true;
97              }
98          }
99          catch (Exception ex)
100         {
101             log.error("Error when registering event listener at target "
102                     + getTargetName(), ex);
103         }
104 
105         if (!success && !isIgnoreFailure())
106         {
107             throw new FormBuilderException(
108                     "Could not register listener of type " + getListenerType()
109                             + " at target " + getTargetName());
110         }
111     }
112 
113     /**
114      * Returns the target name of the object to which the register is to be
115      * added.
116      *
117      * @return the name of the target object
118      */
119     protected String getTargetName()
120     {
121         return targetName;
122     }
123 
124     /**
125      * Returns the event listener type.
126      *
127      * @return the name of the event listener type
128      */
129     protected String getListenerType()
130     {
131         return listenerType;
132     }
133 
134     /**
135      * Returns the event listener to be registered.
136      *
137      * @return the event listener
138      */
139     protected EventListener getListener()
140     {
141         return listener;
142     }
143 
144     /**
145      * Returns a flag whether errors are to be ignored when doing the
146      * registration.
147      *
148      * @return a flag whether errors are to be ignored
149      */
150     protected boolean isIgnoreFailure()
151     {
152         return ignoreFailure;
153     }
154 
155     /**
156      * Performs the actual registration of the event listener. This method is
157      * called by {@code callBack()}. Failures (indicated by the return value)
158      * and exceptions are handled correspondingly.
159      *
160      * @param builderData the {@code ComponentBuilderData} object
161      * @param params additional parameters that have been passed to the {@code
162      *        ComponentBuilderData} object when registering the callback
163      * @return a flag whether registration was successful
164      * @throws Exception if an error occurs
165      */
166     protected abstract boolean register(ComponentBuilderData builderData,
167             Object params) throws Exception;
168 }
