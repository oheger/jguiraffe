25  /**
26   * <p>
27   * A specific input component tag that creates a combo box component.
28   * </p>
29   * <p>
30   * A combo box displays a list of options from which the user can select one.
31   * The options to be displayed are obtained from a
32   * {@link net.sf.jguiraffe.gui.builder.components.model.ListModel ListModel}
33   * object, which is set either by a tag in this tag's body or by specifying the
34   * model's name in the <code>modelRef</code> attribute. A component handler that
35   * wraps a combo box must maintain a single data object of the same type as the
36   * list model (i.e. the return value of the model's <code>getType()</code>
37   * method).
38   * </p>
39   * <p>
40   * The following table lists all attributes that are supported by this tag (in
41   * addition to the default attributes that are allowed for all input component
42   * tags):
43   * </p>
44   * <p>
45   * <table border="1">
46   * <tr>
47   * <th>Attribute</th>
48   * <th>Description</th>
49   * <th>Optional</th>
50   * </tr>
51   * <tr>
52   * <td valign="top">modelRef</td>
53   * <td>With this attribute the name of the list model can be supplied. It is
54   * then fetched from the current {@code BeanContext}. So a list model with this
55   * name must have been created before, either directly by the calling client or
56   * by another tag that has already been executed. If this attribute is not
57   * defined, a list model must be set by a nested tag.</td>
58   * <td valign="top">yes</td>
59   * </tr>
60   * <tr>
61   * <td valign="top">editable</td>
62   * <td>This boolean attribute determines whether the user can enter arbitrary
63   * text in the combo box's text field. If set to <b>false </b>, the user can
64   * only select one of the displayed options. For combo boxes that are editable
65   * the data value is not obtained from the list model, but will be the data
66   * entered by the user.</td>
67   * <td valign="top">yes</td>
68   * </tr>
69   * </table>
70   * </p>
71   *
72   * @author Oliver Heger
73   * @version $Id: ComboBoxTag.java 205 2012-01-29 18:29:57Z oheger $
74   */
75  public class ComboBoxTag extends InputComponentTag implements ListModelSupport
76  {
77      /** Stores a reference to the list model. */
78      private ListModel listModel;
79  
80      /** Stores the value of the modelRef attribute. */
81      private String modelRef;
82  
83      /** Stores the editable flag. */
84      private boolean editable;
85  
86      /**
87       * Returns a flag whether this combo box is editable.
88       *
89       * @return the editable flag
90       */
91      public boolean isEditable()
92      {
93          return editable;
94      }
95  
96      /**
97       * Setter method of the editable attribute.
98       *
99       * @param editable the attribute value
100      */
101     public void setEditable(boolean editable)
102     {
103         this.editable = editable;
104     }
105 
106     /**
107      * Returns the name of the model to use.
108      *
109      * @return the list model's name in the jelly context
110      */
111     public String getModelRef()
112     {
113         return modelRef;
114     }
115 
116     /**
117      * Setter method of the modelRef attribute.
118      *
119      * @param modelRef the attribute value
120      */
121     public void setModelRef(String modelRef)
122     {
123         this.modelRef = modelRef;
124     }
125 
126     /**
127      * Returns the list model to be used by this combo box.
128      *
129      * @return the list model
130      */
131     public ListModel getListModel()
132     {
133         return listModel;
134     }
135 
136     /**
137      * Sets the list model for this combo box. This method is intended to be
138      * called by nested tags.
139      *
140      * @param model the list model
141      */
142     public void setListModel(ListModel model)
143     {
144         listModel = model;
145     }
146 
147     /**
148      * Creates the combo box component.
149      *
150      * @param manager the component manager
151      * @param create the create flag
152      * @return the handler for the newly created component
153      * @throws FormBuilderException if an error occurs
154      * @throws JellyTagException if the tag is incorrectly used
155      */
156     @Override
157     protected ComponentHandler<Object> createComponentHandler(
158             ComponentManager manager, boolean create)
159             throws FormBuilderException, JellyTagException
160     {
161         if (!create)
162         {
163             fetchListModel();
164         }
165         return manager.createComboBox(this, create);
166     }
167 
168     /**
169      * Helper method for obtaining the list model. If the model has not yet been
170      * set by a nested tag, it is fetched from the context.
171      *
172      * @throws FormBuilderException if the model cannot be fetched
173      * @throws JellyTagException if no model has been provided
174      */
175     private void fetchListModel() throws FormBuilderException,
176             JellyTagException
177     {
178         ListModelUtils.initializeListModel(this);
179     }
180 }
