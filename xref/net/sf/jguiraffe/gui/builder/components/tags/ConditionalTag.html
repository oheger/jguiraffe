16  package net.sf.jguiraffe.gui.builder.components.tags;
17  
18  import org.apache.commons.jelly.Tag;
19  
20  /**
21   * <p>
22   * Definition of an interface for tags that are only executed for certain
23   * builders.
24   * </p>
25   * <p>
26   * In theory the XML GUI definition should be portable, i.e. compatible with
27   * different builders. In practice there may be cases where this is not
28   * possible. Then it comes in handy if there is an easy possibility to specify
29   * that certain tags are only executed under certain conditions. This is exactly
30   * the purpose of this interface. Note that with standard Jelly means it is
31   * quite easy to achieve such a conditional execution of tags; the form builder
32   * library however tries to provide an even more convenient mechanism by simply
33   * specifying an attribute with the name of a builder. The tag is then executed
34   * if the current builder either matches or does not match this name.
35   * </p>
36   * <p>
37   * So the major part of tags in this Jelly tag library will support the
38   * following attributes for conditional execution:
39   * <ul>
40   * <li><em>ifName</em>: This attribute can be assigned a comma separated
41   * list of builder names. The tag gets executed only if the name of the current
42   * builder matches one of the specified names.</li>
43   * <li><em>unlessName</em>: This attribute is very similar to the one
44   * explained before. The difference is that the tag gets executed only if the
45   * current builder's name does <strong>not </strong> match a name specified in
46   * the attribute.</li>
47   * </ul>
48   * </p>
49   *
50   * @author Oliver Heger
51   * @version $Id: ConditionalTag.java 205 2012-01-29 18:29:57Z oheger $
52   */
53  public interface ConditionalTag extends Tag
54  {
55      /**
56       * Returns the value of the <code>ifName</code> attribute. Here a list of
57       * builder names can be returned, for which the tag should be executed.
58       *
59       * @return a list of supported builders
60       */
61      String getIfName();
62  
63      /**
64       * Returns the value of the <code>unlessName</code> attribute. Here a list
65       * of builder names can be returned, for which the tag should bot be
66       * executed.
67       *
68       * @return a list of not supported builders
69       */
70      String getUnlessName();
71  }
