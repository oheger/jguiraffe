16  package net.sf.jguiraffe.gui.builder.components.tags;
17  
18  import net.sf.jguiraffe.gui.builder.components.ComponentManager;
19  import net.sf.jguiraffe.gui.builder.components.FormBuilderException;
20  import net.sf.jguiraffe.gui.forms.ComponentHandler;
21  
22  /**
23   * <p>
24   * A specific input component tag that allows to make use of custom input
25   * components.
26   * </p>
27   * <p>
28   * The idea behind this tag is to support a custom implementation of the
29   * {@link net.sf.jguiraffe.gui.forms.ComponentHandler ComponentHandler}
30   * interface that can be defined by a tag in the body of this tag. This
31   * component handler will then be added to the
32   * {@link net.sf.jguiraffe.gui.forms.Form Form} object constructed during the
33   * builder process, and - if it contains a component - to the generated GUI.
34   * </p>
35   * <p>
36   * This mechanism can be used to add components to the generated GUI that are
37   * not supported per default by the form builder library. Therefore it is only
38   * necessary to provide a specific <code>ComponentHandler</code> implementation
39   * that makes the new component accessible for the <code>Form</code> object.
40   * </p>
41   * <p>
42   * Another use case is to provide a different data model for some components
43   * generated by other tags. The default example for this is a group of radio
44   * buttons: Typically one would like to avoid storing the state of each button
45   * in the group; it is preferable to make a conversion, e.g. storing the index
46   * of the selected button as an int. This use case can be addressed by a
47   * concrete
48   * {@link net.sf.jguiraffe.gui.builder.components.CompositeComponentHandler
49   * CompositeComponentHandler} implementation, to which the associated component
50   * handlers can be added. The composite component handler can then be
51   * implemented in a way that it somehow transforms the data of the contained
52   * components and passes this transformed value to the owning form object.
53   * </p>
54   * <p>
55   * This tag handler class inherits from <code>InputComponentTag</code> and thus
56   * supports all attributes defined by its parent class. Especially the
57   * definition of transformers and validators is possible, which can play
58   * together with the component handler set for this tag.
59   * </p>
60   *
61   * @see ComponentHandlerTag
62   * @author Oliver Heger
63   * @version $Id: FieldTag.java 205 2012-01-29 18:29:57Z oheger $
64   */
65  public class FieldTag extends InputComponentTag
66  {
67      /** Stores the explicitly set component handler. */
68      private ComponentHandler<?> bodyComponentHandler;
69  
70      /**
71       * Sets the component handler to be used by this tag. This method is
72       * intended to be called by tags in this tag's body. The
73       * <code>createComponentHandler()</code> method will then return the
74       * component handler passed to this method.
75       *
76       * @param componentHandler the component handler to use
77       */
78      public void setComponentHandler(ComponentHandler<?> componentHandler)
79      {
80          bodyComponentHandler = componentHandler;
81      }
82  
83      /**
84       * Returns the component handler to be used by the component that is
85       * represented by this tag. This implementation just returns the handler
86       * that was set using the
87       * {@link #setComponentHandler(ComponentHandler)} method. So
88       * in this tag's body there must have been a component that has already set
89       * this data.
90       *
91       * @param manager the component manager
92       * @param create the create flag
93       * @return the new component handler
94       * @throws FormBuilderException if no component handler has been set
95       */
96      @Override
97      protected ComponentHandler<?> createComponentHandler(ComponentManager manager,
98              boolean create) throws FormBuilderException
99      {
100         if (create)
101         {
102             return null;
103         }
104         else
105         {
106             if (bodyComponentHandler == null)
107             {
108                 throw new FormBuilderException(
109                         "No component handler has been set!");
110             }
111             return bodyComponentHandler;
112         }
113     }
114 }
