27  /**
28   * <p>
29   * A tag for adding an icon to a component.
30   * </p>
31   * <p>
32   * This tag supports multiple ways for defining an icon:
33   * <ul>
34   * <li>By specifying an URL using the {@code url} attribute.</li>
35   * <li>By specifying a resource name using the {@code resource} attribute. In
36   * this case the icon is searched on the class path. Per default, the current
37   * default class loader is used for resolving the resource name. If a different
38   * class loader is desired, its name (as registered at the current
39   * {@code ClassLoaderProvider}) can be specified using the
40   * {@code resourceLoader} attribute.</li>
41   * <li>By specifying an arbitrary {@link Locator} pointing to the icon data
42   * using the {@code locator} attribute. This is actually the generic form of the
43   * ways described above. The locator is looked up by its name in the current
44   * bean context.</li>
45   * <li>With the {@code ref} attribute an icon can be referenced that has already
46   * been created and stored in the Jelly context (e.g by another {@code icon} tag
47   * that had a {@code var} attribute (see below).</li>
48   * </ul>
49   * Exactly one of these attributes must be present.
50   * </p>
51   * <p>
52   * The icon is created using the component manager. Then the tag searches for an
53   * enclosing tag that implements the {@link IconSupport} interface. If one is
54   * found, this tag is passed the icon. With the {@code var} attribute the name
55   * of a variable can be specified, which is assigned the icon. So it can be
56   * reused. At least one of the ways for storing an icon (the {@code var}
57   * attribute or an {@code IconSupport} tag) must be defined, otherwise an
58   * exception is thrown.
59   * </p>
60   *
61   * @author Oliver Heger
62   * @version $Id: IconTag.java 211 2012-07-10 19:49:13Z oheger $
63   */

219     /**
220      * Executes this tag. Tries to create the icon and set it on an enclosing
221      * {@link IconSupport} tag.
222      *
223      * @throws JellyTagException if an error occurs
224      * @throws FormBuilderException if an error occurs
225      */
226     @Override
227     protected void process() throws JellyTagException, FormBuilderException
228     {
229         int defCount = countIconDefinitions();
230         if (defCount < 1)
231         {
232             throw new MissingAttributeException(
233                     "Either url or resource or ref attribute must be set!");
234         }
235         if (defCount > 1)
236         {
237             throw new FormBuilderException(
238                     "Exactly one of these attributes must be used: "
239                             + "url, resource, ref");
240         }
241 
242         IconSupport parent = (IconSupport) findAncestorWithClass(IconSupport.class);
243         if (parent == null && StringUtils.isEmpty(getVar()))
244         {
245             throw new JellyTagException(
246                     "Icon tag must be nested inside an IconSupport tag!");
247         }
248 
249         Object icon = createIcon();
250         if (parent != null)
251         {
252             parent.setIcon(icon);
253         }
254         if (getVar() != null)
255         {
256             getContext().setVariable(getVar(), icon);
257         }
258     }
259 
260     /**
261      * Creates the icon. This method is called after all syntactic checks were
262      * successful. Here the icon must be created, depending on the defined
263      * attributes.
264      *
265      * @return the icon
266      * @throws FormBuilderException if an error occurs
267      */
268     protected Object createIcon() throws FormBuilderException
269     {
270         if (getRef() != null)
271         {
272             Object icon = getContext().getVariable(getRef());
273             if (icon == null)
274             {
275                 throw new FormBuilderException(
276                         "Cannot find referenced variable: " + getRef());
277             }
278             return icon;
279         }
280         else
281         {
282             return getBuilderData().getComponentManager().createIcon(
283                     fetchLocator());
284         }
285     }
286 
287     /**
288      * Fetches the locator for the icon. This method is called by
289      * {@code createIcon()} when no reference attribute is specified, i.e. the
290      * icon has to be obtained from a locator. If the {@code locator} attribute
291      * is specified, the locator is looked up from the current bean context.
292      * Otherwise a locator is created from the other convenience attributes (
293      * {@code url} or {@code resource}).
294      *
295      * @return the {@code Locator} that defines the icon
296      */
297     protected Locator fetchLocator()
298     {
299         if (getLocator() != null)
300         {
301             return (Locator) getBuilderData().getBeanContext().getBean(
302                     getLocator());
303         }
304         else if (getResource() != null)
305         {
306             ClassLoader cl =
307                     getBuilderData().getBeanContext().getClassLoaderProvider()
308                             .getClassLoader(getResourceLoader());
309             return ClassPathLocator.getInstance(getResource(), cl);
310         }
311         else
312         {
313             assert getUrl() != null : "No locator defined!";
314             return URLLocator.getInstance(getUrl());
315         }
316     }

318     /**
319      * Counts in how many ways the icon is defined. There should be exactly one
320      * way.
321      *
322      * @return the number of attributes that define the icon
323      */
324     private int countIconDefinitions()
325     {
326         int count = 0;
327         if (!StringUtils.isEmpty(getUrl()))
328         {
329             count++;
330         }
331         if (!StringUtils.isEmpty(getResource()))
332         {
333             count++;
334         }
335         if (!StringUtils.isEmpty(getRef()))
336         {
337             count++;
338         }
339         if (!StringUtils.isEmpty(getLocator()))
340         {
341             count++;
342         }
343         return count;
344     }
