21  /**
22   * <p>
23   * A tag handler class that creates an item for a list model.
24   * </p>
25   * <p>
26   * Tags of this class can appear in the body of a
27   * <code>{@link TextListModelTag}</code> tag. Each tag defines one item of the
28   * model with a mandatory display text (which can be set either directly or
29   * through a resource definition) and an optional value. The following table
30   * lists all supported attributes:
31   * </p>
32   * <p>
33   * <table border="1">
34   * <tr>
35   * <th>Attribute</th>
36   * <th>Description</th>
37   * <th>Optional</th>
38   * </tr>
39   * <tr>
40   * <td valign="top">text</td>
41   * <td>Defines the display text of this model item directly.</td>
42   * <td rowspan="2">exactly one of these</td>
43   * </tr>
44   * <tr>
45   * <td valign="top">textres</td>
46   * <td>Allows to define the display text of this model item from a resource
47   * property. If no resource group is specified, the builder's default resource
48   * will be used.</td>
49   * </tr>
50   * <tr>
51   * <td valign="top">resgrp</td>
52   * <td>Sets the resource group to use to resolve the resource provided by the
53   * <code>textres</code> attribute.</td>
54   * <td valign="top">yes</td>
55   * </tr>
56   * <tr>
57   * <td valign="top">value</td>
58   * <td>The corresponding value of this list model item. This can be an arbitrary
59   * object.</td>
60   * <td valign="top">yes</td>
61   * </tr>
62   * <tr>
63   * <td valign="top">valueRef</td>
64   * <td>Using this attribute the value of the corresponding list item can be
65   * specified as a reference. It will be looked up in the current {@code
66   * BeanContext}. This way the full power of the dependency injection framework
67   * can be used for creating and initializing Java objects. If both the {@code
68   * value} and the {@code valueRef} attributes are specified, the {@code value}
69   * attribute takes precedence.</td>
70   * <td valign="top">yes</td>
71   * </tr>
72   * </table>
73   * </p>
74   *
75   * @author Oliver Heger
76   * @version $Id: ListModelItemTag.java 205 2012-01-29 18:29:57Z oheger $
77   */
78  public class ListModelItemTag extends FormBaseTag
79  {
80      /** The text data object for managing the display text. */
81      private TextData textData;
82  
83      /** Stores the value of the model item. */
84      private Object value;
85  
86      /** The reference to the value. */
87      private String valueRef;
88  
89      /**
90       * Creates a new instance of {@code ListModelItemTag}.
91       */
92      public ListModelItemTag()
93      {
94          textData = new TextData(this);
95      }
96  
97      /**
98       * Returns the text data object with the definition of the display text.
99       *
100      * @return the display text data
101      */
102     public TextData getTextData()
103     {
104         return textData;
105     }
106 
107     /**
108      * Setter method of the text attribute.
109      *
110      * @param s the attribute value
111      */
112     public void setText(String s)
113     {
114         getTextData().setText(s);
115     }
116 
117     /**
118      * Setter method of the textres attribute.
119      *
120      * @param s the attribute value
121      */
122     public void setTextres(String s)
123     {
124         getTextData().setTextres(s);
125     }
126 
127     /**
128      * Setter method of the resgrp attribute.
129      *
130      * @param s the attribute value
131      */
132     public void setResgrp(String s)
133     {
134         getTextData().setResgrp(s);
135     }
136 
137     /**
138      * Returns the value of this list item.
139      *
140      * @return the value (can be <b>null </b>)
141      */
142     public Object getValue()

144         return value;
145     }
146 
147     /**

152     public void setValue(Object v)

157     /**

162     public String getValueRef()

164         return valueRef;
165     }
166 
167     /**
168      * Set method of the {@code valueRef} attribute
169      *
170      * @param valueRef the attribute value
171      */
172     public void setValueRef(String valueRef)
173     {
174         this.valueRef = valueRef;
175     }
176 
177     /**
178      * Executes this tag.
179      *
180      * @throws JellyTagException if this tag is used incorrectly
181      */
182     @Override
183     protected void process() throws JellyTagException
184     {
185         if (!getTextData().isDefined())
186         {
187             throw new MissingAttributeException("text");
188         }
189         TextListModelTag parent = (TextListModelTag) findAncestorWithClass(TextListModelTag.class);
190         if (parent == null)
191         {
192             throw new JellyTagException(
193                     "This tag must be nested inside a TextListModelTag!");
194         }
195         parent.addItem(getTextData().getCaption(), fetchValue());
196     }
197 
198     /**
199      * Returns the value of this item. This method is called by {@code
200      * process()}. It inspects the attributes {@code value} and {@code valueRef}
201      * and processes them correspondingly.
202      *
203      * @return the value of this item (can be <b>null</b> if unspecified)
204      */
205     protected Object fetchValue()
206     {
207         Object value = getValue();
208 
209         if (value == null && getValueRef() != null)
210         {
211             value = getBuilderData().getBeanContext().getBean(getValueRef());
212         }
213 
214         return value;
215     }
