16  package net.sf.jguiraffe.gui.builder.components.tags;
17  
18  import net.sf.jguiraffe.gui.builder.components.ComponentManager;
19  import net.sf.jguiraffe.gui.builder.components.FormBuilderException;
20  
21  import org.apache.commons.jelly.JellyTagException;
22  
23  /**
24   * <p>
25   * An abstract base class for tags that create simple GUI components.
26   * </p>
27   * <p>
28   * This base class is intended to be used for components that are not inserted
29   * into a {@link net.sf.jguiraffe.gui.forms.Form Form} object. Derived tag
30   * classes create a component of a specific type. This class ensures that the
31   * new component is added to the next enclosing container element.
32   * </p>
33   * <p>
34   * Simple GUI component can be assigned a name using the <code>name</code>
35   * attribute. If this is done, they are stored in the component store of the
36   * current builder operation. This makes it possible to obtain a
37   * <code>WidgetHandler</code> and manipulate some of their properties later.
38   * </p>
39   *
40   * @author Oliver Heger
41   * @version $Id: SimpleComponentTag.java 205 2012-01-29 18:29:57Z oheger $
42   */
43  public abstract class SimpleComponentTag extends ComponentBaseTag
44  {
45      /** Stores a reference to the created component. */
46      private Object component;
47  
48      /**
49       * Returns the component that was created by this tag.
50       *
51       * @return the component created by this tag
52       */
53      @Override
54      public Object getComponent()
55      {
56          return component;
57      }
58  
59      /**
60       * Callback before processing of the tag's body. Calls
61       * {@link #createComponent(ComponentManager)} for the first time with the
62       * <code>create</code> parameter set to <b>true</b>.
63       *
64       * @throws JellyTagException if a script related error occurs
65       * @throws FormBuilderException if an error occurs
66       */
67      @Override
68      protected void processBeforeBody() throws JellyTagException,
69              FormBuilderException
70      {
71          component =
72                  createComponent(getBuilderData().getComponentManager(), true);
73      }
74  
75      /**
76       * Executes this tag. Calls
77       * <code>{@link #createComponent(ComponentManager)}</code> for the second
78       * time with the boolean parameter <code>create</code> set to <b>false
79       * </b>. The new component is then added to the nesting container element.
80       *
81       * @throws JellyTagException if a script related error occurs
82       * @throws FormBuilderException if an error occurs
83       */
84      @Override
85      protected void process() throws JellyTagException, FormBuilderException
86      {
87          super.process();
88          component = createComponent(getBuilderData().getComponentManager(),
89                  false);
90          insertComponent(getName(), getComponent());
91      }
92  
93      /**
94       * Creates a specific component. This method must be implemented in derived
95       * classes. An implementation may use the passed in component manager to
96       * create an instance of a concrete GUI widget class. Note that this method
97       * is called twice during execution of this tag: Once before the tag's body
98       * is processed and once after that. The boolean argument can be used to
99       * distinguish between the two phases. This allows concrete implementations
100      * of the <code>ComponentManager</code> interface to use various
101      * strategies for creating a component hierarchy.
102      *
103      * @param manager the component manager
104      * @param create a flag whether the component should be newly created
105      * @return the new GUI component
106      * @throws JellyTagException if the tag is incorrectly used
107      * @throws FormBuilderException if an error occurs creating the component
108      */
109     protected abstract Object createComponent(ComponentManager manager,
110             boolean create) throws JellyTagException,
111             FormBuilderException;
112 }
