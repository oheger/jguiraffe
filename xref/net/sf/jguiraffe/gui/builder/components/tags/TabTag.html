25  /**
26   * <p>
27   * A tag handler class for creating register tab pages of a tabbed pane.
28   * </p>
29   * <p>
30   * Instances of this class can be placed in the body of a
31   * <code>{@link TabbedPaneTag}</code>. Each tag defines exactly one tab page.
32   * For this purpose a bunch of attributes can be used:
33   * </p>
34   * <p>
35   * <table border="1">
36   * <tr>
37   * <th>Attribute</th>
38   * <th>Description</th>
39   * <th>Optional</th>
40   * </tr>
41   * <tr>
42   * <td valign="top">title</td>
43   * <td>Defines the title of this register as a string constant.</td>
44   * <td valign="top">Yes</td>
45   * </tr>
46   * <tr>
47   * <td valign="top">titleres</td>
48   * <td>Defines the title of this register as a resource ID.</td>
49   * <td valign="top">Yes</td>
50   * </tr>
51   * <tr>
52   * <td valign="top">resgrp</td>
53   * <td>Defines the resource group for resolving resource IDs. This attribute is
54   * only evaluated if one of the other text attributes is specified using a
55   * resource ID. In this case the corresponding resource group is obtained from
56   * this value. If no resource group is specified, the default resource group
57   * will be used.</td>
58   * <td valign="top">Yes</td>
59   * </tr>
60   * <tr>
61   * <td valign="top">tooltip</td>
62   * <td>Allows to specify a tool tip for this register.</td>
63   * <td valign="top">Yes</td>
64   * </tr>
65   * <tr>
66   * <td valign="top">tooltipres</td>
67   * <td>Allows to specify a tool tip for this register as a resource ID.</td>
68   * <td valign="top">Yes</td>
69   * </tr>
70   * <tr>
71   * <td valign="top">mnemonic</td>
72   * <td>With this attribute a mnemonic key can be defined for this register.</td>
73   * <td valign="top">Yes</td>
74   * </tr>
75   * <tr>
76   * <td valign="top">mnemonicres</td>
77   * <td>Allows to specifiy a mnemonic key that is obtained from a resource.</td>
78   * <td valign="top">Yes</td>
79   * </tr>
80   * </table>
81   * </p>
82   * <p>
83   * In addition to these attributes in the body of this tag an
84   * <code>{@link IconTag}</code> can be placed to define an icon for the
85   * register to create.
86   * </p>
87   * <p>
88   * The content of the register is defined by a component that must also be
89   * defined in the tag's body. This is done by placing one of the component tags
90   * in this tag's body. Note that only a single component is supported. If
91   * multiple components are necessary, they must be placed inside a container
92   * tag, e.g. a <code>{@link PanelTag}</code>.
93   * </p>
94   *
95   * @author Oliver Heger
96   * @version $Id: TabTag.java 205 2012-01-29 18:29:57Z oheger $
97   */

197     /**
198      * Creates the container object. This implementation is a bit different: It
199      * will create an instance of the <code>TabData</code> inner class of
200      * <code>TabbedPaneTag</code>. This <code>TabData</code> object will be
201      * initialized with the component object that was added to this container
202      * tag (only one or zero child elements are allowed). This causes the data
203      * object to be added to the enclosing <code>TabbedPaneTag</code> tag.
204      *
205      * @param manager the component manager
206      * @param create the create flag
207      * @param components a collection with the child elements of this container
208      * @return the container object
209      * @throws FormBuilderException if the tag is not correctly used
210      * @throws MissingAttributeException if a required attribute is missing
211      */
212     @Override
213     protected Object createContainer(ComponentManager manager, boolean create,
214             Collection<Object[]> components) throws FormBuilderException,
215             MissingAttributeException
216     {
217         if (create)
218         {
219             if (!(findContainer() instanceof TabbedPaneTag))
220             {
221                 throw new FormBuilderException(
222                         "Tab tag must be directly nested inside a tabbed pane tag!");
223             }
224             return null;
225         }
226         else
227         {
228             if (!tid.isDefined())
229             {
230                 throw new FormBuilderException(
231                         "A tab must at least have either a text or an icon!");
232             }
233             if (components.size() > 1)
234             {
235                 throw new FormBuilderException(
236                         "Not more than one component can be placed in a tab!");
237             }
238             TabbedPaneTag.TabData tabData = new TabbedPaneTag.TabData();
239             tabData.setTitle(tid.getCaption());
240             tabData.setIcon(tid.getIcon());
241             tabData.setMnemonic(tid.getMnemonic());
242             tabData.setToolTip(tdToolTip.getCaption());
243             if (components.size() > 0)
244             {
245                 Object[] comp = components.iterator().next();
246                 tabData.setComponent(comp[0]);
247             }
248             return tabData;
249         }
250     }
