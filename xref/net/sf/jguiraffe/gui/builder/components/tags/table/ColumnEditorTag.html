16  package net.sf.jguiraffe.gui.builder.components.tags.table;
17  
18  import java.util.Set;
19  
20  import net.sf.jguiraffe.gui.forms.Form;
21  
22  /**
23   * <p>
24   * A tag for defining the editor of a column within a table.
25   * </p>
26   * <p>
27   * In the body of this tag a single component (which can also be a container
28   * with an arbitrary number of children) can be specified that will be used for
29   * editing the column this tag belongs to. No other specific attributes are
30   * supported.
31   * </p>
32   * <p>
33   * A more detailed description can be found in the class comments of the super
34   * class and the <code>TableTag</code> and the <code>ColumnTag</code>.
35   * </p>
36   *
37   * @author Oliver Heger
38   * @version $Id: ColumnEditorTag.java 205 2012-01-29 18:29:57Z oheger $
39   * @see TableTag
40   * @see TableColumnTag
41   */
42  public class ColumnEditorTag extends ColumnComponentTag
43  {
44      /**
45       * Returns the form affected by this tag from the specified table tag. This
46       * implementation will return the edit form.
47       *
48       * @param tabTag the table tag
49       * @return the form used by this tag
50       */
51      @Override
52      protected Form getTableForm(TableTag tabTag)
53      {
54          return tabTag.getRowEditForm();
55      }
56  
57      /**
58       * Initializes the passed in column tag with the results produced by this
59       * tag. This implementation sets the editor component and the set with the
60       * names of the components used by the editor form for this column.
61       *
62       * @param colTag the parent column tag
63       * @param bodyComponent the component defined in the body of this tag
64       * @param componentNames a set with the names of the involved components
65       */
66      @Override
67      protected void initializeColumn(TableColumnTag colTag,
68              Object bodyComponent, Set<String> componentNames)
69      {
70          colTag.initEditFields(componentNames);
71          colTag.initEditorComponent(bodyComponent);
72      }
73  }
