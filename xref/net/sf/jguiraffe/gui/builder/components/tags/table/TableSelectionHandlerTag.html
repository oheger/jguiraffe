16  package net.sf.jguiraffe.gui.builder.components.tags.table;
17  
18  import org.apache.commons.jelly.JellyTagException;
19  
20  import net.sf.jguiraffe.gui.builder.components.tags.UseBeanBaseTag;
21  
22  /**
23   * <p>
24   * A tag handler class for defining selection handlers for a table component.
25   * </p>
26   * <p>
27   * With this tag a concrete implementation of the
28   * <code>{@link TableSelectionHandler}</code> interface can be specified that
29   * will be used for either renderer or editor components (depending on the value
30   * of the boolean <code>editor</code> attribute) of the associated table. The
31   * tag can appear in the body of a <code>{@link TableTag}</code> tag only.
32   * </p>
33   * <p>
34   * By extending <code>UseBeanBaseTag</code> the typical attributes for
35   * defining beans (<code>class</code> or <code>ref</code>) are supported.
36   * In addition the <code>editor</code> attribute defines, which
37   * <code>TableSelectionHandler</code> is to be set on the associated table:
38   * <ul>
39   * <li>a value of <b>false</b> (which is also the default value) indicates
40   * that the <code>TableSelectionHandler</code> for renderer components is to
41   * set.</li>
42   * <li> a value of <b>true</b> in contrast will set the
43   * <code>TableSelectionHandler</code> for editor components.</li>
44   * </ul>
45   * </p>
46   *
47   * @author Oliver Heger
48   * @version $Id: TableSelectionHandlerTag.java 205 2012-01-29 18:29:57Z oheger $
49   */
50  public class TableSelectionHandlerTag extends UseBeanBaseTag
51  {
52      /** Constant for the reserved editor attribute. */
53      private static final String ATTR_EDITOR = "editor";
54  
55      /**
56       * Creates a new instance of <code>TableSelectionHandlerTag</code>.
57       */
58      public TableSelectionHandlerTag()
59      {
60          super();
61          setBaseClass(TableSelectionHandler.class);
62          addIgnoreProperty(ATTR_EDITOR);
63      }
64  
65      /**
66       * Passes the <code>TableSelectionHandler</code> created by this tag to
67       * the enclosing table tag. This implementation looks for the
68       * <code>TableTag</code> this tag is nested into. It then sets the
69       * corresponding handler. If this tag is not nested inside a
70       * <code>TableTag</code> and no <code>var</code> attribute is set, an
71       * exception will be thrown.
72       *
73       * @param bean the resulting bean
74       * @return a flag whether the bean could be passed to a target
75       * @throws JellyTagException if the tag is incorrectly used
76       */
77      @Override
78      protected boolean passResults(Object bean) throws JellyTagException
79      {
80          TableTag tableTag = (TableTag) findAncestorWithClass(TableTag.class);
81          if (tableTag == null)
82          {
83              return false;
84          }
85  
86          if (getAttributes().containsKey(ATTR_EDITOR)
87                  && Boolean.valueOf(String.valueOf(getAttributes().get(
88                          ATTR_EDITOR))))
89          {
90              tableTag.setEditorSelectionHandler((TableSelectionHandler) bean);
91          }
92          else
93          {
94              tableTag.setRendererSelectionHandler((TableSelectionHandler) bean);
95          }
96          return true;
97      }
98  }
