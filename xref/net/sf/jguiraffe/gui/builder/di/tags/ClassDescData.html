16  package net.sf.jguiraffe.gui.builder.di.tags;

18  import net.sf.jguiraffe.di.ClassLoaderProvider;
19  import net.sf.jguiraffe.di.impl.ClassDescription;

23  /**
24   * <p>
25   * A simple data class for managing the components needed for a class
26   * description.
27   * </p>
28   * <p>
29   * A couple of tags of the <em>di</em> builder need to create
30   * {@link ClassDescription} objects. This class encapsulates the
31   * functionality required for this purpose. It defines properties for the
32   * supported components and provides validation methods.
33   * </p>
34   *
35   * @author Oliver Heger
36   * @version $Id: ClassDescData.java 205 2012-01-29 18:29:57Z oheger $
37   */
38  public class ClassDescData
39  {
40      /** Stores the target class of the class description. */
41      private Class<?> targetClass;
42  
43      /** Stores the name of the target class. */
44      private String targetClassName;
45  
46      /** Stores the name of the class loader to be used. */
47      private String classLoaderName;
48  
49      /**
50       * Returns the target class for the class description.
51       *
52       * @return the target class
53       */
54      public Class<?> getTargetClass()
55      {
56          return targetClass;
57      }
58  
59      /**
60       * Sets the target class of the class description.
61       *
62       * @param targetClass the target class
63       */
64      public void setTargetClass(Class<?> targetClass)
65      {
66          this.targetClass = targetClass;
67      }
68  
69      /**
70       * Returns the name of the target class for the class description.
71       *
72       * @return the name of the target class
73       */
74      public String getTargetClassName()
75      {
76          return targetClassName;
77      }
78  
79      /**
80       * Sets the name of the target class for the class description.
81       *
82       * @param targetClassName the name of the target class
83       */
84      public void setTargetClassName(String targetClassName)
85      {
86          this.targetClassName = targetClassName;
87      }
88  
89      /**
90       * Returns the name of the class loader to be used.
91       *
92       * @return the name of the class loader
93       */
94      public String getClassLoaderName()
95      {
96          return classLoaderName;
97      }
98  
99      /**
100      * Sets the name of the class loader to be used.
101      *
102      * @param classLoaderName the name of the class loader to be used
103      */
104     public void setClassLoaderName(String classLoaderName)
105     {
106         this.classLoaderName = classLoaderName;
107     }
108 
109     /**
110      * Checks whether all required data for constructing a class description is
111      * available.
112      *
113      * @return a flag whether enough information for constructing a class
114      * description is available
115      */
116     public boolean isDefined()
117     {
118         return getTargetClass() != null || getTargetClassName() != null;
119     }
120 
121     /**
122      * Checks whether the properties of this object are valid. This
123      * implementation checks whether either a target class or a class name is
124      * set. If both are set, they must specify the same class. This
125      * implementation expects that the properties are defined, i.e.
126      * <code>{@link #isDefined()}</code> has returned <b>true</b>.
127      *
128      * @return a flag whether the properties are valid
129      */
130     public boolean isValid()
131     {
132         if (getTargetClass() != null && getTargetClassName() != null)
133         {
134             return getTargetClassName().equals(getTargetClass().getName());
135         }
136         return true;
137     }
138 
139     /**
140      * Creates a <code>ClassDescription</code> object from the internal data.
141      * If not all required properties are set or if they contain invalid values,
142      * an exception will be thrown.
143      *
144      * @return a <code>ClassDescription</code> object corresponding to the
145      * values of the properties
146      * @throws JellyTagException if not all required properties are set or
147      * some properties are invalid
148      */
149     public ClassDescription createClassDescription() throws JellyTagException
150     {
151         if (!isDefined())
152         {
153             throw new JellyTagException("Required properties are missing!");
154         }
155         if (!isValid())
156         {
157             throw new JellyTagException(
158                     "Properties are not valid (ambigous class definition)!");
159         }
160 
161         return (getTargetClass() != null) ? ClassDescription
162                 .getInstance(getTargetClass()) : ClassDescription.getInstance(
163                 getTargetClassName(), getClassLoaderName());
164     }
165 
166     /**
167      * Returns a <code>ClassDescription</code> object from the internal data
168      * or <b>null</b> if no data is defined. This is a convenience method. If
169      * data of a class is defined, <code>isValid()</code> is called for
170      * checking the validity of the entered data. If this fails, an exception is
171      * thrown. If the class is undefined, simply <b>null</b> is returned. This
172      * functionality is frequently needed when a class definition is optional:
173      * if one is provided, it must be valid; otherwise <b>null</b> can be
174      * returned.
175      *
176      * @return a <code>ClassDescription</code> object corresponding to the
177      *         values of the properties or <b>null</b> if no class is defined
178      * @throws JellyTagException if the class definition is not valid
179      */
180     public ClassDescription getOptionalClassDescription()
181             throws JellyTagException
182     {
183         if (isDefined())
184         {
185             if (!isValid())
186             {
187                 throw new JellyTagException(
188                         "Properties are not valid (ambigous class definition)!");
189             }
190 
191             return createClassDescription();
192         }
193 
194         else
195         {
196             return null;
197         }
198     }
199 
200     /**
201      * A convenience method for resolving the class specified by this data
202      * object. This method creates the corresponding {@link ClassDescription}
203      * and obtains the target class for it. No caching is performed, so this
204      * method should only be used if resolving happens once.
205      *
206      * @param clProvider the {@code ClassLoaderProvider} to be used for class
207      *        loading
208      * @return the class specified by this description object
209      * @throws JellyTagException if an error occurs during resolving
210      * @see #createClassDescription()
211      * @see ClassDescription#getTargetClass(ClassLoaderProvider)
212      */
213     public Class<?> resolveClass(ClassLoaderProvider clProvider)
214             throws JellyTagException
215     {
216         return createClassDescription().getTargetClass(clProvider);
217     }
