16  package net.sf.jguiraffe.gui.builder.di.tags;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  import net.sf.jguiraffe.di.Dependency;
22  import net.sf.jguiraffe.di.impl.ClassDescription;
23  import net.sf.jguiraffe.di.impl.Invokable;
24  
25  import org.apache.commons.jelly.JellyTagException;
26  import org.apache.commons.jelly.TagSupport;
27  import org.apache.commons.jelly.XMLOutput;

29  /**
30   * <p>
31   * A base class for tag handler classes defining invocation objects.
32   * </p>
33   * <p>
34   * This abstract base class provides common functionality for the creation and
35   * initialization of invocation objects. It is especially capable of dealing
36   * with the parameters of the invocation. Concrete sub classes only have to take
37   * care that the correct invocation instance is created, e.g. a method or a
38   * constructor invocation.
39   * </p>
40   *
41   * @author Oliver Heger
42   * @version $Id: InvocationTag.java 205 2012-01-29 18:29:57Z oheger $
43   */
44  public abstract class InvocationTag extends TagSupport
45  {
46      /** A list with the parameter tags found in the body of this tag. */
47      private List<ParameterTag> parameterTags;
48  
49      /** An array with the data types of the parameters. */
50      private ClassDescription[] parameterTypes;
51  
52      /** An array with the dependencies of the parameters. */
53      private Dependency[] parameterDependencies;
54  
55      /** A class description for the target class of the invocation. */
56      private ClassDescription targetClass;
57  
58      /** A data object for defining the target class of this invocation. */
59      private ClassDescData targetClassData;
60  
61      /** Stores the name of the result variable. */
62      private String result;
63  
64      /**
65       * Creates a new instance of <code>InvocationTag</code>.
66       */
67      protected InvocationTag()
68      {
69          targetClassData = new ClassDescData();
70          parameterTags = new ArrayList<ParameterTag>();
71      }
72  
73      /**
74       * Returns the name of the result variable. This variable can be set if this
75       * invocation tag belongs to a <code>ChainedInvocation</code>.
76       *
77       * @return the name of the result variable
78       */
79      public String getResult()
80      {
81          return result;
82      }
83  
84      /**
85       * Set method of the result attribute.
86       *
87       * @param result the attribute's value
88       */
89      public void setResult(String result)
90      {
91          this.result = result;
92      }
93  
94      /**
95       * Returns the name of the source variable. This variable can be set if this
96       * invocation tag belongs to a <code>ChainedInvocation</code>. This base
97       * implementation always returns <b>null</b>. Derived classes can overwrite
98       * it if they support the <code>source</code> attribute.
99       *
100      * @return the name of the source variable
101      */
102     public String getSource()
103     {
104         return null;
105     }
106 
107     /**
108      * Returns the target class data object.
109      *
110      * @return the data object for defining the target class
111      */
112     public ClassDescData getTargetClassData()
113     {
114         return targetClassData;
115     }
116 
117     /**
118      * Returns the class description for the target class of this invocation.
119      * Depending on a concrete sub class this description may or may not be
120      * required.
121      *
122      * @return the class description for the target class
123      */
124     public ClassDescription getTargetClassDescription()
125     {
126         return targetClass;
127     }
128 
129     /**
130      * Set method of the targetClass attribute.
131      *
132      * @param c the attribute's value
133      */
134     public void setTargetClass(Class<?> c)
135     {
136         getTargetClassData().setTargetClass(c);
137     }
138 
139     /**
140      * Set method of the targetClassName attribute.
141      *
142      * @param s the attribute's value
143      */
144     public void setTargetClassName(String s)
145     {
146         getTargetClassData().setTargetClassName(s);
147     }
148 
149     /**
150      * Set method of the targetClassLoader attribute.
151      *
152      * @param s the attribute's value
153      */
154     public void setTargetClassLoader(String s)
155     {
156         getTargetClassData().setClassLoaderName(s);
157     }
158 
159     /**
160      * The main method of this tag. Processes the parameters in the body of this
161      * tag and then delegates to the <code>createInvocation()</code> method
162      * for actually creating the invocation object.
163      *
164      * @param out the output object
165      * @throws JellyTagException if the tag is incorrectly used or an error
166      * occurs
167      */
168     public void doTag(XMLOutput out) throws JellyTagException
169     {
170         invokeBody(out);
171         process();
172     }
173 
174     /**
175      * Adds a parameter to this invocation. This method is called by parameter
176      * tags found in the body of this tag.
177      *
178      * @param paramTag the parameter tag to be added
179      */
180     public void addParameter(ParameterTag paramTag)
181     {
182         parameterTags.add(paramTag);
183     }
184 
185     /**
186      * Executes this tag. This method is invoked by <code>doTag()</code> after
187      * the tag's body has been executed. It does the real work.
188      *
189      * @throws JellyTagException if the tag is incorrectly used or an error
190      * occurs
191      */
192     protected void process() throws JellyTagException
193     {
194         if (getTargetClassData().isDefined())
195         {
196             targetClass = getTargetClassData().createClassDescription();
197         }
198         parameterTypes = new ClassDescription[parameterTags.size()];
199         parameterDependencies = new Dependency[parameterTags.size()];
200         for (int i = 0; i < parameterTypes.length; i++)
201         {
202             parameterTypes[i] = parameterTags.get(i).getParameterClassDesc();
203             parameterDependencies[i] = parameterTags.get(i)
204                     .getParameterDependency();
205         }
206 
207         processInvokable(createInvocation());
208     }
209 
210     /**
211      * Processes the <code>Invokable</code> object after its creation. This
212      * method will add it to the current <code>{@link InvokableSupport}</code>
213      * object.
214      *
215      * @param inv the <code>Invokable</code> object
216      * @throws JellyTagException if an error occurs
217      */
218     protected void processInvokable(Invokable inv) throws JellyTagException
219     {
220         InvocationData.get(getContext()).addInvokable(inv, getResult(),
221                 getSource());
222     }
223 
224     /**
225      * Returns an array with the data types of the parameters of this
226      * invocation. This information is probably needed for creating the
227      * invocation instance.
228      *
229      * @return an array with the data types of the parameters (some elements may
230      * be <b>null</b> if they have not been defined)
231      */
232     protected ClassDescription[] getParameterTypes()
233     {
234         return parameterTypes;
235     }
236 
237     /**
238      * Returns an array with the dependencies of the this invocation's
239      * parameters. This information is probably needed for creating the
240      * invocation instance.
241      *
242      * @return an array with the dependencies of the parameters
243      */
244     protected Dependency[] getParameterDependencies()
245     {
246         return parameterDependencies;
247     }
248 
249     /**
250      * Creates the invocation object. Concrete sub classes must here return the
251      * fully initialized invocation object.
252      *
253      * @return the newly created invocation object
254      * @throws JellyTagException if an error occurs
255      */
256     protected abstract Invokable createInvocation() throws JellyTagException;
257 }
