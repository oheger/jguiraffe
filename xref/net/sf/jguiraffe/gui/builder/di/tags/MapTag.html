28  /**
29   * <p>
30   * A specialized <code>BeanTag</code> implementation for creating maps.
31   * </p>
32   * <p>
33   * This tag handler class can be placed either in the body of an arbitrary tag
34   * derived from {@link DependencyTag}. It will then create a map and pass it to
35   * the enclosing {@link DependencyTag}. Alternatively, the {@code name}
36   * attribute can be specified, then the map bean is directly stored in the
37   * corresponding {@code BeanStore}. The map can be populated by
38   * <code>&lt;entry&gt;</code> tags in the body of this tag. The type of the key
39   * and the value of the map entries can be specified as attributes. A typical
40   * example of using this tag could look as follows:
41   * </p>
42   * <p>
43   *
44   * <pre>
45   *   &lt;constructor&gt;
46   *     &lt;param&gt;
47   *       &lt;map keyClassName=&quot;java.lang.String&quot;
48   *         valueClassName=&quot;java.lang.Integer&quot;&gt;
49   *         &lt;entry key=&quot;k1&quot; value=&quot;1&quot;/&gt;
50   *         &lt;entry key=&quot;k2&quot; value=&quot;2&quot;/&gt;
51   *         &lt;entry key=&quot;k3&quot; value=&quot;3&quot;/&gt;
52   *       &lt;/map&gt;
53   *     &lt;/param&gt;
54   *   &lt;/constructor&gt;
55   * </pre>
56   *
57   * Here a map is created, populated with some key value pairs and passed as
58   * argument to a constructor.
59   * </p>
60   * <p>
61   * <code>MapTag</code> supports the following attributes:
62   * <table border="1">
63   * <tr>
64   * <th>Attribute</th>
65   * <th>Description</th>
66   * <th>Optional</th>
67   * </tr>
68   * <tr>
69   * <td valign="top">name</td>
70   * <td>Using this attribute a name for the map bean can be specified. The bean
71   * can then be queried from the current bean store by this name. If no name is
72   * provided, an anonymous bean is created; in this case the tag must be nested
73   * in the body of a {@link DependencyTag}.</td>
74   * <td valign="top">yes</td>
75   * </tr>
76   * <tr>
77   * <td valign="top">keyClass</td>
78   * <td>Here the class of the keys of the map entries can be specified. When
79   * elements are added, the tag will try to convert the key objects to this class
80   * if necessary. If an {@link EntryTag} in the body of this tag defines a key
81   * class, this class will be used thus overriding the key class specified at the
82   * map level. If neither the map tag nor the entry tag specifies a key class, no
83   * type conversion will be performed.</td>
84   * <td valign="top">yes</td>
85   * </tr>
86   * <tr>
87   * <td valign="top">keyClassName</td>
88   * <td>This attribute has the same effect as <code>keyClass</code>, but the name
89   * of the key class is specified rather than the class object itself.</td>
90   * <td valign="top">yes</td>
91   * </tr>
92   * <tr>
93   * <td valign="top">keyClassLoader</td>
94   * <td>If the name of the key class is specified, with this attribute a class
95   * loader for resolving the class can be defined. The name specified here will
96   * be passed to the current {@link net.sf.jguiraffe.di.ClassLoaderProvider
97   * ClassLoaderProvider} for obtaining the desired class loader.</td>
98   * <td valign="top">yes</td>
99   * </tr>
100  * <tr>
101  * <td valign="top">valueClass</td>
102  * <td>With this attribute the class of the map values can be specified. This is
103  * analogous to the <code>keyClass</code> attribute, but defines a default class
104  * for the values stored in the map.</td>
105  * <td valign="top">yes</td>
106  * </tr>
107  * <tr>
108  * <td valign="top">valueClassName</td>
109  * <td>Analogous to <code>valueClass</code>, but the name of the value class is
110  * specified.</td>
111  * <td valign="top">yes</td>
112  * </tr>
113  * <tr>
114  * <td valign="top">valueClassLoader</td>
115  * <td>Defines the class loader to be used for resolving the value class when
116  * only the name is specified. This attribute has the same meaning as the
117  * <code>keyClassLoader</code> attribute for the value class.</td>
118  * <td valign="top">yes</td>
119  * </tr>
120  * <tr>
121  * <td valign="top">ordered</td>
122  * <td>Per default this tag will create a <code>java.util.HashMap</code> object
123  * that does not maintain a specific order of its elements. If this attribute is
124  * set to <b>true</b>, an instance of <code>java.util.LinkedHashMap</code> will
125  * be created instead. Then the map will remember the order in which its
126  * elements have been added.</td>
127  * <td valign="top">yes</td>
128  * </tr>
129  * </table>
130  * </p>
131  *
132  * @author Oliver Heger
133  * @version $Id: MapTag.java 205 2012-01-29 18:29:57Z oheger $
134  */

311     /**
312      * Sets the value of the ordered attribute. This attribute determines the

316      *
317      * @param ordered the value of the ordered attribute
318      */
319     public void setOrdered(boolean ordered)
320     {
321         this.ordered = ordered;
322     }

324     /**
325      * Adds a new entry to this map. This method is intended to be called by
326      * tags in the body of this tag. Key and value are passed in as
327      * dependencies. Their actual values are resolved when the map bean is
328      * created.
329      *
330      * @param depKey the dependency for the key (must not be <b>null</b>)
331      * @param depValue the dependency for the value (must not be <b>null</b>)
332      * @throws IllegalArgumentException if the key or the value dependency is
333      *         <b>null</b>
334      */
335     public void addEntry(Dependency depKey, Dependency depValue)
336     {
337         if (depKey == null)
338         {
339             throw new IllegalArgumentException(
340                     "Key dependency must not be null!");
341         }
342         if (depValue == null)
343         {
344             throw new IllegalArgumentException(
345                     "Value dependency must not be null!");
346         }
347 
348         getKeyDependencies().add(depKey);
349         getValueDependencies().add(depValue);
350     }
351 
352     /**
353      * Returns the collection with the key dependencies.
354      *
355      * @return the key dependencies
356      */
357     protected Collection<Dependency> getKeyDependencies()
358     {
359         return keyDependencies;
360     }
361 
362     /**
363      * Returns the collection with the value dependencies.
364      *
365      * @return the value dependencies
366      */
367     protected Collection<Dependency> getValueDependencies()
368     {
369         return valueDependencies;
370     }
371 
372     /**
373      * Performs some processing before the body gets executed. This
374      * implementation will initialize the class descriptions for the key and
375      * value class.
376      *
377      * @throws JellyTagException if the tag is incorrectly used
378      */
379     @Override
380     protected void processBeforeBody() throws JellyTagException
381     {
382         super.processBeforeBody();
383         keyClassDesc = getKeyClassData().getOptionalClassDescription();
384         valueClassDesc = getValueClassData().getOptionalClassDescription();
385     }
386 
387     /**
388      * Creates the <code>BeanProvider</code> produced by this tag. This
389      * implementation creates a {@link MapBeanProvider} and initialize it with
390      * the dependencies for the keys and values that have been added so far.
391      *
392      * @return the <code>BeanProvider</code> produced by this tag
393      * @throws JellyTagException if an error occurs
394      */
395     @Override
396     protected BeanProvider createBeanProvider() throws JellyTagException
397     {
398         return new MapBeanProvider(getKeyDependencies(),
399                 getValueDependencies(), isOrdered());
400     }
