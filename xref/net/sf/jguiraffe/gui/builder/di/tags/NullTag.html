16  package net.sf.jguiraffe.gui.builder.di.tags;
17  
18  import org.apache.commons.jelly.JellyTagException;
19  import org.apache.commons.jelly.TagSupport;
20  import org.apache.commons.jelly.XMLOutput;
21  
22  /**
23   * <p>
24   * A tag handler class for specifying <b>null</b> values.
25   * </p>
26   * <p>
27   * This tag handler class can appear in the body of a tag implementing the
28   * {@link ValueSupport} interface. It sets the value of its parent tag to
29   * <b>null</b>.
30   * </p>
31   * <p>
32   * The main use case for this tag is to be placed in the body of a
33   * {@link ParameterTag} or {@link SetPropertyTag}. These tags require a value to
34   * set; otherwise they throw an exception. Using this tag means "there is a
35   * value, but it is <b>null</b>".
36   * </p>
37   *
38   * @author Oliver Heger
39   * @version $Id: NullTag.java 205 2012-01-29 18:29:57Z oheger $
40   */
41  public class NullTag extends TagSupport
42  {
43      /**
44       * Executes this tag. This implementation checks whether the parent tag
45       * implements the {@link ValueSupport} interface. If this is the case, its
46       * value is set to <b>null</b>. Otherwise an exception is thrown.
47       *
48       * @param out the output object
49       * @throws JellyTagException if the tag is used incorrectly
50       */
51      public void doTag(XMLOutput out) throws JellyTagException
52      {
53          if (!(getParent() instanceof ValueSupport))
54          {
55              throw new JellyTagException(
56                      "Null tag must be in the body of a ValueSupport tag!");
57          }
58  
59          ((ValueSupport) getParent()).setValue(null);
60      }
61  }
