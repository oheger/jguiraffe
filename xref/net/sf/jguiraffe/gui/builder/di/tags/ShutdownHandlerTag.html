16  package net.sf.jguiraffe.gui.builder.di.tags;
17  
18  import net.sf.jguiraffe.di.impl.ChainedInvocation;
19  
20  import org.apache.commons.jelly.JellyTagException;
21  import org.apache.commons.jelly.TagSupport;
22  import org.apache.commons.jelly.XMLOutput;
23  
24  /**
25   * <p>
26   * A tag handler class that defines a shutdown script for a {@code BeanProvider}
27   * .
28   * </p>
29   * <p>
30   * This tag can appear in the body of a {@link BeanTag}. In its body an
31   * arbitrary number of tags derived from {@link InvocationTag} can be placed
32   * defining a set of operations to be performed when shutting down the
33   * corresponding singleton bean. All theses operations are added to a
34   * {@link ChainedInvocation} object, which will then become the
35   * <em>shutdown handler</em> of the singleton {@code BeanProvider} created by
36   * the hosting bean tag.
37   * </p>
38   * <p>
39   * Using this tag is an alternative to the {@code shutdownMethod} attribute
40   * supported by {@link BeanTag}. With the attribute only a single,
41   * parameter-less method can be specified to be invoked on shutdown of the bean.
42   * This tag in contrast supports much more complex shutdown scripts: all tags
43   * producing {@code Invokable} objects can be used. However, care must be taken
44   * with the dependencies used by the shutdown script. Because the owning {@code
45   * BeanStore} may already been partly destroyed when the shutdown script is
46   * invoked, access to other beans living in the same bean store is not
47   * permitted. It is possible to invoke arbitrary methods on the bean managed by
48   * the {@code BeanProvider} (this bean is also the default target of the
49   * invocation, so all method invocations or property access operations per
50   * default have this object as target), set or query properties on it, or making
51   * use of script-local variables.
52   * </p>
53   * <p>
54   * This tag does not define any attributes. In its body the
55   * {@link InvocationTag} objects can be placed that make up the shutdown script.
56   * </p>
57   *
58   * @author Oliver Heger
59   * @version $Id: ShutdownHandlerTag.java 205 2012-01-29 18:29:57Z oheger $
60   */
61  public class ShutdownHandlerTag extends TagSupport
62  {
63      /** The chained invocation storing the shutdown script. */
64      private ChainedInvocation shutdownScript;
65  
66      /**
67       * Executes this tag. This implementation delegates to the
68       * {@link #processBeforeBody()} and {@link #process()} methods.
69       *
70       * @param output the output object
71       * @throws JellyTagException if the tag is used incorrectly
72       */
73      public void doTag(XMLOutput output) throws JellyTagException
74      {
75          processBeforeBody();
76          invokeBody(output);
77          process();
78      }
79  
80      /**
81       * Performs some processing of this tag before the body is evaluated.
82       *
83       * @throws JellyTagException if the tag is used incorrectly
84       */
85      protected void processBeforeBody() throws JellyTagException
86      {
87          if (!(getParent() instanceof BeanTag))
88          {
89              throw new JellyTagException(
90                      "This tag must be nested inside a bean tag!");
91          }
92  
93          shutdownScript = new ChainedInvocation();
94          InvocationData.get(getContext()).registerInvokableSupport(
95                  shutdownScript);
96      }
97  
98      /**
99       * Performs the actual processing of this tag. This method is called by
100      * {@code doTag()} after the body has been evaluated.
101      *
102      * @throws JellyTagException if an error occurs
103      */
104     protected void process() throws JellyTagException
105     {
106         ((BeanTag) getParent()).setShutdownHandler(shutdownScript);
107         InvocationData.get(getContext()).unregisterInvokableSupport();
108     }
109 }
