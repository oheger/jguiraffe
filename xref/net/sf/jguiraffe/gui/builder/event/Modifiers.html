16  package net.sf.jguiraffe.gui.builder.event;
17  
18  import java.util.Locale;
19  
20  /**
21   * <p>
22   * An enumeration class representing special modifier keys.
23   * </p>
24   * <p>
25   * A typical key board contains some special modifier keys that can be pressed
26   * together with other keys modifying their state and meaning, e.g. the
27   * <em>SHIFT</em> or the <em>ALT</em> key. This class defines constants for such
28   * keys. Most UI toolkits define similar constants. This enumeration class
29   * provides a portable way of specifying such modifiers.
30   * </p>
31   * <p>
32   * The constants defined by this class can be used by several other classes in
33   * the <em>JGUIraffe</em> library. For instance, accelerators for actions (i.e.
34   * key board short cuts) may be assigned some modifiers that need to be pressed
35   * for activating the accelerator. When a mouse event is received, it is also
36   * possible to query the status of the modifier keys.
37   * </p>
38   * <p>
39   * <strong>Tip:</strong> When working with modifiers often sets are needed
40   * because other objects (e.g. accelerators) can be associated with an arbitrary
41   * number of modifiers. In such cases the {@code java.util.EnumSet} class can be
42   * used. It provides convenience methods for creating sets containing specific
43   * combinations of {@code Modifiers} constants. The following code fragment
44   * demonstrates how such a set can be obtained:
45   *
46   * <pre>
47   * EnumSet&lt;Modifiers&gt; mods = EnumSet.of(Modifiers.ALT, Modifiers.SHIFT);
48   * </pre>
49   *
50   * </p>
51   *
52   * @author Oliver Heger
53   * @version $Id: Modifiers.java 205 2012-01-29 18:29:57Z oheger $
54   */
55  public enum Modifiers
56  {
57      /** Constant representing the ALT modifier key. */
58      ALT,
59  
60      /** Constant representing the ALT GRAPH modifier key. */
61      ALT_GRAPH,
62  
63      /** Constant representing the CONTROL modifier key. */
64      CONTROL,
65  
66      /** Constant representing the META modifier key. */
67      META,
68  
69      /** Constant representing the SHIFT modifier key. */
70      SHIFT;
71  
72      /**
73       * Tries to find a {@code Modifiers} constant for the specified string. The
74       * string is compared with the constants defined in this class ignoring
75       * case. If a match is found, the corresponding {@code Modifiers} constant
76       * is returned. Otherwise, an exception is thrown.
77       *
78       * @param s the string to be parsed
79       * @return the corresponding {@code Modifiers} constant
80       * @throws IllegalArgumentException if no match can be found
81       */
82      public static Modifiers fromString(String s)
83      {
84          if (s == null)
85          {
86              throw new IllegalArgumentException(
87                      "String to be parsed must not be null!");
88          }
89  
90          return valueOf(s.toUpperCase(Locale.ENGLISH));
91      }
92  }
