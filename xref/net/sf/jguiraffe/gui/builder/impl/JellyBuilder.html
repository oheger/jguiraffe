56  /**
57   * <p>
58   * An implementation of the <code>Builder</code> interface that uses <a
59   * href="http://commons.apache.org/jelly">Commons Jelly</a> for processing build
60   * scripts and creating GUI components.
61   * </p>
62   * <p>
63   * This builder implementation will interpret Jelly build scripts that contain
64   * tags for creating bean definitions, form components, actions, and windows. To
65   * fulfill its purpose the builder object must be initialized with a couple of
66   * manager and factory interfaces, which will perform the real create operations
67   * for elements that comprise the generated GUI. Usually a client won't have to
68   * deal with these initialization stuff; instead this is handled by the
69   * dependency injection framework.
70   * </p>
71   * <p>
72   * Implementation note: This class is not thread-safe. The intended usage is
73   * that a new instance is created for each new builder operator.
74   * </p>
75   *
76   * @author Oliver Heger
77   * @version $Id: JellyBuilder.java 211 2012-07-10 19:49:13Z oheger $
78   */
79  public class JellyBuilder extends JellyBeanBuilder implements Builder

340      * Adds a new default converter to this builder. Default converters are
341      * stored internally. When a build operation is executed and a default
342      * {@link ConversionHelper} object has to be created, these default
343      * converters are automatically registered at the helper object. Normally
344      * client code does not have to call this method manually. This is done when

347      * default converters are automatically initialized).
348      *
349      * @param converter the converter to be added (must not be <b>null</b>)
350      * @param targetClass the target class of the conversion (must not be

352      * @throws IllegalArgumentException if a required parameter is missing
353      * @see ConversionHelper#registerConverter(Converter, Class)
354      */
355     public void addDefaultConverter(Converter converter, Class<?> targetClass)
356     {
357         checkRegisterConverterArgs(converter, targetClass);
358         defaultConverters.put(targetClass, converter);
359     }

446     /**
447      * Adds all converters in the specified map as default base class converters
448      * to this object using the specified {@code ClassLoaderProvider} to resolve
449      * class names. This method works like the method with the same name, but
450      * converter class are specified by name and resolved dynamically.
451      *
452      * @param converters the map with the converters to be added (must not be
453      *        <b>null</b>
454      * @param clp the {@code ClassLoaderProvider} (must not be <b>null</b>)
455      * @throws IllegalArgumentException if the map is <b>null</b> or contains
456      *         <b>null</b> entries or the {@code ClassLoaderProvider} is
457      *         <b>null</b>
458      * @since 1.2
459      */
460     public void addDefaultBaseClassConverters(
461             Map<String, ? extends Converter> converters, ClassLoaderProvider clp)
462     {
463         checkConvertersMap(converters);
464         if (clp == null)
465         {
466             throw new IllegalArgumentException(
467                     "ClassLoaderProvider must not be null!");
468         }
469 
470         for (Map.Entry<String, ? extends Converter> e : converters.entrySet())
471         {
472             addDefaultBaseClassConverter(e.getValue(),
473                     clp.loadClass(e.getKey(), null));
474         }
475     }

486     /**
487      * Registers default converters at the specified {@code ConversionHelper}
488      * instance. These are the converters added using the
489      * {@link #addDefaultConverter(Converter, Class)} or
490      * {@link #addDefaultBaseClassConverter(Converter, Class)} methods. If the
491      * {@link BuilderData} object passed to the {@code build()} methods does not
492      * contain an {@link InvocationHelper} instance, a new instance is
493      * automatically created, and the default converters are registered at the
494      * associated {@link ConversionHelper} instance. However, if a custom
495      * {@link InvocationHelper} is set, this registration is not done - the
496      * helper object is used as is. Using this method the default converters can
497      * be registered manually. Clients of the builder that need their own
498      * {@link InvocationHelper} can decide whether they call this method so that
499      * enhanced type conversion facilities are supported or not.
500      *
501      * @param conHlp the {@code ConversionHelper} to be initialized (must not be
502      *        <b>null</b>)
503      * @throws IllegalArgumentException if the {@code ConversionHelper} is
504      *         <b>null</b>
505      */
506     public void registerDefaultConverters(ConversionHelper conHlp)
507     {
508         if (conHlp == null)
509         {
510             throw new IllegalArgumentException(
511                     "ConversionHelper must not be null!");
512         }
513 
514         for (Map.Entry<Class<?>, Converter> e : defaultConverters.entrySet())
515         {
516             conHlp.registerConverter(e.getValue(), e.getKey());
517         }
518         for (Map.Entry<Class<?>, Converter> e : defaultBaseClassConverters
519                 .entrySet())
520         {
521             conHlp.registerBaseClassConverter(e.getValue(), e.getKey());
522         }
523     }

572     /**
573      * Frees all resources associated with the specified {@code BuilderData}
574      * object. The {@code BuilderData} object passed to this method must have
575      * been initialized by this builder. Especially the {@code BeanContext} and
576      * the {@code BeanBuilderResult} properties must have been set. Otherwise an
577      * exception is thrown.
578      *
579      * @param data the {@code BuilderData} object
580      * @throws IllegalArgumentException if the data object is <b>null</b> or is
581      *         not fully initialized
582      */
583     public void release(BuilderData data)
584     {
585         if (data == null)
586         {
587             throw new IllegalArgumentException(
588                     "BuilderData object must not be null!");
589         }
590         if (data.getBeanBuilderResult() == null)
591         {
592             throw new IllegalArgumentException(
593                     "Invalid BuilderData: No BeanBuilderResult found!");
594         }
595         if (data.getBuilderContext() == null)
596         {
597             throw new IllegalArgumentException(
598                     "Invalid BuilderData: No BeanContext found!");
599         }
600 
601         release(data.getBeanBuilderResult());
602         data.getBuilderContext().close();
603     }

605     /**
606      * The main method for executing a builder script. This method is called by
607      * all public <code>build()</code> methods. It executes the specified script
608      * with a newly initialized context and collects all results afterwards.
609      *
610      * @param script specifies the build script
611      * @param data the builder parameter object
612      * @param rootContainer the root container object
613      * @throws BuilderException if an error occurs
614      */
615     protected void performBuild(Locator script, BuilderData data,
616             Object rootContainer) throws BuilderException
617     {
618         checkBuilderData(data);
619         checkState();
620         data.setBuilder(this);
621 
622         JellyContext context = setUpExtendedJellyContext(data, rootContainer);
623         InvocationHelper invHlp = initInvocationHelper(data);
624         MutableBeanStore rootStore =
625                 initBuilderBeanContext(data, context, invHlp);
626 
627         fetchResults(
628                 context,
629                 data,
630                 executeScript(script, context, rootStore, data
631                         .getParentContext().getClassLoaderProvider(), invHlp),
632                 script);
633     }

635     /**
636      * Creates and initializes the Jelly context to be used for executing the
637      * builder script. This method will perform all necessary initialization
638      * steps so that the context can be directly used hereafter. Extended in
639      * this context means that the context will be used for the more complex
640      * build operations (forms, actions, and windows) and not only for bean
641      * definitions.
642      *
643      * @param data the builder parameters
644      * @param rootContainer the root container for the builder process
645      * @return the fully initialized context
646      */
647     protected JellyContext setUpExtendedJellyContext(BuilderData data,
648             Object rootContainer)
649     {
650         JellyContext result = createJellyContext();
651         initProperties(data, result);
652 
653         createComponentBuilderData(data, rootContainer).put(result);
654         createActionBuilderData(data).put(result);
655         createWindowBuilderData(data).put(result);
656         registerExtendedTagLibraries(result, data);
657 
658         return result;
659     }

661     /**
662      * Creates the component builder data object for the current builder
663      * process. This implementation first tries to obtain a {@code
664      * ComponentBuilderData} bean from the parent bean context. If such a bean
665      * cannot be found, a default bean is created.
666      *
667      * @param data the builder parameters
668      * @param rootContainer the root container
669      * @return the component builder data object
670      */
671     protected ComponentBuilderData createComponentBuilderData(BuilderData data,
672             Object rootContainer)
673     {
674         ComponentBuilderData result;
675         if (data.getParentContext().containsBean(BEAN_COMP_BUILDER_DATA))
676         {
677             result = (ComponentBuilderData) data.getParentContext().getBean(
678                     BEAN_COMP_BUILDER_DATA);
679         }
680         else
681         {
682             result = new ComponentBuilderData();
683         }
684 
685         if (data.getBuilderName() != null)
686         {
687             result.setBuilderName(data.getBuilderName());
688         }
689         else
690         {
691             result.setBuilderName(getName());
692         }
693         initComponentManagerOnBuilderData(result);
694         result.setFieldHandlerFactory(getFieldHandlerFactory());
695         result.setDefaultResourceGroup(data.getDefaultResourceGroup());
696         result.setRootContainer(rootContainer);
697         result.initializeForm(data.getTransformerContext(), data
698                 .getBindingStrategy());
699         result.getForm().setFormValidator(data.getFormValidator());
700 
701         return result;
702     }

704     /**
705      * Initializes the {@code InvocationHelper} to be used during the builder
706      * operation. This method is called before the builder script gets executed.
707      * If an {@code InvocationHelper} instance is provided in the specified
708      * {@code BuilderData} object, it is directly returned. Otherwise, a new
709      * instance is created with a specialized {@link ConversionHelper} whose
710      * parent is obtained from the parent bean store. That way all converters
711      * registered for the parent bean store are also available in the current
712      * builder script.
713      *
714      * @param data the builder parameters
715      * @return the {@code InvocationHelper} to use
716      */
717     protected InvocationHelper initInvocationHelper(BuilderData data)
718     {
719         InvocationHelper result = data.getInvocationHelper();
720 
721         if (result == null)
722         {
723             ConversionHelper parent =
724                     DefaultBeanStore.fetchConversionHelper(data

734      * Creates the root bean store for the current builder operation and

736      * the bean definitions defined by the builder script are stored (unless a
737      * different store is explicitly selected). A new {@code BeanContext}
738      * is created and initialized, so that it allows access to the following
739      * bean definitions (in this order):
740      * <ul>
741      * <li>the beans defined in the root store</li>
742      * <li>the beans stored in this {@code ComponentBuilderData}
743      * instance</li>
744      * <li>the beans defined in the parent context</li>
745      * <li>the content of the Jelly context</li>
746      * </ul>
747      *
748      * @param data the builder parameters
749      * @param context the Jelly context
750      * @param invHlp the current {@code InvocationHelper}
751      * @return the new root store for the builder operation
752      */
753     protected MutableBeanStore initBuilderBeanContext(BuilderData data,
754             JellyContext context, InvocationHelper invHlp)
755     {
756         SimpleBeanStoreImpl beanStore = new SimpleBeanStoreImpl();
757         beanStore.setParent(data.getParentContext().getDefaultBeanStore());
758 
759         // Connect all builder data objects with the bean store
760         ComponentBuilderData compData = ComponentBuilderData.get(context);
761         assert compData != null : "No component builder data found!";
762         compData.initBeanStore(beanStore);
763         ActionBuilder actData = ActionBuilder.get(context);
764         assert actData != null : "No action builder data found!";
765         actData.initBeanStore(beanStore);
766         WindowBuilderData wndData = WindowBuilderData.get(context);
767         assert wndData != null : "No widow builder data found!";
768         wndData.initBeanStore(beanStore);
769 
770         // Store the builder data itself
771         context.setVariable(ComponentBuilderData.KEY_BUILDER_DATA, data);
772 
773         DefaultBeanStore store = new DefaultBeanStore();
774         store.setParent(beanStore);
775         store.setConversionHelper(invHlp.getConversionHelper());
776         BeanContext builderContext =
777                 new BeanContextWrapper(data.getParentContext(),
778                         new CombinedBeanStore(store, new JellyContextBeanStore(
779                                 context, null)));
780         registerBeanCreationListeners(builderContext, data);
781         compData.setBeanContext(builderContext);
782         data.setBuilderContext(builderContext);
783         return store;
