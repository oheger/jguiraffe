25  /**
26   * <p>
27   * A specialized {@code Validator} implementation that uses regular expressions
28   * to validate user input.
29   * </p>
30   * <p>
31   * This is a very generic yet powerful {@code Validator} class. It can be
32   * configured with a regular expression string and some flags influencing the
33   * matching behavior. From this data a {@code java.util.regex.Pattern} object is
34   * created which is used to validate the data passed to the
35   * {@link #isValid(Object, TransformerContext)} method. {@code isValid()}
36   * transforms the passed in object to a string and applies the regular
37   * expression to it.
38   * </p>
39   * <p>
40   * {@code RegexValidator} provides a couple of properties for defining the
41   * {@code java.util.regex.Pattern} object that is the base for all validation
42   * operations. By using a {@code <properties>} section in the builder script
43   * these properties can be overridden for a local validation. Here the following
44   * properties are evaluated:
45   * </p>
46   * <p>
47   * <table * border="1">
48   * <tr>
49   * <th>Property</th>
50   * <th>Description</th>
51   * <th>Default</th>
52   * </tr>
53   * <tr>
54   * <td valign="top">regex</td>
55   * <td>The regular expression as a string. This string is used for creating the
56   * {@code Pattern} object. It must be a valid regular expression which conforms
57   * to the syntax expected by the {@code java.util.regex.Pattern} class.</td>
58   * <td valign="top">.*</td>
59   * </tr>
60   * <tr>
61   * <td valign="top">caseInsensitive</td>
62   * <td>A boolean flag which determines whether string comparisons should be case
63   * insensitive.</td>
64   * <td valign="top">false</td>
65   * </tr>
66   * <tr>
67   * <td valign="top">dotAll</td>
68   * <td>A boolean flag which indicates whether strings to be matched can contain
69   * line terminators. This flag corresponds to the {@code DOTALL} flag of the
70   * {@code Pattern} class.</td>
71   * <td valign="top">false</td>
72   * </tr>
73   * </table>
74   * </p>
75   * <p>
76   * If validation fails, the following possible error messages can be produced:
77   * <table border="1">
78   * <tr>
79   * <th>Message key</th>
80   * <th>Description</th>
81   * <th>Parameters</th>
82   * </tr>
83   * <tr>
84   * <td valign="top"><code>{@link ValidationMessageConstants#ERR_PATTERN}</code></td>
85   * <td valign="top">The input is not correct according the specified pattern.</td>
86   * <td valign="top">{0} = the regular expression</td>
87   * </tr>
88   * </table>
89   * </p>
90   * <p>
91   * This class is not thread-safe. However, a single instance can be shared
92   * between multiple input fields. In a builder script, a {@code RegexValidator}
93   * instance can be declared and initialized with default settings. This instance
94   * can then be associated with multiple input fields that all use the same
95   * validation pattern. If an input field requires slightly different settings,
96   * properties can be used to override the default settings. This can look as
97   * follows:
98   *
99   * <pre>
100  * <!-- The bean for the validator -->
101  * <di:bean beanClass="net.sf.jguiraffe.transform.RegexValidator"
102  *   name="regexValidator">
103  *   <di:setProperty property="regex" value="[0-9]*"/>
104  * </di:bean>
105  * ...
106  * <!-- Associate the validator with an input field.-->
107  * <f:textfield name="text1">
108  *   <f:validator phase="syntax" beanName="regexValidator"/>
109  * </f:textfield>
110  * <!-- Another text field with slightly different requirements. -->
111  * <f:textfield name="text2">
112  *   <f:validator phase="syntax" beanName="regexValidator">
113  *     <f:properties>
114  *       <f:property key="dotAll" value="true"/>
115  *     </f:properties>
116  *   </f:validator>
117  * </f:textfield>
118  * </pre>
119  *
120  * In this example there are two text fields which are both assigned the {@code
121  * RegexValidator} instance. The first text field uses the validator's default
122  * settings. The second field overrides a property.
123  * </p>
124  *
125  * @author Oliver Heger
126  * @version $Id: RegexValidator.java 205 2012-01-29 18:29:57Z oheger $
127  */
128 public class RegexValidator implements Validator

130     /** Constant for the regex property. */
131     protected static final String PROP_REGEX = "regex";
132 
133     /** Constant for the caseInsensitive property. */
134     protected static final String PROP_CASE_INSENSITIVE = "caseInsensitive";
135 
136     /** Constant for the multiLine property. */
137     protected static final String PROP_DOT_ALL = "dotAll";
138 
139     /** Constant for the default regular expression string. */
140     private static final String DEF_REGEX = ".*";

142     /** Stores the current regular expression pattern. */
143     private Pattern pattern;

231     /**
232      * Performs the validation as described in the class comment. The passed in
233      * object is transformed to a string and matched against the regular
234      * expression. <b>null</b> objects are accepted.
235      *
236      * @param o the object to be validated
237      * @param ctx the transformer context
238      * @return an object with the results of the validation
239      */
240     public ValidationResult isValid(Object o, TransformerContext ctx)
241     {
242         if (o == null)
243         {
244             return DefaultValidationResult.VALID;
245         }
246 
247         Pattern pat = getPattern(ctx);
248         Matcher m = pat.matcher(o.toString());
249         if (m.matches())
250         {
251             return DefaultValidationResult.VALID;
252         }
253 
254         ValidationMessage msg = ctx
255                 .getValidationMessageHandler()
256                 .getValidationMessage(ctx,
257                         ValidationMessageConstants.ERR_PATTERN, fetchRegex(ctx));
258         return new DefaultValidationResult.Builder().addValidationMessage(msg)
259                 .build();
260     }
261 
262     /**
263      * Returns a {@code Pattern} object to be used for the validation operation.
264      * This class maintains a {@code Pattern} object that corresponds to the
265      * properties set for an instance. If a property is changed, the pattern is
266      * reset. It is then created on demand if it is accessed. It is also
267      * possible to override properties in the {@code TransformerContext}. If
268      * this is the case, a new {@code Pattern} object is always created.
269      *
270      * @param ctx the {@code TransformerContext}
271      * @return the {@code Pattern} object to be used
272      */
273     protected Pattern getPattern(TransformerContext ctx)
274     {
275         if (overridesProperties(ctx))
276         {
277             return createPattern(ctx);
278         }
279         else
280         {
281             if (pattern == null)
282             {
283                 pattern = createPattern(ctx);
284             }
285             return pattern;
286         }
287     }
288 
289     /**
290      * Tests whether properties in the specified {@code TransformerContext}
291      * override properties of this object. This method is called by
292      * {@link #getPattern(TransformerContext)}. If it returns <b>true</b>, a new
293      * {@code Pattern} object is created.
294      *
295      * @param ctx the {@code TransformerContext}
296      * @return a flag whether default properties are overridden in the context
297      */
298     protected boolean overridesProperties(TransformerContext ctx)
299     {
300         Map<?, ?> props = ctx.properties();
301         return props.containsKey(PROP_REGEX)
302                 || props.containsKey(PROP_CASE_INSENSITIVE)
303                 || props.containsKey(PROP_DOT_ALL);
304     }
305 
306     /**
307      * Creates a new {@code Pattern} object. This method is called by
308      * {@link #getPattern(TransformerContext)} whenever a new pattern needs to
309      * be created. This implementation creates a new {@code Pattern} object
310      * based on the properties set in the {@code TransformerContext}. The
311      * properties of this object are used as defaults if a property is not set
312      * in the context. Thus the properties in the context override local
313      * properties if they are set.
314      *
315      * @param ctx the {@code TransformerContext}
316      * @return the newly created {@code Pattern} object
317      */
318     protected Pattern createPattern(TransformerContext ctx)
319     {
320         Configuration config = new MapConfiguration(ctx.properties());
321 
322         int flags = 0;
323         if (config.getBoolean(PROP_CASE_INSENSITIVE, isCaseInsensitive()))
324         {
325             flags |= Pattern.CASE_INSENSITIVE;
326         }
327         if (config.getBoolean(PROP_DOT_ALL, isDotAll()))
328         {
329             flags |= Pattern.DOTALL;
330         }
331 
332         return Pattern.compile(config.getString(PROP_REGEX, getRegex()), flags);
333     }
334 
335     /**
336      * Obtains the current regular expression string. This is either the value
337      * of the corresponding instance property or it is overridden in the
338      * context.
339      *
340      * @param ctx the {@code TransformerContext}
341      * @return the regular expression string
342      */
343     private String fetchRegex(TransformerContext ctx)
344     {
345         return ctx.properties().containsKey(PROP_REGEX) ? String.valueOf(ctx
346                 .properties().get(PROP_REGEX)) : getRegex();
347     }
